1.
וַיְהִי כַּאֲשֶׁר יָרַד יֵשׁוּעַ הָנוֹצְרִי מִן הָהָר וַחֲבוּרוֹת רַבּוֹת אַחֲרָיו

2.
וְהִנֵּה מְצוֹרָע אֶחָד בָּא וְיִשְׁתַּחֲוֶה לוֹ לֵאמֹר אֲדוֹנִי אִם תּוּכַל לִרְפַאת אֵלַי

3.
וְיֵט יֵשׁוּעַ אֶת-יָדוֹ וְיָגַע בּוֹ לֵאמֹר רוֹצֶה אֲנִי שֶׁתְּטַהֵר וּבְאוֹתָהּ שָׁעָה נְטַהֵר הַמְּצוֹרָע מִצִּרְעָתוֹ

4.
וַיֹּאמֶר אֵלָיו יֵשׁוּעַ הִשָּׁמֵר לְךָ פָּן תַּגִּיד לָאָדָם וּלֵךָ לְכֹּהֵן וְהַקְרֵב קֹרְבַּנֶךָ כַּאֲשֶׁר צִוָּה מֹשֶׁה בְּתּוֹרַתְכֶם

5.
וַיְהִי בְּבוֹאוֹ בַּכְּפָר-נַחוּם הֲמָרָתָהּ וְיִבֵּא אֵלָיו שַׂר הַמֵּאוֹת וְיִּתְחַנֵּן לוֹ

6.
לֵאמֹר אֲדוֹנִי בְּנִי שׁוֹכֵב בְּבֵיתִי מֵחוֹלִי הַכָּווּץ בַּלֹעֵז פַּרַלֵאטִיקוֹ וּמִתְחַלְחֵל מֵהַמַּחֲלָה

7.
וַיֹּאמֶר אֵלָיו יֵשׁוּעַ אֲנִי אֵלֵךְ וַאֲרַפְּאֵהוּ

8.
וַיַּעַן שַׂר הַמֵּאוֹת וַיֹּאמֶר לוֹ אֲדוֹנִי אֵינִּי רָאוּי שֶׁתָּבֵא תַּחַת גַּגִּי אֶלָּא שֶׁתִּגְזוֹר אוֹמֵר וִירַפֵּא

9.
וְאֲנִי אָדָם חוֹטֵא וְיֵשׁ לִי מֶמְשֶׁלֶת וְתַּחַת יְדֵי פְּרוּשִׁים וּפָרָשִׁים וְרוֹכְבִים וַאוֹמֵר אֲנִי לְאֶחָד מֵהֶם לְךָ וְיֵלֵךְ וּלְאַחֵר בָּא וְיִבֵּא וּלְעַבְדֵי עָשׂוּ זֶה וְיַעֲשׂוּ

10.
וְיִשְׁמַע יֵשׁוּעַ וְיִתְמַהּ וּלְבָאִים אַחֲרָיו אָמַר אָמֵן אֲנִי אוֹמֵר לָכֶם לֹא מָצָאתִי אֱמוּנָה גְּדוֹלָה בְּיִשְׁרָאֵל כְּמוֹ בָּזֶה

11.
אוֹמֵר אֲנִי לָכֶם שֶׁיָּבוֹאוּ רַבִּים מִמִּזְרָח וּמִמַּעֲרָב וְיָנוּחוּ עִם אַבְרָהָם וְעִם יִצְחָק וְעִם יַעֲקֹב בְּמַלְכוּת הַשָׁמַיִם

12.
וּבְנֵי הַמַּלְכוּת יוּשְׁלְכוּ בְּמַחְשַׁכֵּי גְהִינָם וְשָׁם יִהְיֶה בֶּכִי וְחִירוֹק שִׁנַּיִם

13.
וַיֹּאמֶר יֵשׁוּעַ לַשַּׂר הַמֵּאוֹת לְךָ וְכַּאֲשֶׁר הֶאֱמַנְתָּ יַעֲשֶׂה לְךָ וְנִרְפָּא הַנַּעַר בַּעֵת הַהִיא

14.
בַּעַת הַהיִא בָּא יֵשׁוּעַ לְבֵית פֵּיטְיירוֹס וְהִנֵּה חוֹתַנְתּוֹ שׁוֹכֶבֶת מַקְדֵּחַת

15.
וְיָגַע לְיָדָהּ וְיַעֲזְבֶהָ הַקַּדַּחַת וְתִקֹּם וּתְשָׁרְתָם

16.
וַיְהִי לְעֵת-הַעֶרֶב וְיַבְּאוּ אֵלָיו אֲחוּזֵי הַשֵּׁדִים וִירַפְּאֵם בְּמַאֲמָרוּ לְבַד וְכָל הַחוֹלִים רִיפֵא

17.
לְגַמוֹר מָה שֶׁנֶּאֱמַר עַל פִּי יְשַׁעְיָה הַנָּבִיא זִכְרוֹנוֹ לִבְרָכָה אָכֵן חֳלָיֵינוּ הוּא נָשָׂא וּמַכְאֹבֵינוּ סְבָלָם

18.
וַיְהִי אַחֲרֵי זֹאת וְיָרֵא יֵשׁוּעַ חֲבוּרוֹת רַבּוֹת סְבִיבוֹתָיו וִיצַוֶּה לָלֶכֶת עָבַר הַיָּם

19.
וְיִקְרַב חָכָם אֶחָד וַיֹּאמֶר לוֹ אֲדוֹנִי אֵלַךְ אַחֲרֶיךָ בַּכֹּל מָקוֹם שֶׁתֵלֵךְ

20.
וַיַּעַן אֵלָיו יֵשׁוּעַ לַשּׁוּעָלִים חוֹרִים וְלָעוּף קִנִּים וְלַבֵּן-הָאָדָם בֵּן-הַבְּתוּלָה אֵין מָקוֹם לְהַכְנִיס רֹאשׁוֹ

21.
וְאֶחָד מִתַּלְמִידָיו אָמַר לוֹ עָזוֹב אוֹתִי שֶׁאֵלַךְ וְאֶקָבּוֹר אֶת-אֲבִי

22.
וַיֹּאמֶר לוֹ יֵשׁוּעַ בָּא אַחֲרֵי וְעָזוּב הַמֵּתִים לִקְבּוֹר מֵתֵיהֶם

23.
וַיְהִי כַּאֲשֶׁר בָּא יֵשׁוּעַ בָּאֳנִיָּה וְיָבוֹאוּ תַּלְמִידָיו אַחֲרָיו

24.
וַיְהִי סָעַר גָּדוֹל בַּיָּם וְהַגַּלִּים נִכְנֵסִים בָּאֳנִיָּה וְהָאֳנִיָּה חָשְׁבָהּ לְהִשָּׁבֵר

25.
וְיִקְרְבוּ אֵלָיו תַּלְמִידָיו וִיקִיצוּהוּ לֵאמֹר אֲדוֹנֵינוּ הוֹשִׁיעָנוּ פָּן נֶאֱבַד

26.
וַיֹּאמֶר אֲלֵיהֶם לָמָּה תִּרְאוּ קְּטַנֵּי אֱמֻנָה וְיִקֹּם וְיִצוֹ לַיָּם וְלָרוּחוֹת שֶׁיָּנוּחוּ וּמִיָּד נָחוּ

27.
וְהָאֲנָשִׁים אֲשֶׁר שָׁם תָּמְהוּ וַיֹּאמְרוּ מִי הוּא זֶה שֶׁהָרוּחוֹת וְהַיָּם עוֹשִׂים רְצוֹנוֹ

28.
וַיְהִי כַּאֲשֶׁר עֵבֶר הַיָּם וַיַּעֲבוֹר עֵבֶר לַיָּם בְּמַלְכוּת גָדַרִי נִקְרָאִים בַּלֹעֵז גְיִיְנִיטְרָארוֹס וַיִּפְגְּעוּ בוֹ שְׁנַיִם אֲחוּזֵי שֵׁדִים יוֹצְאִים מִהַקְבָרִים מִשְׁתַגֵעַיִם עַד שֶׁלֹא יוּכָל אִישׁ לְעֲבוֹר בַּדֶּרֶךְ הַהִיא

29.
וַיִּצְעֲקוּ אֵלָיו הַשֵׁדִים לֵאמֹר מָה לָךְ עִמָּנוּ יֵשׁוּעַ בֵּן אֱלֹהִים בָאתָ קוֹדֵם הַזְמָן לִצַעֵרנוּ וְגַּם לְהִשְׁמִידנוּ ויֵשׁוּעַ אָמַר לָהֶם צֵאוּ מִשָׁם מִחֲנוּת רָעוֹת

30.
וְשָׁם קָרוֹב מִהֶם עֶדְרֵי חֲזִירִים רַבִּים רוֹעִים

31.
וַיִּפְגְּעוּ בוֹ הַשֵׁדִים יַעַן יֵשׁ לָנוּ לְצֵאת מִכָּאן תֵּן לָנוּ רְשׁוּת לַבָּא בַּאִלוּ הַחֲזִירִים

32.
וַיֹּאמֶר לָהֶם לֵכוּ וֵיֵּצֵאוּ הַשֵׁדִים מִהַאֲנָשִׁים וְיָּבִאוּ בַּחֲזִירִים וַיֵּלֵכוּ כָּל הַעֵדֶר בַּבֶּהָלָה וְנִשְׁמְטוּ בַּיָם וּמֵתוּ בַּמַיִם

33.
וְיִפְחָדוּ הַרוֹעִים וַיִּבְרְחוּ וַיַּגִּידוּ בַּעִיר הַכֹּל וְתֵּהוֹם כָּל הַעִיר

34.
וֵיֵּצֵאוּ לִקְרַאת יֵשׁוּעַ וַיִּרְאוּ אוֹתוֹ וַיָּחֵלּוּ פָּנָיו לְבִּלְתִּי עֲבוֹר בַּגְּבוּלָם
