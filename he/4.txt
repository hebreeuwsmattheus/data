1.
אָז לוֹקֵחַ יֵשׁוּעַ בְרוּחַ הַקָדוֹשׁ לְמִדְבָּר יְהוּדָה לְהַתִּנַסּוֹת מֵהַשָׂטָן

2.
וַיָּצוֹם אַרְבָּעִים יוֹם וְאַרְבָּעִים לָיְלָה אַחֲרֵי-כֵן נִרְעָב

3.
מִקָרַב הַמְנַסֶּה וַאמֶר לוֹ אִם בֵּן הָאֱלֹקִים אַתָּה אֱמוֹר שֶׁהָאֲבָנִים הַאֵלֶה יָשׁוּבוּ לֶחֶם

4.
וַיַּעַן יֵשׁוּעַ וַיּאׁמֶר כָתוּב כִּי לֹא עַל הַלֶּחֶם לְבַדּוֹ יִחְיֶה הָאָדָם כִּי עַל כָּל-מוֹצָא פִי יַהְוֶה יִחְיֶה הָאָדָם

5.
אָז לָקַח אוֹתוֹ הַשָׂטָן בְּעִיר הָקָדוֹשׁ וַיַּעַמִידֵהוּ עַל מָקוֹם הַיּוֹתֵר גָּבוֹהַ שְׁבִכֹּל הַמִקֹדֶשׁ

6.
וַיּאׁמֶר אֵלַיו אִם בֵּן הָאֱלֹקִים אַתָּה דָלַג לְמַטָּה שְׁהָרַי כָתוּב כִּי מַלְאָכָיו יְצַוֶּה-לָּךְ לִשְׁמָרְךָ בְּכָל-דְּרָכֶיךָ עַל-כַּפַּים יִשָּׂאוּנְךָ פֶּן-תִּגּוֹף בָּאֶבֶן רַגְלֶךָ

7.
וַיַּעַן אֵלַיו יֵשׁוּעַ שֵׁנִית וַיּאמֶר גַּם כָּתוּב לֹא תְנָסֶּה אֶת-יַהְוֶה אֱלֹקֶיךָ

8.
וַיִּשָּׂא אוֹתוֹ הַשָׂטָן עַל בְּהַר גָּבוֹהַּ מְאֹד וַיַּרְאֵהוּ כָּל מַמַלְכוֹת הָאָרֶץ וּכְבוֹדָם

9.
וַיּאׁמֶר לוֹ כָּל אֵלֶּה אֶתֵּן לְךָ אִם תִּפְרָע אֶלִי

10.
אָז עָנַה לוֹ יֵשׁוּעַ לֵךְ הַשָׂטָן בְּלֹעֵז שָׂאטָאנָאס שְׁכֵּן כָּתוּב אֶת-יַהְוֶה אֱלֹקֶיךָ אִתְפַּלֵּל וְאוֹתוֹ לְבַדּוֹ תַעֲבוֹד

11.
אָז עָזַב אוֹתוֹ הַשָׂטָן וְהִנֵּה מַלְאָכִים קָרְבוּ אֶליוֹ וְיִשְׂרֻפוֹהוּ

12.
וַיְהִי בְּיָמִים הַהֵם וַיִּשָׁמַע יֵשׁוּעַ כִּי נִמָסַר יוֹחָנָן בְּמִאָסַר וַיֵּלַך לְגָלִיל

13.
וְיַּעֲבוֹר אֶת-נָאזָרִית וְיִשְׁכּוֹן בִּכְפַר-נַחוּם וְאִיתָּהּ לָעַז מַארִיטְמָה בִּקָצֶה אֶרֶץ זְבוּלוּן וְנַפְתָּלִי

14.
לְקַיֵּים מָה שֶׁנֶאֱמַר עַל פִּי יְשַׁעְיָהוּ הַנָבִיא הַאוֹמֶר

15.
אַרְצָה זְבוּלוּן וְאַרְצָה נַפְתָּלִי דֶּרֶךְ הַיָּם עֵבֶר אֵת הַיַּרְדֵּן גָּלִיל הַגּוֹיִם

16.
הָעָם הַהוֹלְכִים בַּחוֹשָׁךְ רָאוּ אוֹר גָּדוֹל יוֹשְׁבֵי בְּאֶרֶץ צַלְמָוֶת אוֹר נָגַהּ עֲלֵיהֶם

17.
מִכְאָן וָאֵילַךְ הִתְחִיל יֵשׁוּעַ לְדָּרוֹשׁ וְלְדִּבֵּר חָזַרוּ בְּתְּשׁוּבָה כִּי שְׁמַלְכוּת הַשָׁמַיִם קְרוֹבָה

18.
וַיַּלִךְ יֵשׁוּעַ עַל שָׁפַת הַיָּם הַגָּלִיל וַיַּרְא שְׁנֵי אַחִים שִׁמְעוֹן שְׁיִקָרַא סִימוֹן וְנִקִרָא פְיֶיטְרוֹס וְאַנְדְירֶיאָה אָחִיו מַשְׁלִיכֵים מִכְמֶרוֹתִיהֵם בְּיָם שְׁהָיוּ דּיָּיגִים

19.
וַיּאׁמֶר לְהֵם לְכוּ אַחַרִי וְאעָשַׂה אֶתְכֶם מִדּיָּיגִים אֲנָשִׁים

20.
וְיֵעָזְבוּ מִכְמֶרוֹתִיהֵם באוֹתָה שָׁעַה וַיַּלְכוּ אַחֲרָיו

21.
וְיֵט מִשָּׁם וַיַּרְא שְׁנֵי אַחִים אֲחֵרִים יַעֲקֹב וְיוֹחָנָן אַחִים בְּנֵי זַבְדִּיאֵל בְּלֹעֵז זָבָאדָאוֹ וְזָאבָאדָה וְאָבִיהֵם בְּאֳנִיָּה מֵכִינִים מִכְמֶרוֹתִיהֵם וָיַּקרָא אוֹתָם

22.
וְיְמַהֲרוּ וַיַּנִּיחוּ מִכְמֶרוֹתִיהֵם וְאֶת-אֲבִיהֶם וַיַּלְכוּ אַחֲרָיו

23.
וַיָּסָב יֵשׁוּעַ אֶל אֶרֶץ הַגָּלִיל לְלָמַד קַהֵלוֹתֶם וְמְבָשַּׂר לְהֵם זֵבֶד טוּב לֹעֵז מַאוָונְגְיֶילְייוֹ מִמַלְכוּת הַשָׁמַיִם וְמִרָפַא כֹּל חוֹלִים וְכֹּל מַדְוֶה בְּעַם

24.
וַיַּלִךְ שְׁמוּעֹתוֹ בְּכֹּל אֶרֶץ סוּרִיָא וַיִּשְּׂאוּ אֵלַיו כֹּל הַחוֹלִים מִּכֹּל מִינֵּי חָלַאיִם מְשׁוֹנִים אֲחוּזִים הַשֵׁדִים וְהַנִבְעַתִים מִרוּחַ רָעָה וְהֵמתִּרְעַשִׁים וַיְרַפֵּא אוֹתָם

25.
וַיַּלְכוּ אַחֲרָיו רַבּוֹת מַקָפְּוֹלִי וְהַגָּלִיל מִיְרוּשָׁלֵם יְהוּדָה וּעֵבֶר הַיַרְדֵּן
