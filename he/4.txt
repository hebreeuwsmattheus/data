1.
אָז לוּקַּח יֵשׁוּעַ בָּרוּחַ הַקָּדוֹשׁ לְמִדְבַּר לְהִתְנָסוֹת מֵהַשָּׂטָן

2.
וְיָצוּם אַרְבָּעִים יוֹם וְאַרְבָּעִים לַיְלָה וְאַחֲרֵי־כֵן נִרְעָב

3.
וַיִּקְרַב הַמְּנַסֶּה וְאָמַר לוֹ אִם בֶּן־אֱלֹהִים אַתָּה אֱמוֹר שֶׁהָאֲבָנִים הָאֵלֶּה יָשׁוּבוּ לֶחֶם

4.
וַיַּעַן יֵשׁוּעַ וַיֹּאמֶר לוֹ כָּתוּב הוּא כִּי לֹא עַל־הַלֶּחֶם לְבַדּוֹ יִחְיֶה הָאָדָם כִּי עַל־כָּל־מוֹצָא פִי־יְהֹוָה

5.
אָז לָקַח אוֹתוֹ הַשָּׂטָן בָּעִיר הַקָּדוֹשׁ וְיַעֲמִידֵהוּ עַל מָקוֹם הַיּוֹתֵר גָּבוֹהַּ שֶׁבַּכֹל הַמִּקְדָשׁ

6.
וְאָמַר אֵלָיו אִם בֶּן־אֱלֹהִים אַתָּה דָּלַג לְמַטָּה וּכְתִיב כִּי מַלְאָכָיו יְצַוֶּה־לָךְ לִשְׁמָרְךָ בְּכָל־דְּרָכֶיךָ עַל־כַּפַּים יִשָּׂאוּנְךָ פֶּן־תִּגּוֹף בָּאֶבֶן רַגְלֶיךָ

7.
וַיַּעַן אֵלָיו יֵשׁוּעַ שֵׁנִית כֵּן כְּתִיב לֹא תְּנָסֶּה אֶת־יְהֹוָה אֱלֹהֶיךָ

8.
וְיִשָּׂא אוֹתוֹ הַשָּׂטָן בְּהַר גָּבוֹהַּ מְאֹד וְיַּרְאֵהוּ כֹּל מִמַּלְכוֹת הָאָרֶץ וּכְבוֹדָם

9.
וַיֹּאמֶר לוֹ כֹּל אֵלֶּה אֶתֵּן לְךָ אִם תִּכְרַע וְתִשְׁתַּחֲווּ אֵלַי

10.
אָז עָנָה לוֹ יֵשׁוּעַ לֵךְ הַשָּׂטָן בְּלֹעֵז שָׂאטָאנָאס שֶׁכֵּן כְּתִיב אֶת־יְהֹוָה הִתְפַּלֵּל וְאוֹתוֹ לְבַדּוֹ תַּעֲבוֹד

11.
אָז עָזַב אוֹתוֹ הַשָּׂטָן וְהִנֵּה מַלְאָכִים קָרְבוּ אֵלָיו וַיְשָׁרְתוּהוּ

12.
וַיְהִי בַּיָּמִים הָהֵם וְיִשְׁמַע יֵשׁוּעַ כִּי נִמְסַר יוֹחָנָן בְּמַאֲסָר וְיֵּלֵךְ אֶל הַגָלִיל

13.
וְיַעֲבוֹר אֶת־נָאזָרֶאל וְיִשְׁכּוֹן בַּכְּפַר־נַחוּם רַאִיתָה בַּלֹעֵז מָארִיטְמָה בִּקְצֵה אֶרֶץ זְבוּלוּן

14.
לְקִייֵּם מָה שֶׁאָמַר יְשַׁעְיָהוּ הַנָּבִיא

15.
אַרְצָה זְבוּלוּן וְאַרְצָה נַפְתָּלִי דֶּרֶךְ הַיָּם עֵבֶר אֶת־הַיַּרְדֵּן גָּלִיל הַגוֹיִם

16.
הָעַם הַהוֹלְכִים בְּחוֹשֶׁךְ רָאוּ אוֹר גָּדוֹל יוֹשְׁבֵי בָּאָרֶץ צַלְמָוֶת אוֹר נָגַהּ עֲלֵיהֶם

17.
מִכָּאן וְאֵילֵךְ הִתְחֵיל יֵשׁוּעַ לִדְרוֹשׁ וּלְדַבֵּר חָזְרוּ בַּתְּשׁוּבָה שֶׁמַּלְכוּת שָׁמַיִם קְרוֹבָה

18.
וְיֵלֵךְ יֵשׁוּעַ עַל שְׂפַת הַיָּם הַגָּלִיל וַיַּרְא שֵׁנִי אַחִים שִׁמְעוֹן שֶׁנִקְרָא שִׁימוֹן וְנִקְרָא פֶּיטְרוֹשׁ וְאַנְדְרֶיאָה אָחִיו מַשְׁלִיכִים מִכְמַרוֹתֵיהֶם בַּיָּם שֶׁהָיוּ דַּיִיגִים

19.
וַיֹּאמֶר לָהֵם לְכוּ אַחֲרִי וְאֶעֱשֶׂהָ אֶתְכֶם דַּיִיגִים מִאֲנָשִׁים

20.
וְיַעַזְבוּ מִכְמַרוֹתֵיהֶם בְּאוֹתָהּ שָׁעַה וַיֵלְכוּ אַחֲרָיו

21.
וְיֵט מִשָּׁם וַיַרְא שְׁנֵי אַחִים אֲחֵרִים יַעֲקֹב וְיוֹחָנָן אַחִים בְּנֵי זַבְדִיאֵל בְּלֹעֵז זָבָאדָאוֹ וְזָאבָאדָה וַאֲבִיהֶם בָּאֳנִיָּה מְכִינִים מִכְמַרוֹתֵיהֶם וַיִּקְרָא אוֹתָם

22.
וַיְמַהֲרוּ וַיַנִיחוּ מִכְמַרוֹתֵיהֶם וְאֶת-אֲבִיהֶם וַיֵלְכוּ אַחֲרָיו

23.
וַיָסָב יֵשׁוּעַ אֶל אֶרֶץ הַגָּלִיל לְלָמֵּד קְהִלוֹתָם וּמְבַשֵּׂר לָהֶם זֵבֶד טוּב בְּלֹעֵז מַאוָונְגְיֶיֶלְייוֹ מִמַּלְכוּת שָׁמַיִם וּמַרְפֵּא כָּל חוֹלִים וְכָל מַדְוֶה בָּעָם

24.
וַיֵּלֶךְ שְׁמוֹעתוֹ בְּכֹל אֶרֶץ סוּרִיָא וַיִשְּׂאוּ אֵלָיו כָּל הַחוֹלִים מִכֹּל מִינֵי חֳלָאִים מְשׁוּנִּים אֲחוּזִים הַשֵּׁדִים וְהַנִּבְעָתִים מֵרוּחַ רָעָה וְהַמְּתֵרֲעֲשִׁים וִירַפֵּא אוֹתָם

25.
וַיֵלְכוּ אַחֲרָיו רַבּוֹת מִקֲּפּוֹלִי וְהַגָּלִיל מִירוּשֲׁלֵם וִיהוּדָה וְעֵבֶר הַיַּרְדֵּן
