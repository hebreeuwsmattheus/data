1.
אַחֵר זֶה אָמַר יֵשׁוּעַ לְתַלְמִידָיו מַלְכוּת שָׁמַיִם דּוֹמֶה לָאָדָם יָחִיד אָדוֹן בֵּיתוֹ הַמַּשְׁכִּים בַּבָּקָר לַשְׂכּוֹר פּוֹעֲלִים לְכַרְמוֹ

2.
וְהִשְׂכִירָם בַּדִּינָר אֶחָד לַיּוֹם וְיִשְׁלְחֵם לְכַרְמוֹ

3.
וְיָצָא בַּשְּׁלִישִׁית הַיּוֹם וְיָרֵא אֲחֵרִים עוֹמְדִים בְּטֵלִים בַּשּׁוּק

4.
וַיֹּאמֶר לָהֶם לְכוּ גַּם אַתֶּם לְכַרְמִי וְהָרָאוּי אֶתֵּן לָכֶם וַיֵּלֵכוּ

5.
וְיָצָא עוֹד בַּצְּהָרִים וְגַם בַּשָּׁעָה הַתְּשִׁיעִית וַיַּעַשׂ כֵּן

6.
וּבְאַחַת עֲשָׂרָה שָׁעָה יָצָא גַּם כֵּן וְיִמְצָא אֲחֵרִים עוֹמְדִים וַיֹּאמֶר לָהֶם מַדּוּעַ אַתֶּם עוֹמְדִים בַּטֵלִים כָּל הַיּוֹם

7.
וְיַעֲנוּ לוֹ שֶׁלֹּא שָׂכַרְנוּ אָדָם וַיֹּאמֶר אֲלֵיהֶם לְכוּ גַּם אַתֶּם לְכַרְמִי

8.
וַיְהִי לָעֵת עֶרֶב וַיֹּאמֶר בְּעַל הַכֶּרֶם לְנִיצָּב עַל הַפּוֹעֲלִים קָרָא אוֹתָם וְאִתָּן לָהֶם שְׂכָרֶם וְיָחֵל בָּאַחֲרוֹנִים וְיָכֹל בָּרִאשׁוֹנִים

9.
וְהָאַחֲרוֹנִים קִבְּלוּ דִּינָר אֶחָד

10.
וְהָרִאשׁוֹנִים חָשָׁבוּ לָקַחַת יוֹתֵר וְהוּא נָתַן לַכֹּל אֶחָד דִּינָר

11.
וִילוֹנוּ הָרִאשׁוֹנִים עַל בְּעַל הַכֶּרֶם

12.
לֵאמֹר אֵלּוּ הָאַחֲרוֹנִים עָמְלוּ שָׁעָה אַחַת וְהִשְׁוִּית אוֹתָם עִמָּנוּ שֶׁעִמְלְנוּ כָּל הַיּוֹם וְהַחוֹרֶב

13.
וַיַּעַן לָאֶחָד מֵהֶם וַיֹּאמֶר לוֹ אֲהוּבִי אֵינִי עוֹשֶׂה לְךָ עוֹל הַלֹּא בַּדִּינָר אֶחָד שְׂכַרְתִּיךָ

14.
קָחֵנוּ וּלֵךְ אִם אֲנִי רוֹצֶה לָתֵת לְזֶה הָאַחֲרוֹן כְּמוֹתְךָ

15.
הַלֹּא אֶעֱשֶׂה כִּרְצוֹנִי הָרָע בְּעֵינֶיךָ כַּאֲשֶׁר אֲנִי טוֹב

16.
כֵּן יִהְיֶה אַחֲרוֹנִים רִאשׁוֹנִים וְהָרִאשׁוֹנִים אַחֲרוֹנִים רַבִּים הֵם הַקְּרוּאִים וּמְעַטִּים הַנִּבְחָרִים

17.
וְיִקְרַב יֵשׁוּעַ אֶל יְרוּשָׁלַם וְיִקַּח אֶת־שָׁנִים עֶשֶׂר תַּלְמִידָיו בְּסֵתֶר וַיֹּאמֶר אֲלֵיהֶם

18.
הִנֵּה אֲנַחְנוּ עוֹלִים לִירוּשָׁלַםִ וּבֶּן־הָאָדָם יִמְסֹר לִגְדוֹלֵי הַחֲכָמִים וְהַכֹּהֲנִים וַיִּחֵייבוּהוּ לָמוּת

19.
וְגַם יִמְסְרוּ אוֹתוֹ לְגּוֹיִם לְהַכּוֹתוֹ וּלְהַשְׁבִּיתוֹ וּבַיּוֹם הַשְּׁלִישִׁי יָקוּם

20.
אָז בָּא אֵשֶׁת זַבְדִיאֵל עִם בָּנֶיהָ מִשְׁתַּחֲוֶה וּמְבַקֶּשֶׁת בַּקָּשָׁה מִמֶּנּוּ

21.
וַיֹּאמֶר אֵלֶיהָ מָה תִּרְצִי וְתֹאמַר שֶׁתְּצַוֶּה לָשֶׁבֶת שְׁנֵי בְּנֵי אֵלֶּה הָאֶחָד לִימִינְךָ וְהַשֵּׁנִי לִשְׂמֹאלְךָ בְּמַלְכוּתְךָ

22.
וַיַּעַן לָהֶם יֵשׁוּעַ לֹא תִּדְעוֹן מָה תְבַקְשׁוֹן הֲתּוּכָלוּן לִסְבּוֹל הַיִסּוּרִין וְהֵמִיתָה שֶׁאֲנִי עָתִיד לִסְבּוֹל וְיֹּאמְרוּ נוּכַל

23.
וַיֹּאמֶר לָהֶם שָׁתוּ כּוֹסִי וְהוֹשִׁיבוּ לִשְׂמֹאלִי אוֹ לִימִינִי אֵין לִי לְתִתָּה לָכֶם כִּי אִם לַאֲשֶׁר הוּא נָכוֹן לִפְנֵי אֲבִי

24.
וְיִשְׁמְעוּ הָעֲשָׂרָה וְיִחַר בְּעֵינָם מֵעֳנִין שְׁנֵי הַאַחִים

25.
וִיקָרְבֵם יֵשׁוּעַ אֵלָיו וַיֹּאמֶר לָהֶם דְּעוּ שֶׁנְּשִׂיאֵי הַגּוֹיִם רוֹדִים בָּהֶם וְגִדּוּלֵיהֶם מְבַקְּשִׁים לְנַפְשָׁם

26.
לֹא כֵּן יִהְיֶה בֵּינֵיכֶם שֶׁהָרוֹצֶה לִהְיוֹת גָּדוֹל בֵּינֵיכֶם יְשָׁרֶת אֶתְכֶם

27.
וַאֲשֶׁר יִרְצֶה בֵּינֵיכֶם לִהְיוֹת רִאשׁוֹן יִהְיֶה לָכֶם עֶבֶד

28.
כַּאֲשֶׁר בֶּן־הָאָדָם לֹא בָּא שֶׁיְּשָׁרְתוּהוּ כִּי אִם לְשָׁרֵת וְלָתֵת נָפְשׁוֹ כּוֹפֵר לָרַבִּים

29.
הֵמָּה נִכְנָסִים בִּיְרִיחוֹ וְתֵלֵךְ אַחֲרָיו חֲבוּרָה אַחַת

30.
וְהִנֵּה שְׁנֵי עִוְּרִים יוֹצְאִים אֵצֶל הַדֶּרֶךְ וְיִשְׁמְעוּ קוֹל הֶהָמוֹן וְיִשְׁאָלוּ מָה זֶה וְיֹּאמְרוּ לָהֶם הַנָּבִיא יֵשׁוּעַ מִנָאצְרַת בָּא אָז צָעֲקוּ לֵאמֹר בֶּן־דָּוִד חָנְנוּ

31.
וְהַחֲבוּרָה גָּעַרוּ בָּהֶם לֵאמֹר הַסּוּ וְהֵם הָיוּ צוֹעֲקִים וְאוֹמְרִים הָאָדוֹן בֶּן־דָּוִד חָנְנוּ

32.
וְיַעֲמוֹד יֵשׁוּעַ וְיִקְרְאֵם וַיֹּאמֶר מָה תִּרְצוּ לְהֵעָשׂוֹת לָכֶם

33.
וְיֹּאמְרוּ הָאָדוֹן שֶׁתִּפְקַחְנָה עֵינֵינוּ

34.
וְיַחְמוֹל עֲלֵיהֶם יֵשׁוּעַ וְיָגַע בְּעֵינֵיהֶם וַיֹּאמֶר לָהֶם אֱמוּנָתְכֶם רִיפֵּא אֶתְכֶם וּמִיַּד רָאוֹ וְהוֹדוּ לָאֵל וְהָלְכוּ אַחֲרָיו וְכָל הָעָם הוֹדוּ לְאֵל עַל זֶה
