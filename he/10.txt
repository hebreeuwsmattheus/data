1.
אָז קָרָא יֵשׁוּעַ לְשָׁנִים עָשָׂר תַּלְמִידָיו וַיִּתֶּן לָהֶם יְכוֹלֶת עַל כֹּל רוּחַ טוּמְאָה לְהוֹצִיא מֵהָאָדָם וְלִרְפֹּאת כֹּל חוֹלִי וְכֹל נֶגַע

2.
וְאֵלֶּה שֵׁמוֹת שְׁנַיִם עָשָׂר הַשְּׁלוּחִים נִקְרְאוּ אַפּוֹסְטוֹלוֹס שִׁימוֹן נִקְרָא פֶּיטְרוֹשׁ וְאַנְדְרֶיאָה אָחִיו

3.
פִילִיפּוֹס בַּארְטוֹלוֹמֵיאוֹס יַעֲקֹב נִקְרָא גְאִימִי וְיוֹחָנָן אָחִיו בְּנֵי זַבְדִּיאֵל טוֹמַאס וּמַתִתְיָה הוּא מַאטֵיאוֹ מִלְוֶה בָּרִיבִּית בַּפִּרְסוּם וְיַעֲקֹב אַלוּפֶיאֶי וְטַדֵיאוֹשׂ

4.
שִׁמְעוֹן כְּנַעֲנִי בַּלֹעֵז שִׁימוֹן קַאנָאנֵאיוֹס וְיְהוּדָה אַסְקָארְיוֹטָה אֲשֶׁר אַחֵר זֶה מָסְרֵהוֹ

5.
אֵלֶּה שְׁנַיִם עָשָׂר שָׁלַח יֵשׁוּעַ וַיְצַוֹ אֲלֵיהֶם לֵאמֹר בְּאַרְצוֹת הַגּוֹיִם אַל תֵּלְכוּ וּבְעָרֵי הַשֹׁמְרוֹנִים אַל תָּבוֹאוּ

6.
לֵכוּ לַצֹּאן אֲשֶׁר נִדְּחוּ מִבֵּית יִשְׂרָאֵל

7.
וּבַשְּׂרוּ לָהֶם שֶׁתִּתְקַרֵב מַלְכוּת שָׁמַיִם

8.
רַפְאוּ הַחוֹלִים וְהַחֲיוּ הַמֵּתִים וְטִהֲרוּ הַמְּצוֹרָעִים וְהוֹצִיאוּ הַשֵּׁדִים מִבְּנֵי אָדָם וְאַל תְּקַבְּלוּ שָׂכַר חִנָּם קִבַּלְתֶּם וּבְחִנָּם תִּתְּנוּ

9.
אַל תִּצְבֹּרוּ כֶּסֶף וְזָהָב וְלֹא מָמוֹן בְּכִיסְכֶם

10.
וְלֹא חֲלִיפוֹת שְׂמָלוֹת וְלֹא מִנְעָלִים וְלֹא מַקֵּל בְּיַדְכֶם רָאוּי הַפּוֹעֵל לְקַבֵּל דַּי אֲכִילָתוֹ

11.
וּבְכֹל עִיר וּבְכֹל מִגְדָּל אֲשֶׁר תָּבוֹאוּ שָׁאֲלוּ מִי הָאִישׁ הַטּוֹב שֶׁבְּתוֹכָם וְשָׁם תְּנוּחוּ עַד שֶׁתֵּצְאוּ

12.
וּבְבוֹאֲכֶם אֶל הַבַּיִת תְּנוּ לָהֶם שָׁלוֹם לֵאמֹר שָׁלוֹם בְּזֹאת הַבַּיִת שָׁלוֹם לַכָל הַיּוֹשְׁבִים בְּתוֹכָהּ

13.
וְאִם תִּהְיֶה הַבַּיִת הַהִיא רָאוּיָה תָּבֹא שְׁלוֹמְכֶם עָלֶיהָ וְאִם לֹא תִּהְיֶה רְאוּיָה תָּשׁוּב שְׁלוֹמְכֶם לָכֶם

14.
וַאֲשֶׁר לֹא יְקַבֵּל אֶתְכֶם וַאֲשֶׁר לֹא יִשְׁמַע אֲלֵיכֶם תֵּצְאוּ מִן הַבַּיִת הַהִיא וְהֶעְתֵּקָם רַגְלֵיכֶם מִן הַעָפָר

15.
אָמֵן אֲנִי דּוֹבֵר אֲלֵיכֶם יוֹתֵר טוֹב יִהְיֶה אֶל סְדוֹם וְאֶל עֲמוֹרָה בַּיּוֹם הַהוּא מִן הָעִיר הַהִיא

16.
הִנְנִי שׁוֹלֵּחַ אֶתְכֶם כְּצֹאן בֵּין הַזְּאֵבִים תִּהְיוּ עֲרוּמִים כְּנְחָשִׁים וַעֲנָוִים כְּיוֹנִים

17.
הֻזְהֲרוּ בִּבְּנֵי אָדָם לֹא יִמְסְרוּ אֶתְכֶם בְקְהִלוֹתֶם וּבְבָתֵּי כְנֵסִיוֹתַם

18.
וּלְפָחוֹת וְלַמְּלָכִים תּוּכְלוּן בַּעֲדִי לְהָעֵיד לָהֶם וְלַגּוֹיִם

19.
כַּאֲשֶׁר יִתְפְּשׂוּ אֶתְכֶם אַל תַּחְשְׁבוּ מָה שֶׁתֹּאמְרוּ שֶׁבִּשַָּׁעָה שֶׁתִּצְטָרְכוּ יִבֵּא לָכֶם מַעֲנֶה

20.
אֵינְכֶם הַמִּדְבָּרִים כִּי אִם רוּחַ קָדְשׁוּ שֶׁל אֲבִי הוּא הַדּוֹבֵר בָּכֶם

21.
יִמְסוֹר הָאָח אֶת־אֶחָיו לָמוֹת וְהָאָב לִבְנוֹ וְיָקוּמוּ הַבָּנִים עַל אָבוֹתָם וְיוֹבִילוּ אוֹתָם עַד לָמוֹת

22.
וְתִהְיוּ לַלַּעַג וְזַעֲוָה לַכָל הָעַמִּים עַל שְׁמִי אָכֵן מִי שֶׁיִּסְבּוֹל עַד עֵת קָץ יִוָּשַׁע

23.
עוֹד אָמַר יֵשׁוּעַ לְתַלְמִידָיו כַּאֲשֶׁר יִרְדְּפוּ אֶתְכֶם בָּעִיר הַזֹּאת בָּרְחוּ לְאַחֶרֶת אָמֵן אֲנִי אוֹמֵר לָכֶם לֹא תַּשְׁלִימוּ לִסְבוֹב עָרֵי יִשְׂרָאֵל עַד יִבֵּא בֶּן־הָאָדָם

24.
אֵין הַתַּלְמִיד גָּדוֹל מְרַבּוֹ וְלֹא הַעֶבֶד גָּדוֹל מֵאֲדוֹנָיו

25.
דַּי לַתַּלְמִיד לִהְיוֹת כְּרַבּוֹ וְלָעֶבֶד כַּאֲדוֹנָיו אִם לַבַּעַל הַבַּיִת יִקְרְאוּ בַּעַל זְבוּב כֹּל שְׁכֵּן לִבְנֵי בֵּיתוֹ

26.
אַל תִּרְאוּ מֵהֶם שֶׁאֵין דָּבָר שֶׁלֹא יִרְאָה וְלֹא נֶעֱלַם שֶׁלֹא יִוָּדַע

27.
אֲשֶׁר אֲנִי אוֹמֶר לָכֶם בַּחֹשֶׁךְ אָמְרוּ אוֹתוֹ בָּאוֹר וַאֲשֶׁר תִּשְׁמְעוּ לָאוֹזֶן סָפְרוּ אוֹתוֹ בַּשַּׁעַר

28.
וְאַל תִּפְחֲדוּ מֵהוֹרְגֵי הֲגוּיוֹת שֶׁאֵין בְּיָדָם לַהֲרוֹג הַנְּפָשׁוֹת רַק פָּחֲדוּ לַאֲשֶׁר יְכוֹלֶת בְּיָדוֹ לְאַבֵּד הַנֶּפֶשׁ וְהַגּוּף בַגֵהִינָם

29.
הַלֹּא שְׁנֵי צִפּוֹרִים יִמָּכְרוּ בַּפְּרוּטָה אַחַת וְלֹא תִּפּוֹל אַחַת מֵהֶם עַל הָאָרֶץ כִּי אִם בָּרָצוֹן אֲבִיכֶם שֶׁבַּשָּׁמַיִם

30.
הַלֹּא שְׂעָרוֹת רָאשֵׁיכֶם כֻּלָּם סְפוֹרִים

31.
אַל תִּרְאוּ שֶׁטּוֹבִים מִצְפֹּרִים אַתֶּם

32.
הַמְּשַׁבֵּחַ אוֹתִי לִפְנֵי בְּנֵי אָדָם אַשְׁבַּחֵנוֹ לִפְנֵי אֲבִי שֶׁבַּשָּׁמַיִם

33.
וַאֲשֶׁר יְכַחֶשׁ בִּי לִפְנֵי בְּנֵי אָדָם אֵכַחֶשׁ בּוֹ לִפְנֵי אָבִי שֶׁבַּשָּׁמַיִם

34.
בְּאוֹתָהּ שָׁעָה אָמַר יֵשׁוּעַ לְתַלְמִידָיו אַל תַּחְשְׁבוּ שֶׁבָּאתִי לְשִׂים שָׁלוֹם בָּאָרֶץ אֶלָּא חָרֵב

35.
בָּאתִי לְהַפְרִיד הָאָדָם הַבֵּן מֵאָבִיו וְהַבַּת מֵאָמָה

36.
וְהָאוֹיְבִים לִהְיוֹת אוֹהֵבִים

37.
הָאוֹהֵב אָבִיו וְאִמּוֹ יוֹתֵר מִמֶּנִּי אֵינִי רָאוּי לוֹ

38.

39.
הָאוֹהֵב אֶת־נַפְשִׁוֹ יֹאבַדנָה הָאוֹבֵד אוֹתִי בִּשְׁבִילִי יִמְצָאֶנָּה

40.
הַמְּקַבֵּל אֶתְכֶם יְקַבֵּל אוֹתִי וְהַמְּקַבֵּל אוֹתִי יְקַבֵּל אֵת אֲשֶׁר שֶׁלַּחֲנִי

41.
הַמְּקַבֵּל נָבִיא לְשֵׁם נָבִיא יְקַבֵּל שָׂכַר הַנָּבִיא וְהַמְּקַבֵּל צַדִּיק לְשֵׁם צַדִּיק יְקַבֵּל שָׂכַר הַצַּדִּיק

42.
וְהֲנוֹתֵן כְּלִי אֶחָד שֹׁל מַיִם קָרִים לְאַחַד מִתַּלְמִידִּי הַקְּטַנִּים לְשֵׁם תַּלְמִידִי אָמֵן אֲנִי אוֹמַר לָכֶם שֶׁלֹא יְאַבֵּד שְׂכָרוֹ
