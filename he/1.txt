1.
אֵלֶּה הַתּוֹלְדוֹת יֵשׁוּעַ הֵמָּשִׁיחַ בֶּן-דָּוִד בֶּן-אַבְרָהָם

2.
אַבְרָהָם הוֹלִיד אֶת-יִצְחָק וַיִצְחָק הוֹלִיד אֶת-יַעֲקֹב וְיַעֲקֹב הוֹלִיד אֶת-יְהוּדָה וְאֶחָיו

3.
וִיהוּדָה הוֹלִיד אֶת-פֶּרֶץ וְזֶרַח מִתָּמָר וּפֶּרֶץ הוֹלִיד אֶת-חֶצְרוֹן וְחֶצְרוֹן הוֹלִיד אֶת-רָם

4.
וְרָם הוֹלִיד אֶת-עַמִּינָדָב וְעַמִּינָדָב הוֹלִיד אֶת-נַחְשׁוֹן וְנַחְשׁוֹן הוֹלִיד אֶת-שַׂלְמוֹן

5.
וְשַׂלְמוֹן הוֹלִיד אֶת-בּוֹעַז מִרָחָב הַזּוֹנָה וְבּוֹעַז הוֹלִיד אֶת-עוֹבֵד מִרוּת וְעוֹבֵד הוֹלִיד אֶת-יִשַׁי

6.
וְיִשַׁי הוֹלִיד אֶת-דָּוִד הַמֶּלֶךְ וְדָּוִד הַמֶּלֶךְ הוֹלִיד אֶת-שְׁלֹמֹה מִאֵשֶׁת אוּרִיָּה

7.
וְשְׁלֹמֹה הוֹליִד אֶת-רְחַבְעָם וּרְחַבְעָם הוֹלִיד אֶת-אֲבִיָּה וְאֲבִיָּה הוֹלִיד אֶת-אָסָא

8.
וְאָסָא הוֹלִיד אֶת-יְהוֹשָׁפָט וִיְהוֹשָׁפָט הוֹלִיד אֶת-יוֹרָם וִיוֹרָם הוֹלִיד אֶת-עוּזּיָּה

9.
וְעוּזּיָּה הוֹלִיד אֶת-יוֹתָם וְיוֹתָם הוֹלִיד אֶת-אָחָז וְאָחָז הוֹלִיד אֶת-חִזְקִיָּה

10.
וְחִזְקִיָּה הוֹלִיד אֶת-מְנַשֶׁה וּמְנַשֶׁה הוֹלִיד אֶת-אָמוֹן וְאָמוֹן הוֹלִיד אֶת-יֹאשִׁיָּה

11.
וְיֹאשִׁיָּה הוֹלִיד אֶת-יְכָנְיָה וְאֶחָיו בַּגָּלוּת בָּבֶל

12.
וְאַחַר גָּלוּת בָּבֶל יְכָנְיָה הוֹלִיד אֶת-שְׁאַלְתִּיאֵל וּשְׁאַלְתִּיאֵל הוֹלִיד אֶת-זְרוּבָּבֶל

13.
וּזְרוּבָּבֶל הוֹלִיד אֶת-אֲבִיהוּד וְאֲבִיהוּד הוֹלִיד אֶת-אֶלְיָקִים וְאֶלְיָקִים הוֹלִיד אֶת-עַזּוּר

14.
וְעַזּוּר הוֹלִיד אֶת-צָדוֹק וְצָדוֹק הוֹלִיד אֶת-אָקִים וְאָקִים הוֹלִיד אֶת-אֱלִיהוּד

15.
וְאֱלִיהוּד הוֹלִיד אֶת-אֶלְעָזָר וְאֶלְעָזָר הוֹלִיד אֶת-מַתָּן וּמַתָּן הוֹלִיד אֶת-יַעֲקֹב

16.
וְיַעֲקֹב הוֹלִיד אֶת-יוֹסֵף 
הוּא יוֹסֵף אֲבִי מִרְיָם אִם יֵשׁוּעַ הַנִּקְרָא מָשִׁיחַ וּבְלֹעֵז קְרִיסְטוֹס

17.
וְכֹל תּוֹלְדוֹת מִאַבְרָהָם עַד דָּוִד תּוֹלְדוֹת י"ד וּמִדָּוִד עַד גָּלוּת בָּבֶל תּוֹלְדוֹת י"ד וּמִגָּלוּת בָּבֶל עַד יֵשׁוּעַ תּוֹלְדוֹת י"ד

18.
וְלֵידַת מִיֵּשּׁוּעַ הָיָה בְּזֶה הָאוֹפַן וַיְהִי כַּאֲשֶׁר הָיְתָה אִמּוֹ אֲרוּסָה לְיוֹסֵף קוֹדְם שֶׁיָּדַע אוֹתָהּ נִמְצֵאת מֵעוֹבֶרֶת מֵרוּחַ הַקֹּדֶשׁ

19.
וְיוֹסֵף אִישָׁהּ צַדִּיק הָיָה וְלֹא רָצָה לֵישֵׁב עִמָּהּ וְלֹא לְגַּלוֹתָהּ לְהָבִיאָהּ לַבּוּשָׁה וְלֹא לְאוֹסְרָהּ לָמוּת אֲבָל הָיָה רוֹצֶה לְכַסּוֹת עָלֶיהָ

20.
וּבְחָשָׁבוֹ בְּזֶה הַדָּבָר בְּלִבּוֹ וְהִנֵה מַלְאַךְ נִרְאָה אֵלָיו בַּחֲלוֹם לֵאמוֹר יוֹסֵף בֶּן-דָּוִד אַל תִּירָא לִקַּחַת אִשְׁתְּךָ מִרְיָם שְׁמָּרוּחַ הַקָּדוֹשׁ הִיא מֵעוֹבֶרֶת

21.
וְתֵלֵד בֵּן וְתִּקְרָא שְׁמוֹ יֵשּׁוּעַ כִּי הוּא יוֹשִׁיעַ אֶת-עַמִּי מֵעֲוֹנוֹתָם

22.
כֹּל זֶה לְגַמוֹר מָה שֶׁנִּכְתַּב מֵאֵת הַנָּבִיא עַל פִּי יַהְוֶה

23.
הִנֵּה הָעַלְמָה הָרָה וְתֵלֵד בֵּן וְקָרָאת שְׁמוֹ עִמָּנוּאֵל
שֶׁרֹצֵה לוֹמַר עִמָּנוּ אֱלֹהִים

24.
וַיִּקַץ יוֹסֵף מִשְׁנַתוֹ וַיַּעַשׂ כְּכֹל אֲשֶׁר צִּוָּה אוֹתוֹ מַלְאַךְ יַהְוֶה וְיִּקַּח אֶת-אִשׁתּוֹ

25.
וְלֹא יָדַע אוֹתָהּ עַד שֶׁיָלְדָה בְּנָהּ הַבְּכוֹר וַיִּקְרָא אֶת-שְׁמוֹ יֵשׁוּעַ
