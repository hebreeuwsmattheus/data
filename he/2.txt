1.
וַיְהִי כַּאֲשֶׁר שֶׁנוֹלַּד יֵשׁוּעַ בַּבֶּיִת-לֶחָם בַּעִיר יְהוּדָה בִּימֵי הוֹרוֹדוֹס הַמֶּלֶךְ 
וְהִנֵה חוֹזִים בַּכּוֹכבִים בָאִים מִמִּזְרַח לְיְרוּשָׁלַם

2.
לֵאמֹר :אַיֵה מֶלֶךְ עַל הַיְהוּדִים הַנוֹלָד כִּי רָאִינוּ סָבְבוּ כוֹכָבוֹ בַּמִזְרַח וְבְּמַתָּנוֹת חְשׁוֹבוֹת בָּאוּ לְהִשְׁתַּחֲווֹת לוֹ

3.
וְשָׁמַע הוֹרוֹדוֹס הַמֶּלֶךְ וְיּבָּהֵל נתִּמָּלֵא חֵמָּה הוּא וְכָל יוֹשְׁבֵי יְרוּשָׁלַם עַמוֹ

4.
וְיִקִבּוּץ אֶת כָּל הָכֹּהֲנִים וְאֵת סוֹפְרֵי הָעָם וְיִבִּקֵשׁ מֶהֶם אִם הָיוֹ יוִדֻעִים בַּאֵיזָה מֶקוֹם נוֹלָד הַמָּשִׁיחַ

5.
וְיַעֲנוּ אֵלָיו בְּבַּיִת-לֶחֶם יְהוּדָה כְכְּתִיב עַל פִּי הַנָּבִיא

6.
וְאַתָּה בַּיִת-לֶחֶם יְהוּדָה אֶפְרָתָה אֶרֶץ יְהוּדָה הֵן אַתָּה צָעִיר בְּאַלְפֵי יְהוּדָה מִמְּךָ לִי יֵצֵא לִהְיוֹת מוֹשֵׁל בְּיִשְׂרָאֵל

7.
אָז קָרָא הַמֶּלֶךְ הוֹרוֹדוֹס לְקוֹסְמִים בְּסֵתֶר וְיְשָׁאַל מָהֵם הֵיטֵב זְמָן רְאִית הַכּוֹכָב לָהֶם

8.
וְיִשַׁלְּחָם לְבַּיִת-לֶחֶם וַיּאָמַר אֵלַיהֵם לְכוּ וְשִׁאֵלוּ הֵיטֵב בַּעַד הַיֶלֶד וְבַּמוֹצַאֲכֶם אוֹתוֹ הִגִּידוּ לִּי וְגִּם אֲנִי אַבֹא אֲנִיוּ לְהִשְׁתַּחֲווֹת

9.
וַיִּשָּׁמִעוּ אֶל הַמֶּלֶךְ וְיֵלֵכוּ וָהִנֵּה הַכּוֹכָב אֲשֶׁר רָאוּ בַּמֵזְרַח הוֹלֵךְ לִפְנֵיהָם עַד בוֹאֶם אֶל הַמָקוֹם וְכְאַשֵׁר בַּאוּ בַּיִת-לֶחֶם עָמַד מְמַּעל נֶגֶד הַמָקוֹם אֲשֶׁר שָׁם הַיֶלֶד

10.
וַיְהִי כַּאֲשֶׁר רָאוּ אֵת הַכּוֹכָב שָׂמְחוּ שִׂמְחָה גְּדוּלָּה עַד מְאֹד

11.
וַיָּבִיאוּ אֶל הַבַּיִת וַיִמְצְאָהוּ וְאֵת מִרְיָם אִמּוֹ וַיִּכְרְעוּ לִפְנֵיוּ וַיִּשְׁתַּחֲווּ לוֹ וַיִּפְתְּחוּ אֵת אֹמְתְּחוֹתֵיהֶם וְיָבִיוּ אֵלַיוּ מַתָּנוֹת אֵת זָהָב וְאֵת לְבוֹנָה וָמוֹר בַּלֹעֵז מִירָא

12.
וַיְצַוּוּ בַּחֲלוֹם לְבִלְתִּי שׁוּב יְרוּשָׁלַם אֶל הוֹרוֹדוֹס מִהַמַלְאָךְ אֵמֶר לָהֵם וַיִּפְנוּ דֶּרֶךְ אַרְצָם בַּדֶּרֶך אַחֶרֶת

13.
וַמֵאִחַר הֵמָּה הוֹלֵכִים וְהִנֵּה מַלְאָךְ יַהְוֶה נִרְאָה אֶל יוֹסֵף בַּחֲלוֹם לֵאמֹר קוּם וְקָח אֵת הַנַעַר וְאֵת אִמּוֹ וּבְרַח לְמִצְרַיִם וְשָׁם תעָמוֹד עַד אָמַרִי אֵלֶיךָ כִּי הוֹרוֹדוֹס יְבַקֵּשׁ אֵת הַנַעַר לְהֲרוֹג

14.
וְיִּקַץ וַיַּעַשׁ כַּאֲשֶׁר אָמַר לוֹ הַמַלְאָךְ וְיִקָח אֵת הַנַעַר וְאֵת אִמּוֹ בַּלָיְלָה וַיִּלֶךְ מִצְרַיִם

15.
וַיְהִי שָׁם עַד מֵת הוֹרוֹדוֹס לְגְמוֹר מָה שֶׁנֶאֱמַר מִיַהְוֶה עַל פִי הַנָבִיא הָאֹמֶר :וּמִמִּצְרַיִם קָרָאתִי לִבְנִי

16.
אָז רָאָה הוֹרוֹדוֹס שִׁרְאוּ אוֹתוֹ הַקוֹסְמִים וְיִרַע אֵלַיוּ מְאֹד וַיִּתְעַצֵּב אֶל לֵבוּ וַיְצַו וַיִּשְׁלַח לְכֹּל שָׂרָיו לְהָרוֹג לְכֹּל הַיְלָדִים אֲשֶׁר בַּבַּיִת-לֶחֶם וּגְּבוּלֶיהָ הַנוֹלָדִים מִזְמָן אֲשֶׁר אִמְרוּ לוֹ הַקוֹסְמִים שְׁנוֹלַד הַנַעַר

17.
אָז נִשָׁלֵם הַדָּבָר מָה שֶׁאָמַר יִרְמְיָהוּ הַנָּבִיא

18.
קוֹל בַּרָמָה נִשְׁמָע נְהִי בְּכִי תַמְרוּרִים רָחֵל מִבַכָּה עַל בָּנֶיהָ מֵאֲנָה לְהִנָּחֵם עַל בָּנֶיהָ כִּי אֵינֶנּוּ

19.
וַיְהִי כַּאֲשֶׁר מֵת הוֹרוֹדוֹס הַמֶלֶךְ הִנֵה וְמַלְאָך יַהְוֶה נִרְאָה מִחֲלוֹם אֶל יוֹסֵף בַּמִצְרַיִם

20.
לֵאמֹר קוּם קָח אֵת הַנַעַר וְאֵת אִמּוֹ וְלֵךְ אֶל אֶרֶץ יִשְׂרָאֵל כִּי מֵתוֹ הַמְבַקְשִׁים אֵת הַנַעַר לְהָמִית

21.
וַיָּקָם וַיִּקַּח אֵת הַנַעַר וְאֵת אִמּוֹ וְיָשׁוּבוּ אֶל אֶרֶץ יִשְׂרָאֵל

22.
וַיִּשָׁמַע כִּי הוֹרְקַנוֹס שְׁמוֹ אַרגֵילַאֻס מֶלֶךְ בַּיְהוּדָה תַּחַת הוֹרוֹדוֹס אָבִיו וַיִּרָא לָלֶכֶת שָׁם וַיִּזָהַר הַמַלְאָךְ בַּחֲלוֹם וַיֶּפֶן אֶל אֶרֶץ הַגָּלִיל

23.
וַיָּבֵא וַיַּשְׁכּוֹן בַּעִיר הַנִּקְרֵאת נָאזָרִית לְקַיֵּים מָה שֶׁאָמַר עַל פִּי הַנָבִיא נָאזָרֶת יִקָּרֵא
