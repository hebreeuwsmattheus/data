1.
אָז בָּאוּ אֶל־יֵשׁוּעַ הַחֲכָמִים וְהַפְּרוּשִׁים וַיֹּאמְרוּ אֵלָיו

2.
לָמָּה עוֹבְרִים תַּלְמִידֶיךָ תַּקָּנוֹת הָרִאשׁוֹנוֹת שֶׁהֵם אֵינָם רוֹחֲצִים יְדֵיהֶם קוֹדֶם הָאֲכִילָה

3.
וַיֹּאמֶר לָהֶם יֵשׁוּעַ וְלָמָּה אִתָּם עוֹבְרִים מַאַמְרֵי הַאֵל בְּעַד תַּקָּנוֹתֶיכֶם

4.
שֶׁהָאֵל אָמַר כָּבֵד אֶת־אָבִיךָ וְאֶת־אִמֶּךָ וּמַכָּה אָבִיו וְאִמּוֹ מוֹת יוּמָת

5.
וְאַתָּם אוֹמְרִים שֶׁאֵיזֶה דָּבָר יֹאמַר הָאָדָם לְאָבִיו וּלְאִמּוֹ שֶׁבְּאֵיזֶה נְדָבָה שֶׁיִּתֵּן בְּעַד אוֹתוֹ חֵטְא שֶׁיְּכוּפַּר לוֹ אוֹתוֹ עָוֹן

6.
וְלֹא יְכַבֵּד אָבִיו וְאִמּוֹ וְאַתָּם מְבַזִּים מַאֲמָרֵי הַאֵל בְּעַד תַּקָּנוֹתְיכֶם

7.
אוֹי חֲנֵפִים הִנֵּה נִבָּא יְשַׁעְיָה מִכֶּם וְאָמַר

8.
כֹּה אָמַר יְהֹוָה יָעֵן כִּי נִגַּשׁ הָעָם הַזֶּה בְּפִיו וּבִשְׂפָתָיו כִּבְּדוּנִי וְלִבּוֹ רִחַק מִמֶּנִּי

9.
וַתְּהִי יִרְאָתָם אוֹתִי מִצְוַת אֲנָשִׁים מְלוּמָּדָה

10.
וִיֵשׁוּעַ קָרָא לְסִּיעָוֹת וַיֹּאמֶר לָהֶם שָׁמְעוּ וְהִבִּיטוּ

11.
הַנִכְנַס בַּעַד הַפֶּה אֵינוּ מְלַכְלֵך הָאָדָם אֲבָל הַיּוֹצֵא מִהַפֶּה מְלַכְלֵך הָאָדָם

12.
אָז קָרְבוּ אֵלָיו תַּלְמִידָיו וַיֹּאמְרוּ לוֹ דַּע שֶׁהַפְּרוּשִׁים נְבוּכִים בְּעַד דָּבָר זֶה

13.
וַיַּעַן לָהֶם יֵשׁוּעַ כָּל־נְטִיעָה שֶׁלֹא נָטִיעָה אֲבִי שֶׁבַּשָּׁמַיִם תְּשַׁחֵת

14.
הֵנִיחוּ אוֹתָם שֶׁהָעִוְּרִים מַדְרִיכִים לַעִוְּרִים וְאִם עִוֵּר יַדְרִיךְ עִוֵּר אַחֵר יִפְלוּ שִׁנֵּיהֶם בְּבוֹר

15.
וַיַּעַן לוֹ פֶּיטְרוֹשׁ אֲדוֹנִי פְּרוּשׁ לָנוּ זֹאת הַחַדָּה

16.
וַיַּעַן לָהֶם יֵשׁוּעַ עֲדַיִין אַתָּם מִבְּלִי דַּעַת

17.
לֹא תָּבִינוּ אַתָּם שֶׁכָּל הַנִּכְנַס בְּעַד הַפֶּה הוֹלֵךְ לַבֶּטֶן וְהַכֹּל הוֹלֵךְ בְּעַד הַמָּקוֹם הַטִּבְעִי

18.
וְהַיּוֹצֵא בְּעַד הַפֶּה מִתְנוֹעֵעַ מֵהַלֵּב וְזֶהוּ הַמְּלַכְלֵךְ הָאָדָם

19.
לְפִי שֶׁמְּחַלֵּל הַלֵּב יוֹצֵא הַתַּרְמִית וְהָרְצִיחָה וְהַנִּאוּפִים וְהַגַּנָּבוֹת וְעֵדוֹת שְׁקָרִים וְהַקְּלָלוֹת

20.
וְכָל אֵלֶּה הַדְּבָרִים הֵם הַמְּלַכְלְכִים הָאָדָם אָמְנָם הָאֲכִילָה בְּלֹא רְחִיצַת יָדַיִם אֵינָהּ מְלַכְלֶכֶת הָאָדָם

21.
וְאַחֵר שֶׁאָמַר יֵשׁוּעַ זֶה הָלַךְ בִּגְלִילֵי צוֹר וְסִידוֹן

22.
וְתָבוֹא לְפָנָיו אִשָּׁה כְּנַעֲנִית בָּאָה מֵאֲרָצוֹת מִזְרַח צוֹעֶקֶת אֵלָיו אֲדוֹנִי בֶּן־דָּוִד חִנָּנִי שֶׁבִּתִּי אֲחוּזַת הַשֵּׁדִים

23.
וִיֵשׁוּעַ לֹא עָנָה לָהּ דָּבָר וְתַּלְמִידָיו קִרְבוּ אֵלָיו וַיֹּאמְרוּ לוֹ אֲדוֹנֵינוּ לָמָּה אַתָּהּ מֵנִיחַ לְזֹאת הָאִשָּׁה צוֹעֶקֶת אַחֲרֵינוּ

24.
וַיַּעַן לָהֶם יֵשׁוּעַ לֹא שְׁלָחוּנִי כִּי אִם לַצֹּאן אוֹבְדוֹת מִבֵּית יִשְׂרָאֵל

25.
וְהָאִשָּׁה מִשְׁתַּחֲוֶה לוֹ וְאוֹמֶרֶת אֲדוֹנִי עֹזְרֵנִי

26.
וַיֹּאמֶר לָהּ יֵשׁוּעַ לֹא טוֹב שֶׁיִקַּח הָאָדָם הַפַּת מִבָּנָיו וְיִתְנֵנוּ לַכְּלָבִים

27.
וַתַּעַן הָאִשָּׁה פְּעָמִים רַבִּים אוֹכְלִים הַכְּלָבִים הַפְּתִיתִים הַנּוֹפְלִים מֵשֻּׁלְחָן אֲדוֹנֵיהֶם

28.
וַיַּעַן לָהּ יֵשׁוּעַ אִשָּׁה גְּדוֹלָה אֲמוּנַתְךָ יַעֲשֶׂה לְךָ כַּאֲשֶׁר שְׁאֵלַת וּמֵהָעֵת הַהוּא וְהָלְאָה נִרְפֵּאת בַּתֵּה

29.
וְכַאֲשֶׁר הָלָךְ יֵשׁוּעַ מִשָּׁם הָלָךְ עָבַר הַגָּלִיל לָהָר בַּעוֹמְדוּ שָׁם

30.
רָאָה עַם רָב מִבֵינֵיהֶם הַרְבֵּה צוֹלְעִים וּמְנּוּגָּעִים וּפִסָחִים וְרַבִּים אֲחֵרִים וְיִפְלוּ לְרַגְלָיו וִירַפְּאֵם

31.
וְהֵעֵם הָיוּ תְּמֵהִים אֵיךְ הָאִלְּמִים הָיוּ מִדְבָּרִים וְהַפִּסְחִים הוֹלְכִים וְהָעִוְּרִים רוֹאִים וְכֻלָּם מְשַׁבְּחִים לָאֵל

32.
אָז אָמַר יֵשׁוּעַ לְתַּלְמִידָיו יֵשׁ לִי רַחְמָנוּת מֵהֶם שֶׁהֵם מְיֻחָלִים אוֹתִי זֶה שְׁנֵי יָמִים שֶׁעָבְרוּ וְאֵין לָהֶם מָה שֶׁיֹּאכְלוּ וְאֵינִי רוֹצֶה לְהוֹלִיכָם בַּתַּעֲנִית יָעֵן לֹא יֵחָלְשׁוּ בַּדֶּרֶךְ

33.
וַיַּעֲנוּ לוֹ תַּלְמִידָיו וּמֵאַיִן אָנוּ יְכוֹלִים לִמְצוֹא לֶחֶם בְּמִדְבָּר הַזֶּה לִשְׂבִּיעַ הָעָם

34.
וְיֵשׁוּעַ אָמַר לָהֶם כַּמָּה כְּכָרוֹת לֶחֶם לָכֶם וְהֵם עָנוּ שִׁבְעָה וּמְעַט דָּגִים

35.
וְיַצוּ יֵשׁוּעַ לָעָם לֵישֵׁב עַל גַּבֵּי הָעֲשָׂבִים

36.
וְלָקַח הַשִּׁבְעָה כִּכָּרוֹת לֶחֶם וְהוֹדָה לְאֵל וְיִשְׁבְּרֵם וְנַתְנֵם לְתַּלְמִידָיו וְהֵם נָתְנוּ לָעָם

37.
וְיֹּאכְלוּ כּוּלָּם וְיִשְׂבְּעוּ וְמֵהַנִּשְׁאָר מָלְאוּ שִׁבְעָה סֵאִים

38.
וְהָאוֹכְלִים הָיוּ בְמִּסְפָּר אַרְבָּעַת אֲלָפִים אֲנָשִׁים לְבַד הַנָּשִׁים וְהַטַּף

39.
אַחֵר זֶה נִכְנַס יֵשׁוּעַ בַּסְּפִינָה וְבָּא לָאָרֶץ מַגְדַּלָה
