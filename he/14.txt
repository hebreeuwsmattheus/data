1.
בַּעֵתּ הַהִיא שָׁמַע הוֹרוֹדוֹס טִיטְרָאקָה שְׁמוּעוֹת יֵשׁוּעַ

2.
וַיֹּאמֶר לַעֲבָדָיו הִנֵּה אֲנִי מַאֲמִין שֶׁאִלוּ הַפְּלָאוֹת עוֹשֶׂה יוֹחָנָן הַמַּטְבִּיל

3.
שֶׁהָיָה לְפִי שֶׁהוֹרוֹדוֹס תָּפַשׂ לְיוֹחָנָן בַּיָּמִים הָהֵם וַיֶאֶסְרֻהוּ בַּמַאֲסָר לְפִי שֶׁהָיָה מוֹכִיחוֹ שֶׁלֹא יִקַּח לְאוֹרְדִיסָה לָאִשָּׁה שֶׁהָיָתָה אֵשֶׁת אֶחָיו

4.
וְהָיָה אוֹמֵר לוֹ יוֹחָנָן אֵינָהּ רְאוּיהָ לְךָ

5.
וְהִנֵּה הוֹרוֹדוֹס הָיָה רוֹצֶה לְהוֹרֵגוּ לוּלֵי  יִרְאַת הָעָם שֶׁלַּנָּבִיא הָיָה בֵּינֵיהֶם

6.
וּבְמִּשְׁתֶּה יוֹם הוּלֶּדֶת הוֹרוֹדוֹס קָרָא לִגְדוֹלֵי הַמַּלְכוּת לֶאֱכוֹל עִמּוֹ וּבְעוֹד שֶׁהָיוּ אוֹכְלִים הָיְתָה בִּתּוֹ מְרַקֵּדָת בֵּינֵיהֶם וַיִיטַב לְהוֹרוֹדוֹס

7.
וַיִּשָּׁבַע לָהּ שֶׁיִּתֵּן לָהּ כָּל אֲשֶׁר תִּשְׁאֵל מִמֶּנּוּ

8.
וְהַנַּעֲרָה מִיוּסֶּרֶת מִאִמָּהּ שְׁאֵלָה רֹאשׁ יוֹחָנָן הַמַּטְבִּיל בַּאַגָּן אֶחָד

9.
וְהַמֶּלֶךְ נֶעֱצַב מְאֹד בְּעַד הַשְּׁבוּעָה שֶׁעָשָׂה  בִּפְנֵי הַקְּרוּאִים וַיְֹצַו לְהֵעָשׂוֹת כֵּן

10.
וְיִּשְׁלַח לִשְׁחוֹט יוֹחָנָן בְּבֵית הַסּוֹהַר

11.
וְיָבִיאוּ רֹאשׁ יוֹחָנָן בָּאַגָּן וְיַתְנוּהוּ לַנַּעֲרָה וְהַנַּעֲרָה נָתְנָה לְאִמָּהּ

12.
וְיַבְּאוּ תָּלְמִידֵי יוֹחָנָן וְיִשְּׂאוּ הַגּוּף וְיִקְבְּרוּהוּ וְהַתַּלְמִידִים הִגִּידוּ הַדָּבָר לִיֵשׁוּעַ

13.
וּכְשִׁמּוּעַ יֵשׁוּעַ נָסַע מִשָּׁם בָּאֳנִיָּה וְיֵּלֵךְ לַמִּדְבָּר יְהוּדָה וּכְשִׁמּוּעַ הַחֲבוּרוֹת וִיֵּלֵכוּ אַחֲרָיו מִכָּל הַמְּדִינוֹת

14.
וּכְשֶׁיָּצְאוּ רָאָה אַחֲרָיו עַם רָב וְיֵט אֵלָיו חֶסֶד וִירַפֵּא כָּל מְחֹלוּתֶם

15.
וּבַעַת עֶרֶב קִרְבוּ אֵלָיו תַּלְמִידָיו וַיֹּאמְרוּ לוֹ זֶה הַמָּקוֹם צַר וְהַעַת עוֹבֵר עֲזוּב הַחֲבוּרוֹת שֶׁיֵּלְכוּ בַּמִּגְדָּלִים וְיִקְּחוּ הַצּוֹרֶךְ אֲלֵיהֶם

16.
וַיַּעַן לָהֶם יֵשׁוּעַ אֵינָם צְרִיכִים לִילָךְ תְּנוּ לָהֶם לֶאֱכֹל

17.
וְהֵם עָנוּ אֵין־לָנוּ בְּכָאן כִּי אִם חָמֵשׁ כִּכָּרוֹת וְשְׁנֵי דָּגִים

18.
וַיֹּאמֶר לָהֶם הֵבִיאוּ אוֹתָם אֵלַי

19.
וְיִצַוּ שֶׁיָּשָׁבוּ הָעָם עַל הָעֲשָׂבִים וּכְשֶׁיָּשְׁבוּ לָקַח הֶחָמֵשׁ כִּכָּרוֹת וְהַשְׁנֵי דָּגִים וּבִהְיוֹתוֹ מַבִּיט לְשָּׁמַיִם בֶּרֶךְ אוֹתָם וִיחַלְּקֵם וְיַתְנֵם לְתַּלְמִידָיו וְהַתַּלְמִידִים חֶלְקוּ לְסַיְּיעוֹת

20.
וְאָכְלוּ כֻּלָּם וְיִשְׂבְּעוּ וְכֵן מֵהַדָּגִים אָכְלוּ כִּרְצוֹנָם וְאַחֵר שֶׁתְּכַלּוּ לָקְחוּ הַפְּתִיתִים הַנְשָׁארוֹת וִימַלְּאוּ מֵהֶם שָׁנִים עֶשֶׂר סֵאִים

21.
וַיְהִי מִסְפָּר הָאוֹכָלִים חַמְשַׁת אֲלָפִים אֲנָשִׁים מִלְּבַד הַנָּשִׁים וְהַטָּף

22.
וְאַחֲרֵי זֶה צִוָּה לְתַּלְמִידָיו לַכֶּנֶס בָּאֳנִיָּה וְיִשְׁלֵיחוּ קוֹדֵם מִמֶּנּוּ בָּעִיר שֶׁהַחֲבוּרוֹת הוֹלְכוֹת

23.
וְאַחֵר שֶׁעָזַב הַחֲבוּרוֹת עָלָה לָהָר וְהִתְפַּלֵּל לְבַדּוּ וַיְהִי לָעֵת עֶרֶב וְהוּא לְבַדּוּ עוֹמֵד

24.
וְהָאֳנִיָּה בָּאֶמְצַע הַיָּם וְהַגַּלִים הָיוּ דּוֹחֲפוֹת אוֹתָהּ לְפִי שֶׁהַרוּחַ הָיָה מִנֶּגֶד

25.
וְלְאַשְׁמוֹרֶת הָרְבִיעִית מֵהַלַּיְלָה בָּא לָהֶם יֵשׁוּעַ הוֹלֵךְ בַּיָּם

26.
וְכַאֲשֶׁר רָאוּהוֹ תַּלְמִידָיו הוֹלֵךְ בַּיָּם נִבְהֲלוּ בַּשַּׂר בְּחוֹשֵׁבָם שֶׁהָיָה שֵׁד וּמֵרוֹב פַּחֲדָם הָיוּ צוֹעֲקִים

27.
וְאָז עָנָה לָהֶם יֵשׁוּעַ וַיֹּאמֶר לָהֶם יִהְיֶה אֱמוּנָה בָּכֶם שֶׁאֲנִי הוּא וְאַל־תִּירָאוּ

28.
וַיַּעַן פֶּיטְרוֹשׁ וַיֹּאמֶר לוֹ אֲדוֹנִי אִם אִתָּהּ הוּא צִוָּה אוֹתִי לְבָא אֵלֶיךָ בַּמַּיִם

29.
וַיֹּאמֶר לוֹ יֵשׁוּעַ בָּא וְיָרַד פֶּיטְרוֹשׁ מֵהַסְּפִינָה וְהָלַךְ בַּיָּם וְבָא לִיֵשׁוּעַ

30.
וּבִרְאוֹתוֹ חוֹזֶק הָרוּחַ פָּחַד מְאֹד וּבְהַתְחִילוּ לְיִּטְבַּע צָעַק וְאָמַר אֲדוֹנִי הוֹשִׁיעַנִי

31.
וּמִיַּד יֵשׁוּעַ הֶאֱרִיךְ יָדוֹ וְלָקְחוּ וְאָמַר לוֹ אָדָם מֵאֱמוּנָה מִעוּטָּהּ לָמָּה נִסְתַּפֵּקֶת

32.
וְכַאֲשֶׁר עָלוּ בָּאֳנִיָּה נָח הָרוּחַ

33.
וַאֲשֶׁר בַּסְּפִינָה הִשְׁתַּחֲווּ לוֹ וְאָמְרוּ בֶּאֱמֶת אִתָּהּ הוּא בֵּן הָאֱלֹהִים

34.

35.
וְכַאֲשֶׁר הִכִּירוּהוּ אַנְשֵׁי הַמָּקוֹם שָׁלְחוּ בְּכָל אוֹתוֹ הַמַּלְכוּת וְהֵבִיאוּ לוֹ כָּל־הַחוֹלִים מִכָּל־מַדְוִים

36.
וְחָלוּ פָּנָיו יִרְצוּ לְעוֹזְבָם יָגְעוּ בַּכָּנָף מְעִילוֹ וְכָל אֲשֶׁר נָגַעוּ נִתְרַפֵּאוּ
