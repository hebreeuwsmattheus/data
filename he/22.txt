1.
וַיַּעַן יֵשׁוּעַ וַיֹּאמֶר לָהֶם עוֹד בְּדִבְרֵי מָשָׁל

2.
מַלְכוּת שָׁמַיִם דּוֹמֶה לַמֶּלֶךְ אֲשֶׁר עוֹשֵׂה חוּפָּהּ

3.
וַיִשְׁלַח אֶת־עֲבָדָיו בְּעַד הַקְּרוּאִים לַחוּפָּה וְלֹא אָבוּ לָבֹא

4.
וַיִשְׁלַח עוֹד עֲבָדִים אֲחֵרִים לֵאמֹר אִמְרוּ לַקְּרוּאִים הִנֵּה הֲכִינוֹתִי הַמִּשְׁתֶּה וְזָבַחְתִּי  שְׁוָרִים וְעוֹפוֹת וְהַכֹּל מוּכָן בּוֹאוּ אֶל הַחוֹפָּה

5.
וְהֵם בָּזוּ וַיֵלְכוּ מִקְצָתָם בָּעִיר וּמִקְצָתָם בְּעֲסָקֵיהֶם

6.
וְהָאֲחֵרִים לָקְחוּ אֶת־עֲבָדָיו וְהִתְעַלְּלוּ בָּם וַהֲרָגוּם

7.
וַיִשְׁמַע הַמֶּלֶךְ וַיִחַר אָפּוֹ וַיִשְׁלַח חֵיל וַיַּשְׁמִיד הָרוֹצְחִים הָהֵם וְאֶת־בֵּיתָם שָׂרַף בָּאֵשׁ

8.
אָז אָמַר לַעֲבָדָיו הַחוּפָּה מוּכָנָת הִיא רַק הַקְּרוּאִים לֹא הָיוּ רְאוּיִים

9.
וְעַתָּה צְאוּ אֶל הַדְּרָכִים וְכֹל אֲשֶׁר תִּמְצְאוּ קָרְאוּ לְחוּפָּה

10.
וַיֵצְאוּ עֲבָדָיו אֶל הַדְּרָכִים וְיִקְבְּצוּ כֹּל הַנִּמְצְאִים טוֹבִים וְרָעִים וַתִּמָּלֵא הַחוּפָּה מֵהָאוֹכְלִים

11.
וַיָּבֹא הַמֶּלֶךְ לִרְאוֹת הָאוֹכְלִים וַיַּרֵא שָׁם אָדָם אֲשֶׁר לֹא הָיָה מְלוּבָּשׁ בִּגְדֵי הַחוּפָּה

12.
וַיֹּאמֶר לוֹ אֲהוּבִי אֵיךְ בָּאתָה לְכַּאן בְּלֹא לְבוּשֵׁי הַחוּפָּה וְהוּא הֶחֱרִישׁ

13.
אָז אָמַר הַמֶּלֶךְ לִמְשָׁרְתָיו אִסְרוּ יָדָיו וְרַגְלָיו וְהַשְׁלִיכוּהוּ בְּמַחֲשַׁכֵּי תַּחְתִּיּוֹת וְשָׁם יִהְיֶה בֶּכִי וְחֲרוֹק שִׁינַּיִם

14.
הַקְּרוּאִים רַבִּים וְהַנִּבְחָרִים מְעַטִּים

15.
אָז הָלְכוּ הַפְּרוּשִׁים וַיִוָּעֲצוּ לְקַחְתּוֹ בִּדְבַר

16.
וַיִשְׁלֵּיחוּ אֵלָיו מִתַּלְמִידֵיהֶם עִם פָּרִיצִים מֵהוֹרוֹדוֹס לֵאמֹר רַבִּי יָדַעְנוּ שֶׁנֶּאֱמָן אַתָּה וְתַּלְמוֹד בָּאֱמוּנָה דָּרֶךְ הָאֱלֹהִים וְאֵינְךָ חוֹשֵׁשׁ לַשּׁוּם דָּבָר וְלֹא נוֹשֵׂא פָּנִים

17.
אֱמוֹר לָנוּ מָה יִרְאֶה לְךָ הַנְכוֹן לָתֶת מַס לְצֵיזָר אִם־אָיִן

18.
וַיַּכֵּר יֵשׁוּעַ אֶת־נִכְלוֹתֶם וַיֹּאמֶר לָמָּה תְּסִיתוּנִי חֲנֵפִים

19.
הַרְאוּ לִי מַטְבֵּעַ הַמַּס וַיָבִיאוּ לוֹ

20.
וַיֹּאמֶר אֲלֵיהֶם לְמִי הַצּוּרָה הַזֹּאת וְהָרָשׁוּם

21.
וְיֹאמְרוּ לְצֵיזָר אָז אָמַר אֲלֵיהֶם יֵשׁוּעַ הֵשִׁיבוּ לְצֵיזָר אֶת־אֲשֶׁר לְצֵיזָר וְאֶת־אֲשֶׁר לֵאלֹהִים לֵאלֹהִים

22.
וַיִשְׁמְעוּ וַיִתְמְהוּ וַיַּעֲזֹבוּהוּ וַיֵלְכוּ

23.
בַּיּוֹם הַהוּא קָרְאוּ אֵלָיו הַצְּדּוּקִים וְהַכּוֹפְּרִים בִּתְּחִייַת הַמֵּתִים וְיִשְׁאֲלוּהוּ

24.
לֵאמֹר רַבִּי אָמוֹר לָנוּ אָמַר מֹשֶׁה כִּי־יֵשְׁבוּ אַחִים יַחְדָּיו וּמֵת אַחַד מֵהֶם וּבֵן אֵין־לוֹ שֶׁיִּקַּח אָחִיו אֶת־אִשְׁתּוֹ לְקִייֵם זָרַע אָחִיו

25.
וְהִנֵּה שִׁבְעָה אַחִים הָיוּ בֵּינֵינוּ וְנָשָׂא הָרִאשׁוֹן אִשָּׁה וּמֵת בְּלֹא זֶרַע וְיַבֵּם אָחִיו אֶת־אִשְׁתּוֹ

26.
וְכֵן הַשֵּׁנִי וּשְׁלִישִׁי עַד הַשְּׁבִיעִי

27.
וְאַחֲרֵיהֶם מֵתָה הָאִשָּׁה

28.
שֶׁכְּבָר הָיְתָה לְכּוּלָּם אֶל מִי מֵהַשִּׁבְעָה תִּהְיֶה הָאִשָּׁה

29.
וַיַּעַן יֵשׁוּעַ וַיֹּאמֶר אֲלֵיהֶם תִּשְׁגּוּ וְלֹא תָּבִינוּ הַסְּפָרִים וְעוֹז הָאֱלֹהִים

30.
בַּיּוֹם הַתְּקוּמָה לֹא יִקְּחוּ הָאֲנָשִׁים נָשִׁים וְלֹא הַנָּשִׁים אֲנָשִׁים רַק יִהְיוּ כְּמַלְאֲכֵי אֱלֹהִים  בַּשָּׁמַיִם

31.
הַלֹּא קְרָאתֶם מִתְּחִיַּית הַמֵּתִים שֶׁאָמַר יְהֹוָה לָכֶם 

32.
אֲנִי יְהֹוָה אֱלֹהֵי אַבְרָהָם אֱלֹהֵי יִצְחָק וֶאֱלֹהֵי יַעֲקֹב וְאִם כֵּן אֵינוּ אֱלֹהֵי הַמֵּתִים כִּי אִם אֱלֹהֵי הַחַיִּים

33.
וְיִשְׁמְעוּ הַחֲבוּרוֹת וְיִתְמְהוּ מֵחָכְמָתוֹ

34.
וְכַאֲשֶׁר רָאוּ הַפְּרוּשִׁים כִּי אֵין מַעֲנֶה לַצְּדּוּקִים הִתְחַבְּרוּ יַחְדָּיו

35.
וְיִשְׁאֲלוּהוּ חָכָם אֶחָד לְנַסּוֹתוֹ

36.
רַבִּי אָמוּר אֵיזֶה הִיא מִצְוָה גְּדוֹלָה שֶׁבַּתּוֹרָה

37.
אָמַר לוֹ וְאָהַבְתָּ אֵת יְהֹוָה אֱלֹהֶיךָ בְּכָל־לְבָבְךָ וּבְכָל־נַפְשְׁךָ וּבְכָל־מְאֹדֶךָ

38.
זוֹ הִיא הָרִאשׁוֹנָה הַמִּצְוָה גְדוֹלָה

39.
וְהַשֵּׁנִית דּוֹמֶה אֵלֶיהָ וְאָהַבְתָּ לְרֵעֲךָ כָּמוֹךָ

40.
וְעַל שְׁתֵּי הַמִּצְווֹת הָאֵלֶּה הַתּוֹרָה כּוּלָּהּ תְּלוּיָה וְהַנְּבִיאִים

41.
וְיֶאֶסְפוּ הַפְּרוּשִׁים וְיִשְׁאֲלֵם יֵשׁוּעַ

42.
לֵאמֹר מָה יִרְאָה לָכֶם מִן הַמָּשִׁיחַ וּבֵן מִי יִהְיֶה וְיֹאמְרוּ לוֹ בֵּן דָּוִד

43.
וַיֹּאמֶר לָהֶם אֵיךְ קָרְאוּ אוֹתוֹ דָּוִד בָּרוּחַ הַקֹּדֶשׁ לֵאמֹר אָדוֹן

44.
דִכְתִיב נְאֻם יְהֹוָה לַאדוֹנִי שֵׁב לִימִינִי עַד־אָשִׁית אוֹיְבֶיךָ הֲדוֹם לְרַגְלֶיךָ

45.
אִם דָּוִד קָרְאוּ אוֹתוֹ אָדוֹן אֵיךְ יִהְיֶה בְּנוֹ

46.
וְלֹא יָכְלוּ לְהָשִׁיבוֹ דָּבָר מִכָּאן וְאֵילָךְ פָּחֲדוּ לִשְׁאוֹל מִמֶּנּוּ דָּבָר
