1.
אַל תִּדַּיְּנוּ פָּן תִּדּוֹנוּ

2.
בְּאֵיזֶה דִּין תָּדּוּנוּ וּבְאֵיזֶה מִדָּה תָּמוֹדּוּ יֵּמוֹדּוּ לָכֶם

3.
וְלָמָה תִּרְאוּ קַשׁ בְּעַיִן זוּלָתְךָ וְלֹא תִּרְאֶה קוֹרָה שֶׁבְּעֵינֶיךָ

4.
וְאֵיךְ תּאׁמַר לְזוּלָתְךָ כַּתַּר-לִי זְעֵיר וְאוֹצִיא קַשׁ מֵעֵינֶיךָ וְהִנֵּה הַקָּרָה בְּעֵינֶיךָ

5.
הֶחָנֵף תּוֹצִיא קוֹדֶם הַקּוֹרָה מֵעֵינֶיךָ וְאַחַר תּוֹצִיא הַקַּשׁ מֵעֵיִן זוּלָתְךָ

6.
עוֹד אָמַר לָהֶם אַל תַּתְנוּ בַּשַּׂר קֹדֶשׁ לַכְּלָבִים וְאַל תָּשִׂימוּ פְּנֵינִים לִפְנֵי חֲזִיר פָּן יְכַרְסְמְנוּ אוֹתָהּ לְעֵינֵיכֶם וִיחַזֵּרוּ אוֹתָהּ לִקְרוֹעַ אֶתְכֶם

7.
שָׁאֲלוּ מֵהָאֵל וְיִנָּתֵן לָכֶם בַּקְּשׁוּ וְתִמְצְאוּ דָּפְקוּ וְיִפְתְּחוּ לָכֶם

8.
כֹּל הַשּׁוֹאֵל יְקַבֵּל וּלַאֲשְׁר יְבַקֵּשׁ יִמָּצֵא וְלַקּוֹרֵא יְפָּתֵּחַ

9.
מִי בָּכֶם שֶׁיְּבַקֵּשׁ בָּנוֹ מִמֶּנּוּ לִפְרוֹס לֶחֶם וְיִתֵּן לוֹ אֶבֶן

10.
אוֹ אִם יְבַקֵּשׁ דָּג וְיִתֵּן נָחָשׁ

11.
וְאִם אִתָּם עִם הֱיוֹתְכֶם רָעִים תָּבוֹאוּ לָתֵת מַתָּנוֹת טוֹבוֹת לִבְנֵיכֶם כֹּל שְׁכֵּן אֲבִיכֶם שֶׁבַּשָּׁמַיִם שֶׁיִתֵּן רוּחוֹ הַטּוֹבָה לִמְבַקְּשָׁיו

12.
וְכָל מָה שֶׁתִּרְצוּ שֶׁיַּעֲשׂוּ לָכֶם בְּנֵי אָדָם עֲשׂוּ לָהֶם זֹאת הַתּוֹרָה וְדִבְרֵי הַנְּבִיאִים

13.
בַּזְּמַן הַהוּא אָמַר יֵשׁוּעַ לְתַלְמִידוֹ בָּאוּ בַּשֵּׁעָר הַצַּר שֶׁשֵּׁעָר הָאֲבַדּוֹן רָחָב וּמְצוּלָה וְרַבִּים הוֹלְכִים בָּהּ

14.
כַּמָּה הַשֵּׁעָר צַר וְכַמָּה כָּבֵד הַדֶּרֶךְ הַמִיּישֶׁר לְחַיִּים וּמְעַטִּים הַמוֹצָאִים אוֹתוֹ

15.
עוֹד אָמַר לָהֶם הֻזְהֲרוּ מִנְּבִיאֵי הַשֶּׁקֶר הַבָּאִים לָכֶם בְּמַלְבּוּשֵׁי הַצֶמֶר דּוֹמִים לַצֹּאן שֶׁבְתוֹכָם זְאֵבִים טוֹרְפִים

16.
וּבְמַעֲשֵׂיהֶם תַּכִּירוּם הַיַּלְקוּט אָדָם מִן הַקּוֹצִים עֲנָבִים וּמִן הַבַּרְקוֹנִים תְּאֵנִים

17.
שֶׁכָּל עֵץ טוֹב יַעֲשֶׂה פְּרִי טוֹב וְכָל עֵץ רַע יַעֲשֶׂה פְּרִי רַע

18.
וְעֵץ הַטּוֹב לֹא יוּכַל לַעֲשׂוֹת פְּרִי רַע וְעֵץ רַע לֹא יוּכַל לַעֲשׂוֹת פְּרִי טוֹב

19.
וְכָל עֵץ אֲשֶׁר לֹא יַעֲשֶׂה פְּרִי טוֹב כַּאֲשֶׁר יִשְׂרֹף

20.
לָכֵן כַּפְרִיִּים רֹצֵה לוֹמַר בְּמַעֲשֵׂיהֶם תַּכִּירוּם

21.
שֶׁכָּל הַאוֹמֵר אֵלַי אֲדוֹנִי לֹא יָבֵּא בְּמַלְכוּת הַשָׁמַיִם אֲבָל כָּל הָעוֹשֶׂה רָצוֹן אֲבִי שֶׁבַּשָּׁמַיִם יְכַנֵּס בְּמַלְכוּת הַשָׁמַיִם

22.
רַבִּים יֹאמֵרוּ אֵלַי בַּיּוֹם הַהוּא אֲדוֹנִי אֲדוֹנִי הֲלֹא בַּשֶּׁמֵךָ נִבֵּאנוּ וּבַשֶּׁמֵךָ שֵׁדִים הוֹצָאָתֵנוּ וְאוֹתוֹת רַבּוֹת עַל שֶׁמֵךָ עָשִׂינוּ

23.
וְאָז אוֹמָר לָהֶם מֵעוֹלָם לֹא יָדַעְתִּי אֶתְכֶם סוּרוּ מִמֶּנִּי כָּל-פּוֹעֲלֵי אוֹן

24.
עוֹד אוֹמָר לָהֶם כָּל הַשּׁוֹמֵעַ דְּבָרִים אִלוּ וְעוֹשֶׂה אוֹתָם דּוֹמֶה לָאִישׁ חָכָם שֶׁבָּנָה בֵיתוֹ בַּסֶּלַע

25.
וְיָרַד הַגֶּשֶׁם עָלָיו וְהָרוּחוֹת מַקִּישׁוֹת אוֹתוֹ וְלֹא יִפוֹל לְפִי שֶׁיְּסוֹדוֹ אֶבֶן

26.
וְכָל הַשּׁוֹמֵעַ דִּבְרֵי אֵלֶּה וְלֹא יַעֲשֵׂם דּוֹמֶה לָאִישׁ שׁוֹטֶה אֲשֶׁר בָּנָה בֵּיתוֹ עַל הַחוֹל

27.
וְיָרְדוּ גְּשָׁמִים וְיָבֹאוּ זְרָמִים וְיַפִּילוּהוּ וְנֹפֵל מַפָּלָה גְּדוֹלָה

28.
וּבְעוֹד שֶׁיֵשׁוּעַ הָיָה מְדַבֵּר דְּבָרִים אֵלּוּ כָּל הָעָם הָיוּ תְּמֵהִים מְגוּדָל טוֹב הַנְהָגָתוֹ

29.
לְפִי שֶׁהוּא שֶׁהָיָה דּוֹרֵשׁ לָהֶם בְּכֹחַ גָּדוֹל שֶׁלֹּא כַּאֲשֶׁר הַחֲכָמִים
