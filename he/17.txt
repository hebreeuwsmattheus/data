1.
אַחַר שִׁשָּׁה יָמִים לָקַח יֵשׁוּעַ לְפֶּיטְרוֹשׁ וְגַם יַעֲקֹב לֹעֵז גִאימִי וְיוֹחָנָן אָחִיו וַיוֹלִיכֵם אֶל הַר גָּבוֹהַּ מִשָּׁם לְהִתְפַּלֵּל הוּא

2.
וּבְעוֹד שֶׁהָיָה מִתְפַּלֵּל הִשְׁתַּנָּה לִפְנֵיהֶם וְקֶרֶן עוֹר פָּנָיו כְּשֶּׁמֶשׁ וּמַלְבּוּשָׁיו לְבָנִים כְּשֶׁלֶג

3.
וְנִגְלוּ אֲלֵיהֶם מֹשֶׁה וְאֵלִיָּהוּ מְדַבְּרִים עִמּוֹ וְהִגִּידוּ לִיֵשׁוּעַ כָּל מָה שֶׁיִּקְרְאֵהוּ בִיְרוּשָׁלָם וּפֶּיטְרוֹשׁ וְחֲבֵירָיו הָיוּ נִרְדָּמִים מְאֹד נִים וְלֹא נִים תִּיֵּר וְלֹא תִּיֵּר רָאוּ גּוּפוֹ וְשֵׁנִי אֲנָשִׁים עִמּוֹ

4.
וְכַאֲשֶׁר הָלְכוּ אָז אָמַר פֶּיטְרוֹשׁ לְיֵשׁוּעַ אֲדוֹנִי טוֹב לִהְיוֹת בְּכָאן וְנַעֲשָׂה פֹּה שֶׁלָּשׁ מִשְׁכָּנוֹת לְךָ אֶחָד וּלְמֹשֶׁה אֶחָד וּלְאֵלִיָּה אֶחָד שֶׁלֹּא הָיָה יוֹדַע מָה הָיָה דּוֹבֵר

5.
עוֹדֶנּוּ מְדַבֵּר וְהִנֵּה עָנָן שֶׁכָּסָה אוֹתָם וְיִבָּהֲלוּ עַד־מְאֹד וּבְעוֹד שֶׁהֵם תַּחַת הֶעָנָן שָׁמֵעוּ מִתּוֹךְ הֶעָנָן קוֹל מִדְבָּר וְאוֹמֵר הִנֵּה זֶה בְּנִי יַקִּירִי וְחֶפְצִי בּוֹ אֵלָיו תִשְׁמָעוּן

6.
וְיִּשְׁמְעוּ הַתַּלְמִידִים וְיִפְלוּ עַל פְּנֵיהֶם אַרְצָה וְיִירְאוּ מְאֹד

7.
וְכַאֲשֶׁר נִפְסַק הַקּוֹל וַיֹּאמֶר לָהֶם יֵשׁוּעַ קוּמוּ אַל־תֵּירָאוּ

8.
וַיִּשְּׂאוּ עֵינֵיהֶם וְלֹא רָאוּ כִּי אִם יֵשׁוּעַ בַּלְבַד

9.
וְיָרַד יֵשׁוּעַ מִן הָהָר וַיְצוֹ לָהֶם לֵאמֹר אַל תְּדַבְּרוּ לָאִישׁ הַמַּרְאֶה אֲשֶׁר רְאִיתֶם עַד שֶׁיָקוּם בֶּן־הָאָדָם מִן הַמָּוֶת 

10.
וְיִשְׁאֲלוּ לוֹ תַּלְמִידָיו לֵאמֹר מָה חֲכָמִים אוֹמְרִים שֶׁאֵלִיָּה יִבֵּא רִאשׁוֹנָה

11.
וַיַּעַן לָהֶם וַיֹּאמֶר אָמְנָם אֵלִיָּה יִבֵּא וְיוֹשִׁיעַ כָּל הָעוֹלָם

12.
אוֹמֵר אֲנִי לָכֶם שֶׁכְּבַר בָּא וְלֹא הִכִּירוּהוּ וְעָשׂוּ בּוֹ כִּרְצוֹנָם כֵּן יַעֲשׂוּ בַּבֶּן־הָאָדָם

13.
אָז הֵבִינוּ הַתַּלְמִידִים שֶׁבַּשֹּׁבֶל יוֹחָנָן הַמַּטְבִּיל הָיָה מִדְבָּר זֶה

14.
וַיְהִי בְּבוֹאוֹ אֶל הַחֲבוּרוֹת וְיִבֵּא לְפָנָיו אִישׁ כּוֹרֵעַ בִּרְכָּיו וַיֹּאמֶר

15.
אֲדוֹנִי חִנָּנִי וְחוֹסָה עַל בְּנִי כִּי נִבְעַת מָרוּחַ רָעָה וְחוֹלֶה מְאֹד וְחוֹרֵק אֶת־שִׁינָּיו וּמִקֶצֶף  בְּפִיו וְנוֹפֵל מִקּוֹמָתוֹ אַרְצָה וְפַּעֲמִַים נוֹפֵל בָּאֵשׁ וּפְעָמִים נוֹפֵל בַּמַּיִם

16.
וְהִבִיאוֹתָיו לְתַּלְמִידֶיךָ וְלֹא יוּכְלוּ לִרְפֵאתוֹ

17.
וַיַּעַן יֵשׁוּעַ וַיֹּאמֶר דּוֹר רַע אוֹי לָכֶם אַתֶּם הַכּוֹפֵרִים עַד־מָתַי אֶהֱיֶה עִמָּכֶם וְעַד־מָתַי אֶשָּׂא טְרַחְכֶם הֱבִיאוּהוּ אֵלַי

20.
וֶהֱבִיאוּהוּ אֵלָיו וּמִיָּד שֶׁיֵשׁוּעַ רָאָהוּ הַשָׂטָן מַכְנִיעַוּ וּמַפִּילוֹ לָאָרֶץ וְהַתְּחִיל מִתְעַפֵּר וּמִתַּקַצֵּף

21.
וְיֵשׁוּעַ שָׁאַל לַאֲבִי הַנַּעַר כַּמָּה זְמַן יֵשׁ שֶׁהַשָּׂטָן לָקְחוּ וְהָאָב הֵשִׁיבוּ מִזְּמַן פְּלוֹנִי וְהָלְאָה וְמִיַּד

22.
וְהַרְבֵּה פְּעָמִים הִפִּילוֹ בָּאֵשׁ וּבַמַּיִם בְּעִנְיַין יוּכָל לְהִשְׁמִידוֹ וְאִם אַתָּה אֲדוֹנִי בַּשׂוּם עִנְיֵין תּוּכַל לְעַזְרוֹ עָזְרֵהוֹ וְיִשָּׂא הָאִישׁ חֵן בְּעֵינָיו וְנִתְמַלֵּא רַחֲמִים עָלָיו

23.
וְאָמַר לוֹ אִם אַתָּה תּוּכַל לְהַאֲמִין כָּל דָּבָר תּוּכַל לְהַשְׁלִים לְפִי שֶׁלְּמַאֲמִין כָּל הַדְּבָרִים אֶצְלוֹ קַלִּים

24.
וּמִיָּד בָּכָה בַּצְּעָקָה אֲבִי הַנַּעַר וְאָמַר אֲדוֹנִי אֲנִי מַאֲמִין אָמְנָם עוֹזְרֵנִי לְפִי אֱמוּנָתִי

25.
וְכַאֲשֶׁר רָאָה יֵשׁוּעַ שֶׁהָעָם מִתְקַבְּצִים צָוַה לְשֹׁד וְאָמַר לוֹ חָזָק וְאֵלָם הִנְנִי מִצַּוְּךָ שֶׁתֵּצֵא מִכָּאן וּמִכָּאן וְהָלְאָה לֹא תָּשׁוּב כָּאן עוֹד

26.
וְהַשָּׂטָן יָצָא צוֹעֲק וּמַכְאִיב וְהַנַּעַר נִשְּׁאָר כַּמָּת בָּעִנְיָין שֶׁרַבִּים הָיוּ אוֹמְרִים שֶׁהוּא מֵת

27.
וְיֵשׁוּעַ לָקְחוֹ מִהַיָּד וְהֶעֱמִידוֹ וְקָם

28.
וְכַאֲשֶׁר נִכְנַס יֵשׁוּעַ לַבַּיִת

19.
אָז קָרְבוּ הַתַּלְמִידִים לִיֵשׁוּעַ בַּסֵּתֶר וְיֹאמְרוּ אֵלָיו מַדּוּעַ לֹא נוֹכֵל אֲנַחְנוּ לְהוֹצִיאוֹ

20.
וַיֹּאמֶר לָהֶם לְמִּיעוּט אֱמוּנַתְכֶם אָמֵן אֲנִי אוֹמֵר לָכֶם אִם יִהְיֶה בָּכֶם מִן הָאֱמוּנָה כְּגַרְגִּיר חַרְדָּל אִם תַּאֲמִינוּ לָהָר הַזֶּה תֹּאמְרוּ סוּרוּ וְיָסוּר וְכָל דָּבָר לֹא יְבַצֵּר מִכֶּם

21.
וְזֶה הַמִּין מִן הַשֵּׁדִים לֹא יָצָא כִּי אִם בַּתִּפְלָה וְצוֹם

22.
הֵמָּה בַּגָּלִיל וַיֹּאמֶר יֵשׁוּעַ בֶּן־הָאָדָם יִמְסֹר לְיַד בְּנֵי הָאָדָם

23.
וְיַהַרְגוּהוּ וּבַיּוֹם הַשְּׁלִישִׁי יְקוּם

24.
וְיָבוֹאוּ כְּפַר־נַחוּם מָרְתָה וִיקָרְבוּ מְקַבְּלֵי הַמֶּכֶם לְפֶּיטְרוֹשׁ וְיֹאמְרוּ אֵלָיו רַבִיכֶם אֵינוּ נוֹתֵן מֶכֶס

25.
וְיֹאמְרוּ כֵּן וְיָבוֹאוּ בַּבַּית וְהִקְדִּים יֵשׁוּעַ אֵלָיו לֵאמֹר לְפֶּיטְרוֹשׁ מָה נִרְאָה לְךָ פֵּיטְרוֹשׁ מַלְכֵי אֶרֶץ מִמִּי לוֹקֵחִים מֶכֶס מִן בְּנֵיהֶם אוֹ מִן הַנָּכְרִים

26.
וַיַּעַן אֵלָיו מִן הַנָּכְרִים וַיֹּאמֶר לָהֶם יֵשׁוּעַ אִם כֵּן הַבָּנִים חָפְשִׁים 

27.
וַיֹּאמֶר לֹא תִּהְיוּ בֶּעָבָר זֶה נִבְהָלִים וַיֹּאמֶר לְפֶּיטְרוֹשׁ לֵךְ לַיָּם וְהֻשְׁלַךְ חַכָּה וְאוֹתוֹ דָּג שֶׁתִּקַּח רִאשׁוֹנָה תִּמְצָא בְּפִיו מַטְבֵּעַ כֶּסֶף וְאוֹתוֹ תִּתֵּן בְּעֲדֵינוּ
