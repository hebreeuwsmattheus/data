1.
וְיָבוֹאוּ אֵלָיו הַחֲכָמִים וְהַפְּרוּשִׁים מֵנִסִּים אוֹתוֹ וְיִלְמְדֵם אֵיזֶה אוֹת מֵהַשָּׁמַיִם

2.
וַיַּעַן לָהֶם יֵשׁוּעַ חֲנֵפִים אַתֶּם אוֹמְרִים עֶרֶב מָחָר יוֹם צַח יִהְיֶה לְפִי שֶׁהַשָּׁמַיִם אֲדוּמִּים

3.
וּבַבָּקָר אַתֶּם אוֹמְרִים הַיּוֹם יַמְטִיר שֶׁהַשָּׁמַיִם חֲשׁוּכִים אִם כֵּן אַתֶּם יוֹדְעִים מִשְׁפָּט מָרְאִית הַשָּׁמַיִם וְאֵין אַתֶּם יוֹדְעִים מִשְׁפָּט הַזְּמַנִּים

4.
זֶרַע מִרְעִים שֶׁאַל אוֹת וְאוֹת לֹא יִנָּתֵן לָהֶם כִּי אִם אוֹת שֶׁל יוֹנָה הַנָּבִיא וְאָז נִפְרָד וְהָלַךְ לוֹ

5.
וְכַאֲשֶׁר יֵשׁוּעַ הָיָה בִּשְׂפַת הַיָּם אָמַר לְתַּלְמִידָיו שֶׁיָּכִינוּ לֶחֶם וְהוּא נִכְנַס לַסְּפִינָה עִם תַּלְמִידָיו וְתַּלְמִידָיו שָׁכְחוּ וְלֹא הִכְנִיסוּ שׁוּם לֶחֶם

6.

7.

8.
וְיֵשׁוּעַ אָמַר לָהֶם אַתֶּם מָעַטֵי הַשֵּׂכֶל חוֹשְׁבִים שֶׁאֵין לָכֶם לֶחֶם

9.
וְאֵין אַתֶּם זוֹכְרִים מֵהַחֲמְשָׁה כִּכָּרוֹת 

10.
וְאַרְבַּע אֶלֶף אִישׁ וְכַמָּה סֲאִים נִשְׁאֲרוּ

11.
וְלָכֵן תָּבִינוּ שֶׁאֵינֵי מִדְבָּר מֵהַחַלָּמִישׁ וְגַם מֵהַלְּחָמִים הַטִּבְעִיִּים אֲבָל אֲנִי אוֹמֵר לָכֶם שֶׁתִּשָּׁמֵרוּ מֵהַנַּהֶגֶת הַפְּרוּשִׁים וְהַצְּדוּקִים

12.
אָז הֵבִינוּ תַּלְמִידָיו שֶׁלֹא הָיָה מִדְבָּר מֵהוֹלֶכֶת הַלֶּחֶם רַק מֵהַנַּהֶגֶת הַפְּרוּשִׁים וְהַצְּדוּקִים

13.
וַיָצָא יֵשׁוּעַ אֶל אָרֶץ סוּרִיאָה וְאָרֶץ פִילוֹף נִקְרָא פִילִיפּוֹס וְיִשְׁאַל לְתַּלְמִידָיו לֵאמֹר מָה אוֹמְרִים בְּנֵי אָדָם בִּשְׁבִילִי

14.
וְיֹאמְרוּ אֵלָיו מֵהֶם אוֹמְרִים שֶׁהוּא יוֹחָנָן הַמַּטְבִּיל וּמֵהֶם אוֹמְרִים שֶׁהוּא אֵלִיָּהוּ וּמֵהֶם יִרְמְיָהוּ אוֹ אֶחָד מֵהַנְּבִיאִים

15.
וַיֹּאמֶר לָהֶם יֵשׁוּעַ וְאַתֶּם מָה אוֹמְרִים בִּשְׁבִילִי

16.
וַיַּעַן שִׁמְעוֹן נִקְרָא פֶּיטְרוֹשׁ וַיֹּאמֶר אַתָּה הַמָּשִׁיחַ בַּלֹעֵז קְרִישְׁטוֹס בֶּן־אֱלֹהִים חַיִּים שֶׁבְּאִתָּהּ בְּזֶה הָעוֹלָם

17.
וַיֹּאמֶר אֵלָיו יֵשׁוּעַ אַשְׁרֶיךָ שִׁמְעוֹן בַּר־יוֹנָה שֶׁבָּשָׂר וְדָם לֹא גָּלָּה לְךָ כִּי אִם אֲבִי שֶׁבַּשָּׁמַיִם

18.
וַאֲנִי אוֹמֵר לְךָ שֶׁאַתָּה אֶבֶן וַאֲנִי אֶבְנֶה עָלֶיךָ בֵּית תְּפִלָּתִי וְשַׁעֲרֵי גֵהִינֹם לֹא יוּכְלוּ נֶגְדְּךָ

19.
לְפִי שֶׁאֲנִי אַתֶּן לְךָ מַפְתְּחוֹת מַלְכוּת הַשָּׁמַיִם וְכָל אֲשֶׁר תִּקְשׁוֹר מֵאָרֶץ יִהְיֶה קָשׁוּר בַּשָּׁמַיִם וְכָל אֲשֶׁר תַּתִּיר בָּאָרֶץ יִהְיֶה מוּתָּר בַּשָּׁמַיִם

20.
אָז צִוָּה לְתַּלְמִידָיו לְבַל יֹאמְרוּ שֶׁהוּא מָּשִׁיחַ

21.
מִכָּאן וְאֵילֵךְ הִתְחִיל יֵשׁוּעַ לְגַלּוֹת לְתַּלְמִידָיו שֶׁהוּא צָרִיךְ לָלֶכֶת לִירוּשָׁלֶם וְלָשֵׂאת עָוֶל רַבִּים מֵהַכֹּהֲנִים וְזִקְנֵי הֵעֵם עַד שֶׁיַּהַרְגוּהוּ וְיוֹם הַשְּׁלִישִׁי יְקוּם

22.
וְיִקְּחוּהוּ פֶּיטְרוֹשׁ בֵּינוּ לְבֵינוּ וְהִתְחִיל לְהוֹכִיח לֵאמֹר חֲלִילָהּ לְךָ לִהְיוֹת לְךָ כֵּן אֲדוֹנִי

23.
וְיָשַׁב יֵשׁוּעַ וַיַּבֵּט אֵלָיו וַיֹּאמֶר לוֹ לֵךְ הַשָּׂטָן לֹא תַּמְרֶה בִּי שֶׁאַתָּה אֵינְךָ מַכִּיר דָּבָר הָאֵל כִּי אִם דִּבְרֵי הָאָדָם

24.
אָז דָּבָר יֵשׁוּעַ לְתַּלְמִידָיו מִי שֶׁיִּרְצֶה לְבָא אַחֲרֵי יְבַזֶּה עָצְמוּ וְיּקַּח אֶת־הַשְׁתִי וְעָרֵב רֹצֵה לוֹמַר שֶׁקָּרַב עָצְמוּ לַמִּיתָה וְיֵלֵךְ אַחֲרִי

25.
כָּל הַרוֹצֶה לְהוֹשִׁיעַ נָפְשׁוֹ יְאַבֵּד אוֹתָהּ בַּעֲדִי וְהַמְּאַבֵּד אֶת־חַיָּיו בַּעוֹלָם הַזֶה בִּשְׁבִילִי יוֹשִׁיעַ נַפְשׁוֹ לְחָיֵי הַעוֹלָם הָבֵא

26.
מָה בָּצַע לָאָדָם אִם יַרְוִיחַ אֵת כָּל הָעוֹלָם אִם נַפְשׁוֹ יְאַבֵּד לָעַד וְאֵיזֶה תְּמוּרָה טוֹבָה יַעֲשֶׂה הָאָדָם אִם בְּעַד הַדְּבָרִים הַהוֹוִים וְהַנִּפְסָדִים יִתֵּן נַפְשׁוּ לַדִּין גֵהִינֹם

27.
כִּי בֵּן הָאֵל יִבֵּא בַּכָּבֵד אָבִיו שֶׁבַּשָּׁמַיִם עִם מַלְאָכָיו לְהָשִׁיב לְכָל אִישׁ כְּפָעֳלוֹ

28.
אָמֵן אֲנִי אוֹמֵר לָכֶם שֶׁיֵּשׁ מֵהָעוֹמְדִים פֹּה שֶׁלֹּא יִטְעֲמוּ מוֹת עַד שֶׁיָּרְאוּ בֵּן אֱלֹוהַּ בָּא בְּמַלְכוּתוֹ
