1.
וַיְהִי אַחֲרֵי זֶה בְּעֵת הַהִיא וַיַּרְא הַחֲבוּרוֹת וַיַּעַל הָהָר וַיֵּשֶׁב 
וְיַקְרִיבוֹ לוֹ תַּלְמִידִיוֹ

2.
וַיִפְתַּח פִּיו וַיִדָבַר אֶליהֵם לֵאמֹר

3.
אַשְׁרֵי שְׁפַלִי-רוּחַ שֶׁלָהֶם מַלְכוּת הַשָׁמַיִם

4.
אַשְׁרֵי הַחוּכִים שֶׁיִנוּחָמוּ

5.
אַשְׁרֵי הַעֲנָוִים שֶׁהֵם יִרְשׁוּ אָרֶץ

6.
אַשְׁרֵי שֶׁרְעֵבִים וּצְמֵאִים אֶת-צֶדֶק כִּי הֵם יִשְׂבָּעוּ

7.
אַשְׁרֵי רַחֲמָנִיִם כִּי לָהֶם יְרְחָם

8.
אַשְׁרֵי זִכִּי הַלֵב וְהֵמָּה יִרְאוּ אֱלֹקִים

9.
אַשְׁרֵי רוֹדֵפִי שָׁלוֹם שֶׁבְּנֵי הָאֱלֹקִים יִּקָּרָאוּ

10.
אַשְׁרֵי הַנִרְדֶּפִים לְצֶדֶק שֶׁלָהֶם מַלְכוּת הַשָׁמַיִם

11.
אַשְׁרֵיכֶם כַּאֲשֶׁר יִרְדֹּפוּ וְיִגִּדְּפוּ אֵתְכֶם וַיִּאמְרוּ אֵלַיכֶם כָּל-רֹעַ בַּעֲדִי וַיְכַזְּבוּ

12.
שִׂישׂוּ וְשִׂמְחוּ שֶׁשְׂכַרְכֶם רַב מְאֹד בַּשָׁמָיִם שֶׁכֵּן רִדְּפוּ הָנְבִיאִים

13.
בַּעִתּ הַהיִא אָמַר יֵשׁוּעַ לְתַּלְמִידִיוֹ מֶלַח אִתַּם בַּעוֹלָם אִם הַמֶּלַח יִבָטַל טַעְמוֹ בִמָּה יוּמֶלַח וְאִינוֹ שָׁוַה כִּלּוּם אַלָא שֶׁיוּשְׁלַךְ בַּחוּץ לְהֱיוֹת מִרְמָס רְגָלִים

14.
מָאוֹר אַתֶּם בַּעוֹלָם עִיר בְּנוּיָה עַל הָהָר לֹא תּוּכַל לְהִסָּתֵר

15.
וְלֹא יַדְלִיקוֹ נֵר לְהַשִׂים אוֹתוֹ בַּמָקוֹם נִסְתָּר שֶׁלֹא תָּאִיר רַק מֵשִׂימִים אוֹתוֹ עִל הַמְּנוֹרָה לְהָאִיר לְכָל בְּנֵי הַבָּיִת

16.
כֵּן יָאִיר מִאוֹרְכֶם לִפְּנֵי כֹּל אָדָם לְהָרָאוּתֶם מַעֲשֶׂיכֶם הַטּוֹבִים הַמִשׁוּבַּחוֹת וְמִכָּבֵדוֹת לְאֲבִיכֶם שֶׁבַּשָׁמָיִם

17.
בַּעִתּ הַהיִא אָמַר יֵשׁוּעַ לְתַּלְמִידִיוֹ אַל-תְּחַשְׁבוּ שְׁבָאתִי לְהֵפֵר אֵת הַתוֹרָה אוֹ אֵת הַנְּבִיאִים אַלָא לְהַשְׁלִים

18.
בַּאֱמֶת אָנִי אוֹמַר לְכֶם כִּי עַד שֶׁיָּעְבִרוּ שָׁמַיִם וְאֶרֶץ אוֹת אַחַת וְנְקּוּדָּה אַחַת לֹא תְבַטֵּל מִהַתוֹרָה אוֹ מִהַנְבִיאִים שֶׁהַכֹּל יִתְקַיֵּים

19.
וְאֲשֶׁר יִעֲבוֹר מַאֲמָר א מִהַמִצְווֹת אִלּוּ אֲשֶׁר אֶלְמַד אֲחֵרִים בֵּן הָבַל יִקָּרֵא בְּמַלְכוּת הַשָׁמַיִם וְהַמִקַיֵּים וְהַמִלְמַד גָּדוֹל יִקָּרֵא בַּמַלְכוּת הַשָׁמַיִם

20.
בַּעִתּ הַהיִא אָמַר יֵשׁוּעַ לְתַּלְמִידִיוֹ בַּאֱמֶת אָנִי אוֹמַר לְכֶם אִם לֹא תַּגְדֵּל צִדְקַתְכֶם יוֹתֵר מִהַפְרוּשִׁים וְהַחֲכָמִים לֹא תָּבוֹאוּ בַּמַלְכוּת הַשָׁמַיִם

21.
הֲלֹא שְׁמַעְתֶּם מָה שֶׁנֶאֱמַר לְקַדְמוֹנִים לֹא תִרְצָח וְאֲשֶׁר יִּרְצַח חַיָּיב הוּא לַמִּשְׁפָּט מָוֶת

22.
וַאֲנִי אוֹמַר לְכֶם שֶׁהַמַכְעִיס לְחֶבִירוֹ חַיָּיב הוּא לְמִּשְׁפָּט וְאֲשֶׁר יִקְרָא לְאָחִיו פִּחוּת יִחַיָּיב בַּמִּשְׁפָּט בַּהַקָהָל וְאֲשֶׁר יִקָּרְאוּהוֹ שׁוֹטֶה חַיָּיב לְאֵשׁ גֵהִינֹם

23.
וְאִם תַּקְרִיב קָרְבָּנְךָ לְהַמִזְבֵּחַ וְתִזָּכִיר שֶׁהָיַה לְךָ עִם חָבֵרךְ דִּין וְהוּא מִתְּרֹעֵם מִמֶּךָ מִאֵיזֶה דָּבָר

24.
הַנַּח קָרְבָּנְיךָ שָׁם לִפְּנֵי הַמִזְבֵּח וְלְךָ לְרָצוּתוֹ קוֹדֵם וְאַחַר כְּךָ הִקְרַב קָרְבָּנִיךָ

25.
אָז אָמַר יֵשׁוּעַ לְתַּלְמִידִיוֹ רָאָה שֶׁתְּמַהֵר לִרְצוֹת שׂוֹנַאֲךָ בַּלֶכֶתךָ עִמּוֹ בַּדֶּרֶךְ פֶּן יִמַסוּר אוֹתְךָ לְשׁוֹפֵט וְזֶה הַשׁוֹפֵט יִמַסוּרךָ לְעֶבֶד לְתֵּת אוֹתְךָ לְבֵּית הַסּוֹהַר

26.
בַּאֱמֶת אָנִי אוֹמַר לְךָ לֹא תֵצֵא מִשָׁם עַד תִנָּתֶן פְּרוּטָה אַחֲרוֹנָה

27.
עוֹד אָמַר לְהֵם שְׁמַעְתֶּם מָה שֶׁנֶאֱמַר לְקַדְמוֹנִים לֹא תִנְאָף

28.
וַאֲנִי אוֹמַר לְכֶם שֶׁכָּל הַרוֹאֶה אִשָּׁה וְיַחְמוֹד אוֹתָה כְּבָר נְּאָף עִמָּה בַּלִבָּה

29.
וְאִם יְסִיתְךָ עֵינֶיךָ הַיָמִין נַקֵּר אוֹתָּהּ וְתַּשְׁלִיכָה מִמֶּךָ כִּי טוֹב לְךָ שֶׁיּאבֵד אֶחָד מִן אֵיבָרֶיךָ מֵהִשָֹׁלֵחָ כָּל-גּוּפְךָ אֶל גֵּיהִינֹם

30.
וְכֵן אִם יְסִיתְךָ יָדְךָ חַתּוּךָ אוֹתָּהּ כִּי טוֹב לְךָ שֶׁתַפְסִיד אֶחַד מִאִבָרֶיךָ מִכָּל גּוּפְךָ בַּגֵּהִינֹם

31.
עוֹד אָמַר יֵשׁוּעַ לְתַּלְמִידִיוֹ שְׁמַעְתֶּם מָה שֶׁנֶאֱמַר לְקַדְמוֹנִים שֶׁכֹּל הַעוֹזִב אִשְׁתּוֹ וְשִׁלַּח לָתֵת לָהּ גֵט כְּרִיתוּת וְבַּלֹעֵז לִיבֵילָ רִיפוּדְײו

32.
וַאֲנִי אוֹמַר לְכֶם שֶׁכָּל הַעוֹזִב אִשְׁתּוֹ יֵשׁ לוֹ לָתֵת לָהּ גֵט כְּרִיתוּת כִּי אִם עַל דָּבָר נָאוֹף הוּא הַנוֹאֵף וְהַלוֹקֵחַ אוֹתָּהּ יִנֵאַף

33.
עוֹד שְׁמַעְתֶּם מָה שֶׁנֶאֱמַר לְקַדְמוֹנִים לֹא תִשָּׁבְעוּ בַּשְׁמִי לַשָּׁקֶר וְתָּשִׁיב לְיַהְוֶה שְׁבוּעַתךָ

34.
וַאֲנִי אוֹמַר לְכֶם לְבִלְתִּי הִשָּׁבֵעַ בַּשׂוּם עִניָין לַשָּׁוְא לֹא בַּשָׁמַיִם שֶׁכֵּסֵּא הָאֱלֹקִים הִיא

35.
וְלֹא בְּאֶרֶץ שֶׁהֲדוֹם רִגְלָיוֹ הוּא לֹא בַּיְרוּשָׁלָם שֶׁעִיר אֱלֹקִים הִיא

36.
וְלֹא בַּרֹאשׁךָ שֶׁלֹא תּוּכַל לְעָשׁוֹת שֵׂעָר א לָבָן אוֹ שָׁחוֹר

37.
אֲבָל יְהִיוֹ דִּבְרֵיכֶם כֵּן כֵּן וְגַּם לֹא לֹא כָּל הַנוֹסַף עַל זֶה הוּא רַע

38.
וְעוֹד שְׁמַעתֶם מָה שֶׁנֶאֱמַר בַּתוֹרָה עַיִן תַּחַת עַיִן שֵׁן תַּחַת שֵׁן

39.
וַאֲנִי אוֹמַר לְכֶם לְבִלְתִּי שִׁלַּם רַע תַּחַת רַע אֲבָל הַמַכָּה בַּלְחִיךָ הַיָמִין הָכֵן לוֹ הַשְׂמאׁל

40.
וְאֲשֶׁר יִרְצֶה לְחֲלוֹק עַמְּךָ בַּמִשְׁפָּט וְלְגְזוֹל כֻּתֹּנִיךָ עֲזוֹב אֶלָיוֹ מְעִילךָ

41.
וְאֲשֶׁר יִשְׁאָל אוֹתךָ לְיֵלֵךְ עִמּוֹ אָלֶף פְסִיעָוֹת לֵךְ עִמּוֹ אַלְפַיִּים

42.
הַשׁוֹאֵל מִמֶּךָ תֵּן לוֹ וְהַרוֹצֶה לְלֹוֶת מִמֶּךָ אַל תִּמְנַע

43.
עוֹד אָמַר יֵשׁוּעַ לְתַּלְמִידִיוֹ שְׁמַעְתֶּם מָה שֶׁנֶאֱמַר לְקַדְמוֹנִים וְאָהַבְתָּ לְאוֹהַבךָ וְתִּשְׂנָא לְשׂוֹנֵאךָ

44.
וַאֲנִי אוֹמַר לְכֶם אֶהֱבוּ אוֹיְבֵיכֶם וְעֲשׂוּ טוֹבָה לְשׂוֹנֵאכֶם וְמִכְעִיסכֶם וְהִתְפַּלְּלוּ בַּשְׁבִיל רוֹדְפֵיכֶם וְלוֹחֲצֵיכֶם

45.
לְמַעַן תִּהְיוּ בְנֵי אֲבִיכֶם שֶׁבֵּשָׁמַיִם שֶׁמִזְרִיחַ שֶׁמֶשׁוֹ עַל טוֹבִים וְרָעִים וּמַמְטִיר עַל רְשָׁעִים וְצַדִּיקִים

46.
אִם תֶּאֱהָבוּ אוֹהַבֵיכֶם אֵיזֶה שָׂכָר לְכֶם הֲלֹא עָזִּי פָּנִים אוֹהַבִים אוֹהַבֵיהֶם

47.
וְאִם-תִּשְׁאֲלוּ בַּשָׁלוֹם אֶחֵיכֶם לְבַד מַה-תַּעֲשׂוּ עַל יָתֶר הֲלֹא גַם הַגּוֹיִם עוֹשִׂים זּאֹת

48.
הָיוֹ אַתֶּם תְמִימִים כַּאֲשֶׁר תַּם אֲבִיכֶם
