1.
בַּיּוֹם הַהוּא יָצָא יֵשׁוּעַ מִהַבָּיִת וַיֵּשֶׁב עַל־שְׂפַת  הַיָּם

2.
וְיִתְחַבָּרוּ אֵלַיו חֲבוּרוֹת עַד שֶׁנִצְטָרֵך לָבֹא בָּאֳנִיָּה וְכֹל הַחֲבוּרָה עוֹמְדֹת בַּחוּץ

3.
וַיְדַּבֵּר לָהֶם דְּבָרִים רַבִּים בִּמְשָׁלִים וַיֹּאמֶר לָהֶם אִישׁ יוֹצֵא מִבֵּיתוֹ בְּבוֹקֶר לַזְּרוֹעַ אֶת־זַרְעוֹ

4.
וּבִזְרעוֹ נָפַל מִמֶּנּוּ בַּדֶּרֶךְ וְאָכַל אוֹתוֹ הָעוֹף

5.
וּמִמֶּנּוּ נָפְלָה בָּאָבֶן שֶׁאֵין שָׁם עוֹבִי עָפָר וּבְצִמְחוֹ נִתְיַיבֵּשׁ לְפִי שֶׁאֵין שָׁם עָפָר לָרוֹב

6.
וּבַחוֹם הַשֶּׁמֶשׁ עָלָיו נִשְׂרַף וְנִתְיַיבֵּשׁ שֶׁאֵין לוֹ שׁוֹרֶשׁ

7.
וּמִמֶּנּוּ נָפַל בֵּין הַקּוֹצִים וִיגַדְּלוּ הַקוֹצִים וִיעַמְּמוּהוּ

8.
וּמִמֶּנּוּ נָפַל בָּאָרֶץ טוֹבָה וְיַעֲשֶׂה פְּרִי וּתְבוּאָה הָאֶחָד מֵאָה וְהַשֵּׁנִי שִׁשִׁים וְהַשְּׁלִישִׁי שֶׁלֹּשִׁים

9.
לְמִי אָזְנַיִם לִשְׁמוֹעַ יִשְׁמַע

10.
וְיִקְרְבוּ אֵלָיו תַּלְמִידָיו וְיֹאמְרוּ לוֹ לָמָּה תְדַבֵּר בַּמְּשָׁלִים

11.
וַיֹּאמֶר שֶׁלָּכֶם נִתָּן מַלְכוּת הַשָׁמַיִם לְהַכִּיר וְלֹא לָהֶם

12.
לְמִי שֶׁיֵּשׁ לוֹ יִנָּתֵן לוֹ עוֹד וּלְמִי שֶׁאֵין לוֹ מָה שֶׁהוּא חוֹשֵׁב שֶׁיֵשׁ לוֹ יִלָּקַח מִמֶּנּוּ

13.
לְזֶה אֲנִי מִדָבַר בַּמְּשָׁלִים שֶׁהֵם רוֹאִים וְאֵינָם רוֹאִים שׁוֹמְעִים וְאֵינָם שׁוֹמְעִים

14.
לִגְמוֹר מָה שֶׁנֶאֱמַר עַל פִּי יְשַׁעְיָה הַנָּבִיא לֵךְ וְאָמַרְתָּ לָעָם הַזֶּה שִׁמְעוּ שָׁמוֹעַ וְאַל־תָּבִינוּ וּרְאוּ רָאוֹ וְאַל־תֵּדָעוּ

15.
הַשְׁמֵן לֵב הָעָם הַזֶּה וְאָזְנָיו הַכְבֵּד וְעֵינָיו הָשַׁע פֶּן־יִרְאֶה בְעֵינָיו וּבְאָזְנָיו יִשְׁמָע וּלְבָבֹו יָבִין וָשָׁב וְרָפָא לֹו

16.
וְאַשְׁרֵי עֵינֵיכֶם שֶׁרוֹאוֹת וְאָזְנֵיכֶם שֶׁשּׁוֹמְעוֹת

17.
אָמֵן אֲנִי דּוֹבֵר לָכֶם שֶׁנְּבִיאִים רַבִּים וְצַדִּיקִים הִתְאַוּוּ לִרְאוֹת מָה שֶׁאִתָּם רוֹאִים וְלֹא רָאוּ וְלִשְׁמוֹעַ מָה שֶׁאִתָּם שׁוֹמְעִים וְלֹא שָׁמְעוּ

18.
וְאַתֶּם שָׁמֵעוּ מְשַׁל הַזּוֹרֵעַ

19.
הַזּוֹרֵעַ הוּא בֶּן־הָאָדָם וְהַזֶּרַע שֶׁנָּפַל בַּדֶּרֶךְ כָּל הַשּׁוֹמֵעַ מַלְכוּת שָׁמַיִם וְלֹא יָבִין יִבֵּא הַשָּׂטָן וְיִחְתוֹף מִלִּבּוֹ כָּל מָה שֶׁנִּזְרַע בּוֹ וְזֶהוּ הַזֶּרַע שֶׁנָּפַל עַל הַדֶּרֶךְ

20.
וַאֲשֶׁר נָפַל עַל הָאֶבֶן הוּא הַשּׁוֹמֵעַ אֶת-דָּבָר הַאֵל וִיקַבְּלָנוּ מִיָּד בַּשִּׂמְחָה

21.
וְהוּא בְּלֹא שׁוֹרֶשׁ וּבַּמְבוּכָה וּבַבָּא מְעַט צַעַר וְצָרָה לָהֶם הַשָּׂטָן מִשָׂכְחוּ מִלִּבָּם

22.
וַאֲשֶׁר נָפַל בַּקּוֹצִים זֶה הַשּׁוֹמֵעַ אֶת־הַדָּבָר וּבְחֶמְדָּתוּ לֶאֱסוֹף עוֹשֶׁר הַשָּׂטָן מִשָׁכְחוּ דָּבָר אֱלֹהִים וְלֹא יַעֲשֶׂה פְּרִי

23.
וַאֲשֶׁר נָפַל בָּאָרֶץ הַטּוֹבָה הוּא הַשּׁוֹמֵעַ אֶת־הַדָּבָר וּמֵבִין וְעוֹשֵׂה פְּרי רֹצֵה לוֹמַר מִמַּעֲשִׂים טוֹבִים וְיוֹצִיא מִן הָאֶחָד מֵאָה וּמִן הַשֵּׁנִי שִׁשִּׁים וּמִן הַשְּׁלִישִׁי שְׁלֹשִׁים הָאֶחָד מֵאָה זֶהוּ מְטַהֶרֶת הַלֵּב וּקְדוּשַּׁת הַגּוּף וּמֵהָאֶחָד שִׁשִּׁים זֶהוּ מִפְּרִישׁוֹת הָאִשָּׁה וּמֵהָאֶחָד שְׁלֹשִׁים זֶהוּ מִקְּדוּשָּׁה בַּזִּיוּוּג בַּגּוּף וּבַלֵּב

24.
וְיִשֵּׂם לִפְנֵיהָם מָשָׁל אַחֵר מַלְכוּת שָׁמַיִם דּוֹמֶה לָאִישׁ הַזּוֹרֵעַ בְּשָׂדֵהוּ זֶרַע טוֹב

25.
וַיְהִי כַּאֲשֶׁר בְּנֵי אָדָם יְשָׁנִים בָּא שׂוֹנְאוּ וִיזְרַע עַל הַחִטִּים זוּן בַּלֹעֵז בְּרִייָאגָה וַיֵלֶךְ

26.
וַיְהִי כַּאֲשֶׁר גָּדְלָה הָעֵשֶׂב לַעֲשׂוֹת פְּרִי וְרָאָה הַזּוּן

27.
וְיִקְרְבוּ עַבְדֵי בַּעַל־הַשָּׂדֶה אֵלָיו וַיֹּאמְרוּ לוֹ אֲדוֹנֵינוּ הֲלֹא־זֶרַע טוֹב זָרַעְתָּ וּמֵאַיִן הָיָה הַזּוּן

28.
אָמַר לָהֶם שׂוֹנְאַי עָשָׂה זֶה וַיֹּאמְרוּ לוֹ עֲבָדָיו נִעֲקוֹר הַזּוּן

29.
וַיֹּאמֶר לָהֶם לֹא פֶן תַּעַקְּרוּ הַחִטָּה

30.
אֶלָּא הֵנִיחוּ זֶה וְזֶה וְיִגְדַּל עַד הַקָּצִיר וּבַעַת הַקָּצִיר אָמַר לַקּוֹצְרִים לִקְטוּ אֶת־הַזּוּן רִאשׁוֹנָה וְקָשְׁרוּ אוֹתוֹ חֲבִילוֹת חֲבִילוֹת לִשְׂרוֹף וְהַחִטָּה תְּנוּ בָּאוֹצָר

31.
וְיָשֵּׂם לִפְנֵיהֶם מָשָׁל אַחֵר מַלְכוּת שָׁמַיִם דּוֹמֶה לַגַּרְגִּיר חַרְדָּל שֶׁיִּקַּח אוֹתוֹ אָדָם וַיִּזְרַעהוּ בַּשָׂדֶה

32.
וְהוּא דַּק מִכֹּל הַזֵרְעוֹנִים וּבְגַדְּלוֹ יִגְדַּל עַל כֹּל הָעֲשָׂבִים וְנַעֲשָׂה עֵץ גָּדוֹל עַד שֶׁעוֹף הַשָּׁמַיִם יַאצֶילוּ בַּעֲנָפָיו

33.
וַיְדַבֵּר לָהֶם מָשָׁל אַחֵר וַיֹּאמֶר לָהֶם מַלְכוּת שָׁמַיִם דּוֹמֶה לִשְׂאוֹר שֶׁמֵּבִיא אוֹתוֹ הָאִשָּׁה  בִּשְׁלֹשׁ סֵאִים קֶמַח וְיַחְמִיץ אֶת־כּוּלּוֹ

34.
כֹּל הַמְּשָׁלִים הַאֵלּוּ דִּבֵּר יֵשׁוּעַ לַחֲבוּרוֹת וּבְלִי מָשָׁל לֹא הָיָה דּוֹבֵר לָהֶם

35.
לְקִייֵּם מָה שֶׁנֶּאֱמַר עַל פִּי הַנָּבִיא אֶפְתְּחָה בְּמָשָׁל פִּי אַבִּיעָה חִידוֹת מִנִּי־קֶדֶם

36.
אָז נִפְרָד יֵשׁוּעַ מִן הַחֲבוּרוֹת וַיָבֹא אֶל הַבַּיִת וְיִקְרְבוּ אֵלָיו תַּלְמִידָיו וַיְבַקְשׁוּ מִמֶּנּוּ לִפְרוֹשׁ לָהֶם מָשָׁל הַזּוּן

37.
וַיַּעַן לָהֶם וַיֹּאמֶר הַזּוֹרֵעַ זֶרַע טוֹב הוּא בֶּן־הָאָדָם

38.
וְהַשָּׂדֶה הוּא הָעוֹלָם הַזֶּה וּפְּרִי הַטּוֹב הֵם הַצַּדִּיקִים וְהַזּוּן הֵם הָרְשָׁעִים

39.
וְהַשּׂוֹנֵא שֶׁזָּרַע אוֹתוֹ הוּא הַשָּׂטָן וְהַקֹּמָה אַחֲרִית הָעוֹלָם וְהַקּוֹצְרִים הֵם הַמַּלְאָכִים

40.
וְכַּאֲשֶׁר לָקְטוּ הַקּוֹצְרִים הַזּוּן לִשְׂרוֹף כֵּן יִהְיֶה בְּאַחֲרִית הַיָּמִים

41.
יִשְׁלַח בֶּן־הָאָדָם אֶת־מַלְאָכָיו לַעֲקוֹר מִמַּלְכוּתוֹ כֹּל־רָשָׁע וְכֹל־פּוֹעֲלֵי אָוֶן

42.
וַיִשְׁלִיכוּ אוֹתָם בִּמְדּוּרַת אֵשׁ וְשָׁם יִהְיֶה בְּכִי וַחֲרִיקֹת שִׁינַּיִם

43.
אָז יַזְהִירוּ הַצַּדִּיקִים כְּשֶׁמֶשׁ בְּמַלְכוּת אֲבִיהֶם לְמִי אָזְנַיִם לִשְׁמוֹעַ יִשְׁמַע

44.
עוֹד אָמַר יֵשׁוּעַ לְתַּלְמִידָיו מַלְכוּת שָׁמַיִם הִיא דּוֹמֶה לָאָדָם הַמּוֹצָא מַטְמוֹן אֲשֶׁר יַסְתִּירוּהוּ וּבְשִׂמְחַת הַמָּמוֹן יִמְכּוֹר כָּל־אֲשֶׁר־לוֹ וְיִקְנֶה הַשָּׂדֶה בַּעֲדוֹ

45.
וְעוֹד מַלְכוּת שָׁמַיִם דּוֹמֶה לָאָדָם סוֹחֵר הַמְּבַקֵּשׁ אֲבָנִים יְקָרוֹת

46.
וְכַּאֲשֶׁר יִמְצָא אַחַת טוֹבָה יִמְכּוֹר כָּל־אֲשֶׁר־לוֹ וְיִקְנֶה אוֹתָהּ

47.
מַלְכוּת שָׁמַיִם דּוֹמֶה הִיא לָרֶשֶׁת בְּתוֹךְ הַיָּם שֶׁכָּל מִינֵי דָּגִים נֶאֱסָפִים בָּהּ

48.
וְכַּאֲשֶׁר תְּמַלֵּא יוֹצִיאוּה לַחוּץ וְיוֹצְאִים הַדַּייָּגִים וּבּוֹחֲרִים הַטּוֹבִים בִּכְלֵיהֶם וְהָרָעִים מַשְׁלִיכִים חוֹצֶה

49.
כֵּן יִהְיֶה בָּאַחֲרִית הַיָּמִים יָצְאוּ הַמַּלְאָכִים וְיַבְדִּילוּ הָרְשָׁעִים מִתּוֹךְ הַצַּדִּיקִים

50.
וְיַשְׁלִיכוּ אוֹתָם בַּמְּדוּרָת אֵשׁ וְשָׁם יִהְיֶה בֶּכִי וַחֲרִיקָת שִׁינַּיִם

51.
וַיֹּאמֶר לָהֶם הֵבַנְתֶּם זֶה וַיֹּאמְרוּ כֵּן

52.
לְזֹאת כָּל חָכָם יִדְמֶה בְּמַלְכוּת שָׁמַיִם לָאָדָם אֲבִי הַטָּף הַמּוֹצִיא מֵאוֹצָרוֹ דְּבָרִים חֲדָשִׁים גַּם יְשָׁנִים

53.
וַיְהִי אַחֲרֵי זֹאת כַּאֲשֶׁר כִּלָּה יֵשׁוּעַ הַדְּבָרִים הָאֵלֶּה עָבַר מִשָּׁם

54.
וּבָא לְאַרְצוֹ וְהָיָה מְלַמֵּד לָאֲנָשִׁים בְּבָתֵּי כְּנֵסֵיוֹת וְהַפְּרוּשִׁים נִפְלַאוּ וַיֹּאמְרוּ בְּלִבָּם מֵאַיִן בָּא לְזֶה הַחָכְמָה וְכֹחַ לַעֲשׂוֹת אֵלּוּ הַפְּעוּלּוֹת

55.
אֵין זֶה בֵּן הַנֶּפַח וּמִרְיָם הֲלֹא יְדַעְתֶּם שֶׁאִמּוֹ מִרְיָם וְאֶחָיו ג' יוֹסֵף וְשִׁמְעוֹן וִיהוּדָה

56.
וְאַחְיוֹתָיו הֲלֹא יְדַעְתֶּם שֶׁכָּל אֵלּוּ עִמָּנוּ מֵאַיִן בָּא לְזֶה כָּל־אֵלֶּה

57.
וְהָיוֹ נָבוֹכִים בוֹ וַיַּעַן לָהֶם יֵשׁוּעַ אַיִן נָבִיא בְעִירוֹ שֶׁאַיִן לוֹ כָּבוֹד כִּי אִם בְּאַרְצוֹ וְעִירוֹ וְבֵיתוֹ

58.
וְלֹא רָצָה לַעֲשׂוֹת שָׁם שׁוּם אוֹת לַמִּיעוּט אֱמוּנָתָם
