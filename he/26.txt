1.
וַיְהִי כַּאֲשֶׁר כִּלָּה יֵשׁוּעַ לְדַבֵּר כָּל הַדְּבָרִים הָאֵלֶּה אָמַר לְתַלְמִידָיו

2.
הַלֹּא תֵּדְעוּ שֶׁאַחַר שֵׁנִי יָמִים יִהְיֶה הַפֶּסַח וּבֶּן־הָאָדָם יִּמָּסְר בַּיָּד הַיְּהוּדִים לַצְּלִיבָה

3.
אָז נֶאֶסְפוּ סְגָנֵי הַכֹּהֲנִים וּגְדוֹלֵי הָעָם בֶּחָצֵר נָגִיד הַכֹּהֲנִים וּשְׁמוֹ קַאיַפַשׂ

4.
וַיִוָּעֲצוּ יַחְדָּיו לִתְפּוֹשׂ אֶת־יֵשׁוּעַ בְּעָרְמָה וּלְהוֹרְגוֹ

5.
וְיֹאמְרוּ לֹא יִהְיֶה בַּחַג פָּן שָׁאוֹן יִהְיֶה בָּעָם

6.
וַיְהִי כַּאֲשֶׁר הָיָה יֵשׁוּעַ בַּכְּפָר־חֲנַנְיָה בַּבַּית שִׁימוֹן הַמְּצוֹרָע

7.
נִגְּשָׁה אֵלָיו אִשָּׁה אַחַת בַּפַּךְ מְשִׁיחָה יְקָרָה וּתִּיָצַק אוֹתוֹ עַל־רֹאשׁוֹ וְהוּא מְסִבָּה לַשֻּׁלְחָן

8.
וְיָרַע לָהֶם מְאֹד מַדּוּעַ הָאֲבַדּוֹן הַזֶּה

9.
יוּכַל לְמוֹכְלָהּ בַּמְּחִיר רָב וְלָתֵת לָעֲנִיִּים

10.
ויֵשׁוּעַ הַיּוֹדֵעַ כָּל דָּבָר לְאֵיזֶה עִנְיָן נַעֲשָׂה אָמַר לָהֶם אַתָּם מַאֲשִׁימִים אֶת־הָאִשָּׁה הַזֹּאת בֶּאֱמֶת מַעֲשֶׂה טוֹב וְנִפְלָא עָשְׂתָה עִמָּדִי

11.
כִּי הָעֲנִיִּים יִהְיוּ עִמָּכֶם תָּמִיד וַאֲנִי לֹא אֶהֱיֶה עִמָּכֶם תָּמִיד

12.
וְשֶׁמָּה זֹאת בַּגּוּפִי רוֹמֵז לִקְבוּרָתִי

13.
אָמֵן אֲנִי אוֹמֵר לָכֶם בְּכָל מָקוֹם אֲשֶׁר תִּקְרָא בְּשׂוֹרָה זוֹ לֹעז אוּוַנְגֵיל בְּכָל הָעוֹלָם יֹאמַר אֲשֶׁר עָשְׂתָה זֹאת בְּזִכְרִי

14.
אָז הָלַךְ אֶחָד מֵהַשָּׁנִים עֶשֶׂר שֶׁשְּׁמוֹ יְהוּדָה אַיסְכָרְיוֹטָא לִגְדוֹלֵי הַכֹּהֲנִים

15.
וַיֹאמֶר לָהֶם מָה תַּתְנוּ לִי וַאֲנִי אֶמְסוֹר יֵשׁוּעַ לָכֶם וְיִפְסְקוּ אִתּוֹ שְׁלֹשִׁים כָּסֶף

16.
וּמִכָּאן וְאֵילָךְ בַּקַשׁ עִנְיָן לַמַּסּוֹר אוֹתוֹ

17.
וּבַיּוֹם הָרִאשׁוֹן שֶׁל חַג הַמַּצּוֹת קָרְבוּ הַתַּלְמִידִים לִיֵשׁוּעַ לֵאמֹר אָנָה נָכִין לְךָ אֲכִילַת הַפֶּסַח

18.
וַיֹאמֶר לָהֶם לְכוּ אֶל הָעִיר לְאֵיזֶה אִישׁ שֶׁיִדְּבֶנּוּ לִבּוֹ לַעֲשׂוֹת וְאָמְרוּ לוֹ כֹּה אָמַר הָרַב זְמַנִּי קָרוֹב הוּא עִמְּךָ אֶעֱשֶׂה פֶּסַח עִם תַּלְמִידִי

19.

20.
וַיְהִי לָעֵת עֶרֶב וְהוּא יוֹשֵׁב לְשֻׁלְחָן עִם י''ב תַּלְמִידָיו

21.
כַּאֲשֶׁר הָיוּ אוֹכְלִים אָמַר לָהֶם אוֹמֵר אֲנִי לָכֶם שֶׁאֶחָד מִכֶּם יִמְסְרֵנִי

22.
וְיִתְעַצְּבוּ מְאֹד וְיֹאמְרוּ לוֹ כָּל אֶחָד לֵאמֹר אֲדוֹנִי הָאֲנִי זֶה

23.
וַיַּעַן לָהֶם הַטּוֹבֵל יָדוֹ עִמִּי בַּקְּעָרָה הוּא יִמְכְּרֵנִי וְכוּלָּם הָיוּ אוֹכְלִים בַּקְּעָרָה אַחַת לָכֵן לֹא הִכִּירוּהוּ שֶׁאִלּוּ הִכִּירוּהוּ הִשְׁמִידוּהוּ

24.
וַיֹאמֶר לָהֶם יֵשׁוּעַ אֱמֶת שֶׁבֶּן־הָאָדָם הוֹלֵךְ כְּכָתוּב בּוֹ אוֹי לָאָדָם הַהוּא אֲשֶׁר בִּשְׁבִילוֹ בֶּן־הָאָדָם יִמְסֹר טוֹב לוֹ שֶׁלֹּא נוֹלַד לָאִישׁ הַהוּא

25.
וַיַּעַן יְהוּדָה אֲשֶׁר מְכָרוֹ וַיֹאמֶר לוֹ רַבִּי הַאֲנִי זֶה וַיֹאמֶר אַתָּהּ דַּבֶּרֶת

26.
הֵמָּה אוֹכְלִים וְיִקַּח יֵשׁוּעַ אֶת־לֶחֶם וִיבָרֵךְ וִיחַלְּקֵהוּ וְיִתֵּן לַתַּלְמִידָיו וַיֹאמֶר קְחוּ וְאָכְלוּ זֶה הוּא גּוּפִי

27.
וְיִקַּח אֶת־הַכּוֹס וְיִתֵּן שְׁבָחִים לְאָבִיו וְיִתֵּן לָהֶם וַיֹאמֶר שָׁתוּ מִזֶּה כּוּלְּכֶם

28.
זֶהוּ דָּמִי מִבְּרִית חֲדָשָׁה אֲשֶׁר יִשְׁפֹּךְ בַּעֲבוּר הָרַבִּים לְכַפָּרַת עוֹנוֹת

29.
אוֹמֵר אֲנִי לָכֶם לֹא אִשְׁתָּהּ אֲנִי מִכָּאן וְאֵילָךְ מִפְּרִי הַגֶּפֶן הַזֹּאת עַד הַיּוֹם הַהוּא שֶׁאִשְׁתָּהּ אוֹתוֹ חָדָשׁ עִמָּכֶם בְּמַלְכוּת שָׁמַיִם

30.
וְיֵלְכוּ וְיָצְאוּ לָהָר הַזֶּתִים

31.
אָז אָמַר לָהֶם יֵשׁוּעַ בָּאוּ כֻּלְּכֶם הִתְעַצְּבוּ עָלַי הַלַּיְלָה שֶׁכֵּן כְּתִיב הַךְ אֶת־הָרוֹעֶה וּתְפוּצֶינָה הַצֹּאן

32.
וְאַחֲרֵי קוֹמִי מֵהַמִּיתָה אֲגַלֶּה לָכֶם בַּגָּלִיל

33.
וַיַּעַן פֶּיטְרוֹשׁ וַיֹּאמֶר לוֹ אִם כֻּלָּם יְעַצְּבוּ עָלֶיךָ אֲנִי לֹא אֶתְעַצֵּב לְעוּלָם

34.
וַיֹּאמֶר לוֹ יֵשׁוּעַ אָמֵן אֲנִי אוֹמֵר לְךָ שֶׁבְּזֶה הַלַּיְלָה קוֹדֶם קְרִיאַת הַגֶּבֶר תִּכְפוֹר בִּי שְׁלֹשָׁה פְּעָמִים

35.
וַיֹּאמֶר לוֹ פֶּיטְרוֹשׁ אִם יִתָּכֵן לִי לָמוּת עִמְּךָ לֹא אֶכְפּוֹר בְּךָ וְכָזֶה אָמְרוּ לוֹ כָּל הַתַּלְמִידִים

36.
אָז בָּא יֵשׁוּעַ עִמָּהֶם לַכְּפָר גֵּיא־שְׁמֵנִים וַיֹּאמֶר שָׁבוּ נָא עַד שֶׁאֵלֵךְ לְשָׁם וְאֶתְפַּלֵּל

37.
וְיִקַּח אֶת־פֶּיטְרוֹשׁ וְאֶת־שֵׁנִי בְּנֵי־זַבְדִאֵל וְהִתְחִיל לְהִתְעַצֵּב וְלִהְיוֹת זָעַף

38.
אָז אָמַר לָהֶם נַפְשִׁי מִתְעַצֶּבֶת עַד מוֹת סִמְכוּנִי וְשָׁמְרוּ עִמִּי

39.
וְיֵלֵךְ לְאַט לְאַט מְעַט וְיִפּוֹל עַל פָּנָיו וְיִתְפַּלֵּל וַיֹּאמֶר אֲבִי אִם יוּכַל לִהְיוֹת הָסֵר נָא מִמֶּנִּי הַכּוֹס הַזֶּה אָמְרָם לֹא כְּמוֹ שֶׁאֲנִי רוֹצֶה יִהְיֶה אֶלָּא כִּרְצוֹנְךָ

40.
וְיִבֵּא אֶל הַתַּלְמִידִים וְיִמְצְאֵם יְשֵׁנִים וַיֹּאמֶר לְפֶּיטְרוֹשׁ כָּךְ הַאִינֶךָ יָכוֹל לִשְׁמוֹר עִמָּדִי שָׁעָה אַחַת

41.
שָׁמְרוּ וְהִתְפַּלְּלוּ פָּן תְּבִאוּ בַּנִּסָּיוֹן שֶׁהָאֱמֶת שֶׁהָרוּחַ נָכוֹן לִילָךְ לְבּוֹרֵאוֹ אַךְ הַבָּשָׂר חַלָּשׁ וְחוֹלֶה

42.
וְיֵלֵךְ שֵׁנִית לְהִתְפַּלֵּל לֵאמוֹר אִם-לֹא תּוּכַל לְהָסִיר הַכּוֹס הַזֶּה אֶלָּא שֶׁאֶשְׁתֵּהוּ יִהְיֶה עָשׂוּי כִּרְצוֹנְךָ

43.
וְיָשַׁב אַחֲרֵי־כֵּן וְיִמְצְאֵם יְשֵׁנִים שֶׁהָיוּ עֵינֵיהֶם כְּבֵדִים

44.
וְיַעֲזוֹב אוֹתָם וְיֵלֵךְ לְהִתְפַּלֵּל פַּעַם שְׁלִישִׁית כִּדְבָרִים הָרִאשׁוֹנִים

45.
אָז בָּא יֵשׁוּעַ לַגָּלִיל לְתַלְמִידָיו וַיֹּאמֶר לָהֶם שְׁנוּ וְנוּחוּ הִנֵּה הַקְּרָב הָעֵת וּבֶּן־הָאָדָם יִמְסֹר בַּיָּד הַחֲטָאִים

46.
קוּמוּ וְנֵלֵךְ שֶׁהִנּוֹ קָרוֹב מִי שֶׁיִּמְסְרֵנוּ

47.
עוֹדֶנּוּ מִדְבָּר וְהִנֵּה יְהוּדָה אַסְכָרְיוֹטָא אֶחָד מִי''ב תַּלְמִידָיו בָּא וְעִמּוֹ חֲבוּרָה אַחַת רָבָה בַּחֲרָבוֹת וּבַשּׁוֹטִים שְׁלוּחִים מֵאֵת גְּדוֹלֵי הַכֹּהֲנִים וְשָׂרֵי הָעָם

48.
וַאֲשֶׁר מְסָרוּהוּ נָתַן לָהֶם אוֹת הַאִישׁ אֲשֶׁר אֶשָּׁקְנּוּ הוּא הוּא וּתְפָשׂוּהוּ

49.
וּמִיָּד נִגַּשׁ אֶל יֵשׁוּעַ וַיֹּאמֶר לוֹ שָׁלוֹם עָלֶיךָ רַבִּי וְיַשְׁקֵהוּ

50.
וַיֹּאמֶר אֵלָיו יֵשׁוּעַ אֲהוּבִי מָה עָשִׂיתָ וְיִקְרְבוּ וְיִשְׁלְחוּ יָדָם בּוֹ וְיִתְפְשוּהוּ

51.
וְהִנֵּה אֶחָד מֵאֲשֶׁר הָיָה עִם יֵשׁוּעַ נָטָה יָדוֹ וְיִשְׁלוֹף חַרְבּוֹ וַיַּךְ עֶבֶד אֶחָד מְעַבְּדֵי הַכֹּהֲנִים וְיִכְרוֹת אָזְנוֹ

52.
וַיֹּאמֶר אֵלָיו יֵשׁוּעַ הָשֵׁב חַרְבְּךָ אֶל נְדָנָהּ שֶׁהַשְׁלֵפִים חֶרֶב בַּחֶרֶב יִפּוֹלוּ

53.
הֲלֹּא תָּבִין שֶּׁאוּכָל לִפְגוּעַ בְּאוֹיְבִי וְאָכֵן לִי עַתָּה יָתַר מִי''ב לִגְיוֹנוֹת  שֶׁל מַלְאָכִים

54.
וְאֵיךְ יִמַּלְאוּן הַכְּתוּבִים שֶׁכֵּן רָאוּי לַעֲשׂוֹת

55.
אַחֵר אָמַר יֵשׁוּעַ לַחֲבוּרָה כְּמוֹ אִם הָיִינוּ גַּנָּבִים בְּאַתָּם לָקַחַת אוֹתִי בַּחֲרָבוֹת וּבַשּׁוֹטִים וְהַלֹּא בְּכָל יוֹם הָיִיתִי עִמָּכֶם בַּמִּקְדָּשׁ מְלַמֶּדְכֶם וְלֹא עַכְבַתוּנִי

56.
אָמְנָם כָּל זֶה נַעֲשָׂה יַעַן יִמַּלְאוּן הַכְּתוּבִים מֵהַנְּבִיאִים אָז כָּל תַּלְמִידָיו הֱנִיחוּהוּ וּבָרְחוּ

57.
וְהֵם הוֹלִיכוּ לְיֵשׁוּעַ לַבֵּית קַאיַפַשׁ גָּדוֹל הַכֹּהֲנִים וְאָז כָּל הַסּוֹפְרִים וְהַפְּרוּשִׁים נִקְהֲלוּ

58.
וְפֶּיטְרוֹשׁ הָיָה הוֹלֵךְ אַחֲרָיו מֵרָחוֹק עַד בֵּית גָּדוֹל הַכֹּהֲנִים וְנִכְנַס לַבֵּית וְיָשַׁב לוֹ אֵצֶל הַאוּמָנִים עַד יִרְאָה הַתַּכְלִית

59.
וּגְדוֹלֵי הַכֹּהֲנִים וְהַפְּרוּשִׁים הָיוּ רוֹצִים לִמְצוֹא עֲדֵי שֶׁקֶר נֶגֶד יֵשׁוּעַ יַעַן יְמִיתוּהוּ וְלֹא הָיוּ מוֹצָאִים

60.
וְאֶחָד אֲשֶׁר הֵכִינוּ הַרְבֵּה עֲדֵי שֶׁקֶר נֶגֶד יֵשׁוּעַ

61.
לַסּוֹף בָּאוּ שֵׁנִי עֵדִים שְׁקָרִים וְיֹאמְרוּ זֶה אָמַר יֵשׁ לִי יְכוֹלֶת לְהַשְׁחִית מִקְדָּשׁ הָאֵל וְאַחֵד שְׁלֹשָׁה יָמִים לְתַקֵּן אוֹתוֹ

62.
וְגָדוֹל הַכֹּהֲנִים קָם וַיֹּאמֶר לוֹ אֵינְךָ עוֹנָה דָּבָר נֶגֶד הָעֵדוֹת שֶׁאִלוּ מְעֵידִים נֶגְדְּךָ

63.
וִיֵשׁוּעַ לֹא עָנָה דָּבָר וְגָדוֹל הַכֹּהֲנִים אָמַר לוֹ אֲנִי מַשְׁבִּיעֶךָ בָּאֵל חַי שֶׁתֹּאמַר לָנוּ אִם־אַתָּהּ מָשִׁיחַ בֶּן־הָאֵל

64.
וַיַּעַן לוֹ יֵשׁוּעַ אַתָּה אוֹמֵר וְעוֹד אֲנִי אוֹמֵר לָכֶם עֲדַיִן תִּרְאוּ בֶּן־הָאֵל יוֹשֵׁב לַיָּמִין גְּבוּרַת הָאֵל בָּא בָּעֲבֵי שְׁחָקִים

65.
אָז גָּדוֹל הַכֹּהֲנִים קָרַע בְּגָדָיו וְאָמַר זֶה בֵּרַךְ  אֱלֹהִים וּמָה לָנוּ צוֹרֶךְ לָעֵדִים אֲחֵרִים וְהִנֵּה כּוּלְּכֶם שְׁמַעְתֶּם אֵיךְ בֵּרַךְ אֱלֹהִים

66.
מָה יִרְאָה לָכֶם שֶׁיִּתָּכֵן לַעֲשׂוֹת וְהֵם עָנוּ שֶׁחַיָּיב מִיתָה

67.
וְאָז רָקְקוּ בְּפָנָיו וְהִלְקוּהוּ עַל שִׁכְמוֹ וַאֲחֵרִים טִפֵּחַוּ לוֹ בְּפָנָיו

68.
אוֹמְרִים אָמוּר לָנוּ הַמָּשִׁיחַ מִי הַכָּךָ

69.
וְפֶּיטְרוֹשׁ הָיָה עוֹמֵד לַפֶּתַח הֶחָצֵר וְנִגְּשָׁה אֵלָיו שִׁפְחָה אַחַת וְאוֹמֶרֶת לוֹ וְהַלֹּא אַתָּהּ עִם יֵשׁוּעַ הַגַּלִילִי הָיִיתָ עוֹמֵד

70.
וְפֶּיטְרוֹשׁ כָּחַשׁ לָהּ בִּפְנֵי הַכֹּל וְאָמַר לָהּ אִשָּׁה אֵינִי יוֹדֵעַ מָה אֶת־אוֹמֶרֶת

71.
וְכַאֲשֶׁר עָבַר הַשַׁעַר רָאָהוּ שִׁפְחָה אַחֶרֶת וְאָמַרְתָּ לְעוֹמְדִים שָׁם זֶה הָאִישׁ הָיָה עוֹמֵד עִם יֵשׁוּעַ בְּנָאצְרַת

72.
וּפַעַם אַחֶרֶת כָּחַשׁ יֵשׁוּעַ בַּשְּׁבוּעָה שֶׁלֹּא הִכִּירוּ

73.
וְאַחַר כֵּן לַזְּמַן מְעַט נְגַשֵּׁוּ אֶל פֶּיטְרוֹשׁ הָעוֹמְדִים בֶּחָצֵר וְיֹאמְרוּ לוֹ אַתָּהּ הוּא מֵחֲבוּרַת זֶה הַנָּבִיא שֶׁמִּדְבָּרְךָ נֵכָר שֶׁאַתָּהּ מֵהֶם

74.
אָז הִתְחִיל לַכְּפוֹר וּלִישָׁבֵע שֶׁבָּשּׂוּם זְמַן לֹא הִכִּירוּ וּמִיָּד קָרָא הַתַּרְנְגוֹל

75.
וּפֶּיטְרוֹשׁ נִזְכַּר מֵאֲשֶׁר אָמַר לוֹ יֵשׁוּעַ שֶׁקּוֹדֶם קְרִיאַת הַגֶּבֶר יִכְפוֹר בּוֹ ג' פְּעָמִים וְאָז יָצָא לַחוּץ וּבָכָה בִּמְרִירוּת נָפְשׁוֹ
