1.
וַיְהִי בַּבָּקָר כָּל גְּדוֹלֵי הַחֲכָמִים וְהַקַּדְמוֹנִים לָקְחוּ עֵצָה נֶגֶד יֵשׁוּעַ שֶׁמִּכָּל וְכָל יַהַרְגוּהוּ

2.
וְקָשׁוּר הוֹלִיכוּהוּ לְבֵית פוטץ פִּילָאט שֶׁהָיָה גִּזְבָּר

3.
וְאָז כַּאֲשֶׁר רָאָה יְהוּדָה אַסְכָרְיוֹטָא שֶׁהָיָה נִדּוֹן הִתְחִיל לָשׁוּב בַּתְּשׁוּבָה וְחָזַר הַשִּׁלֵּשִׁים דִּינָרִים לַגָּדוֹל הַכֹּהֲנִים וּלְזִקְנֵי הֵעֵם

4.
אָמַר אֲנִי חָטָאתִי שֶׁשָּׁפַכְתִּי דָּם נָקִי וְהֵם אָמְרוּ לוֹ מָה־לָנוּ אַתָּה תִּרְאֶה

5.
וְזָרַק הַמָּעוֹת בַּמִּקְדָּשׁ וְהָלַךְ לוֹ וְלָקַח חֶבֶל אַחַת וְתָלָה עָצְמוּ

6.
וּגְדוֹלֵי הַכֹּהֲנִים כַּאֲשֶׁר לָקְחוּ הַמָּעוֹת אָמְרוּ לֹא יִתָּכֵן שֶׁנִשִׂים אֵלּוּ הַמָּעוֹת בַּמִּקְדָּשׁ שֶׁדְּמֵי דָּם הֵם שֶׁנָּתְנוּ בַּעַד דְּמֵי יֵשׁוּעַ

7.
וְיִוֹעֲצוּ וְיַתְנוּ אוֹתָם בַּעַד שָׂדֶה אָדָם יוֹצֵר חֶרֶס בַּעַד שֶׁיִקְבְּרוּ שָׁם הַגֵּרִים

8.
וְלָכֵן נִקְרָא אוֹתוֹ שָׂדֶה אָהָל־דָּם עַד הַיּוֹם הַזֶּה

9.
אָז נִשְׁלַם מַאֲמַר זְכַרְיָה הַנָּבִיא וָאוֹמַר לָהֶם אִם־טוֹב בְּעֵינֵיכֶם הָבוּ שְׂכָרִי וְאִם־לֹא חֲדָלוּ וַיִּשְׁקְלוּ אֶת־שְׂכָרִי שְׁלֹשִׁים כָּסֶף וַיֹּאמֶר יְהֹוָה אֵלַי הַשְׁלִיכֵהוּ אֶל־הַיּוֹצֵר אָדָם הַיְקָר אֲשֶׁר יָקַרְתִּי מֵהֶם

10.
וְזֶהוּ מֵהָאָדָם הַיּוֹצֵר חֶרֶס כַּאֲשֶׁר אֲדוֹנָי צִוָּה

11.
ויֵשׁוּעַ הָיָה עוֹמֵד לִפְנֵי פִּילָאט וְשָׁאַל לוֹ הֲאַתָּה הוּא מֶלֶךְ הַיְּהוּדִים וְיֵשׁוּעַ אָמַר אַתָּהּ אוֹמֵר

12.
וְכַאֲשֶׁר יֵשׁוּעַ הָיָה רוּדַּף בַּעַד גְּדוֹלֵי הַכֹּהֲנִים וְזִקְנֵי הֵעֵם לַשּׁוּם דָּבָר שֶׁהָיוּ אוֹמְרִים עָלָיו לֹא הָיָה עוֹנָה

13.
וְפִּילָאט אָמַר לוֹ אֵינְךָ רוֹאֶה כַּמָּה עֵדָיוֹת יֵשׁ נֶגְדְּךָ

14.
וְיֵשׁוּעַ לֹא עָנָה אֵלָיו דָּבָר וְפִּילָאט הָיָה נִפְלָא מִזֶּה מְאֹד

15.
וּבַיּוֹם הַחַג הַנִּכְבָּד שֶׁל פֶּסַח הָיָה מִנְהָגָם שֶׁגִּזְבָּר הָעִיר הָיָה לָתֵת לָעָם אָסוּר אֶחָד מֵהָאֲסוּרִים אוֹתוֹ אֲשֶׁר יִרְצוּ

16.
וּבַיָּד פִּילָאט הָיָה חָבוֹשׁ אֶחָד שֶׁהָיָה כִּמְעַט שׁוֹטֶה שְׁמוֹ בַּרַאבַּאשׂ וְנִלְקַח עַל רָצְחָה וְשָׂם אוֹתוֹ בַּבּוֹר

17.
וְכַאֲשֶׁר נֶאֶסְפוּ אָמַר לָהֶם פִּילָאט אֵיזֶה מֵאֵלוּ תִּרְצוּ שֶׁאָנִיחַ בַּרַאבַּאשׂ אוֹ יֵשׁוּעַ שֶׁנִּקְרָא מָשִׁיחַ

18.
לְפִי שֶׁפִּילָאט הָיָה יוֹדֵעַ שֶׁעַל שִׂנְאַת חִנָּם נִלְקַח

19.
וּבְעוֹדוֹ יוֹשֵׁב בַּכִּסֵּא אִשְׁתּוֹ שָׁלְחָה לוֹ שָׁלִיחַ לֵאמֹר אֲחִילֶה אֲנִי מִמְּךָ שֶׁבַּשּׂוּם עִנְיָן לֹא תֹּאמַר דָּבָר כְּנֶגֶד אוֹתוֹ צַדִּיק שֶׁבְּזֹאת הַלַּיְלָה סָבַלְתִּי עִנְיָנִים רַבִּים בַּמַּרְאֶה בַּעֲדוֹ

20.
וּגְדוֹלֵי הַכֹּהֲנִים וְזִקְנֵי הַדָּת הִקְהִילוּ לָעָם שֶׁיִשְׁאֲלוּ אֶת־בַּרַבַּאשׂ וְשֶׁיֵשׁוּעַ יָמִית

21.
וַיַּעַן לָהֶם פִּילָאט אֵיזֶה מֵהֶם תִּרְצוּ שֶׁנָּנִיחַ וְהֵם אָמְרוּ בַּרַבַּאשׂ

22.
וַיֹּאמֶר לָהֶם פִּילָאטוֹשׂ אִם כֵּן מָה אֶעֱשֶׂה מִיֵשׁוּעַ הַנִּקְרָא מָשִׁיחַ וְכוּלָּם עָנוּ שֶׁיִּתְלֶה

23.
וּפִּילָאטוֹשׂ אָמַר לָהֶם אֵיזוּ רָעָה עָשָׂה וְהֵם בַּחוֹזֶק הָיוּ זוֹעֲקִים יִתְלוֹהוֹ יִתְלוֹהוֹ יִתְלוֹהוֹ

24.
וּפִּילָאטוֹשׂ בִּרְאוֹתוֹ שֶׁלֹּא הָיָה תְּקוּמָה וְלֹא יָכוֹל לְהִשְׁלִים שׂוּם דָּבָר עִמָּהֶם קּוֹדֶם שֶׁיְּקוּם בַּעַד זֶה קְטָטָה גְּדוֹלָה בָּעָם לָקַח מַיִם וְרָחַץ יָדָיו בִּפְנֵי הָעָם וְאָמַר אֲנִי נָקִי מִהָדָּם שָׁמְרוּ לָכֶם מָה תַּעֲשׂוּ

25.
וְעָנוּ כָּל הָעָם וְאָמְרוּ דָּמוֹ יִהְיֶה עָלֵינוּ וְעַל זָרַעֵינוּ

26.
וְאָז הֵנִיחַ לָהֶם בַּרַאבַּאשׂ וּמָסַר לָהֶם יֵשׁוּעַ לָקוּי וּמֵעוֹנָה שֶׁיִתְלוֹהוֹ

27.
אָז פָּרָשֵׁי הֶחָצֵר לָקְחוּ לִיֵשׁוּעַ בַּמִשְׁמָר וְיִקָּהֲלוּ בִּפְנֵי קָהָל רָב מֵעַמִּים רַבִּים

28.
וְיַלְבִּישׁוּהוּ לִיֵשׁוּעַ בִּגְדֵי מֶשִׁי וְיַעַטְפוּהוּ מְעִיל מֶשִׁי יָרוֹק

29.
וְעָשׂוּ עֲטָרָה מִקּוֹצִים וַיְשִׂימֵהוּ עַל־רֹאשׁוֹ וּשֵׂמוֹ לוֹ קָנָה אַחַת בַּיָּדוֹ הַיְּמָנִית וְכוֹרְעִים הָיוּ מַלְעִיגִים מִמֶּנּוּ שָׁלוֹם עָלֶיךָ מֶלֶךְ הַיְּהוּדִים

30.
וְרוֹקְקִים לוֹ בְּפָנָיו וְהָיוּ לוֹקְחִים הַקְנָה וּמַכִּים בְּרֹאשׁוֹ

31.
וְכַאֲשֶׁר הִלְעִיגוּ מִמֶּנּוּ מְאֹד הִפְשִׁיטוּ מִמֶּנּוּ הַמְּעִיל וְהִלְבִּישׁוּהוּ מַלְבּוּשָׁיו וְצוּוּ לִתְלוּתוֹ

32.
וְעוֹדָם יוֹצְאִים מֵהָעִיר פָּגְעוּ בָּאִישׁ שֶׁשְׁמוֹ שִׁמְעוֹן הַכְּנַעֲנִי וַאֲנָסוּהוּ שֶׁיּוֹלִיךְ הַצְּלִיבָה רֹצֵה לוֹמַר הַשְּׁתִי וְעָרֵב

33.
וּבָאוּ לְמָקוֹם נִקְרָא גוֹלְגוֹטַא הוּא הַר קָאלוּוַארִי

34.
וְנָתְנוּ לוֹ יַיִן מִזּוּג בַּמָּרָה וְכַאֲשֶׁר הִתְחִיל לִשְׁתּוֹת הִרְגִּישׁוּ וְלֹא יִרְצֶה לִשְׁתּוֹת

35.
וְאַחֵר כַּאֲשֶׁר שֵׂמוּהוּ בַּצְּלִיבָה חֶלְקוּ בִּגְדָיו בַּגּוֹרָל

36.

37.
וְאַחֵר הֵנִיחוּ לוֹ עַל־רֹאשׁוֹ מִכְתָּב אֶחָד שֶׁהָיָה אוֹמֵר זֶה יֵשׁוּעַ נָזָארַת מֶלֶךְ יִשְׂרָאֵל

38.
אָז נִתְלוּ עִמּוֹ שֵׁנִי גַּנָּבִים הָאֶחָד לִימִינוֹ וְהָאֶחָד לִשְׂמֹאלוֹ

39.
וְהָעוֹבְרִים הָיוּ מַלְעִיגִים מִמֶּנּוּ וּמְנִיעִים רֹאשָׁם

40.
וְאוֹמְרִים רְאֵה אֲשֶׁר חָרְבַּת מִקְדָּשׁ הָאֵל וּבְעוֹד שְׁלֹשָׁה יָמִים תִּבְנֶהוֹ תּוֹשִׁיעַ עַצְמְךָ וְאִם אַתָּה בֶּן־הָאֵל רַד מִן הַצְּלִיבָה

41.
וּגְדוֹלֵי הַכֹּהֲנִים וְזִקְנֵי הֵעֵם הָיוּ מַלְעִיגִים מִמֶּנּוּ וְאוֹמְרִים

42.
הָאֲחֵרִים הוֹשִׁיעַ וְעָצְמוֹ לֹא יוּכַל לְהוֹשִׁיעַ אִם מֶלֶךְ יִשְׂרָאֵל הוּא יָרַד מִן הָעֵץ וְנַאֲמִין בוֹ

43.
כִּי הוּא נִשְׁעַן בָּאֵל יוֹשִׁיעֵהוּ עַתָּה אִם יִרְצֶה שֶׁהוּא אָמַר שֶׁהוּא בֶּן־הָאֱלֹהִים

44.
וְאוֹתָם הַדְּבָרִים עַצְמָם אָמְרוּ לוֹ הַגַּנָּבִים שֶׁהָיוּ נִתְלִים עִמּוֹ

45.
וְלַשָּׁעָה שִׁשִּׁית נַעֲשׂוּ חֲשׁוּכוֹת בְּכָל הָעוֹלָם וְעָמְדוּ עַד שָׁעָה תְּשִׁיעִית

46.
יֵשׁוּעַ צָעַק בַּקּוֹל גָּדוֹל אוֹמֵר בַּלָּשׁוֹן הַקוֹדֶשׁ אֵלִי אֵלִי לָמָה עֲזַבְתָּנִי

47.
וְאֶחָד מֵהָעוֹמְדִים שָׁם אָמַר זֶה קוֹרֵא לְאֵלִיָה

48.
וּמִיָּד לָקַח אִסְפּוֹג וּמַלְּאֵהוּ חוֹמֶץ וְנָתַן לוֹ לִשְׁתּוֹת

49.
וְהָאֲחֵרִים הָיוּ אוֹמְרִים נִרְאָה אִם יִבֵּא אֵלִיָה וְיוֹשִׁיעֵהוּ

50.
וְיֵשׁוּעַ צָעַק פַּעַם אַחֶרֶת בַּקּוֹל גָּדוֹל וְשָׁלַח  נִשְׁמָתוֹ לְאָבִיו

51.
וּמּיָּד נִקְרַע פָּרֹכֶת הַמִּקְדָּשׁ לִשְׁנֵי קְרָעִים מִלְּמַעְלָה לְמַטָּה וְרַעֲשָׁהּ הָאָרֶץ וְנִשְׁתַּבְּרוּ הָאֲבָנִים

52.
וְהַקְּבָרִים נִפְתְּחוּ וְרַבִּים מִישֵׁינֵי אַדְמַת עָפָר קָמוּ

53.
וְיָצְאוּ מִקְּבוּרָתָם וְאַחַר שֶׁחַיוּ בָּאוּ בָּעִיר הַקָּדֵשׁ וְנִגְלוּ לָרַבִּים

54.
וְשָׂר הַמֵּאָה וְהָעוֹמְדִים עִמּוֹ לִשְׁמוֹר יֵשׁוּעַ רָאוּ הַרְעָשַׁת הָאָרֶץ וְהַדְּבָרִים שֶׁנַּעֲשׂוּ וְיִפְחֲדוּ מְאֹד וְאוֹמְרִים בֶּאֱמֶת זֶה הָיָה בֶּן־הָאֱלֹוהַּ

55.
וְהָיוּ שָׁם נָשִׁים רַבּוֹת שֶׁהָיוּ עוֹמְדוֹת מֵרָחוֹק מֵאוֹתָן אֲשֶׁר שִׁמְשׁוֹ לִיֵשׁוּעַ מֵהַגָּלִיל עַד־כֹּה

56.
וּבִכְלָלָן הָיְתָה מַארִיאַה מַגְדַלֵינַה וּמִרְיָם אֵם יַעֲקֹב וְיוֹסֵף וְאֵם בְּנֵי זַבְדִאֵל

57.
וְלָעֵת עָרֵב בָּא אָדָם עָשִׁיר שֶׁהָיָה מִכַרְנַאסִיאַה שְׁמוֹ יוֹסֵף וְהָיָה תַּלְמִיד מִיֵשׁוּעַ

58.
וְהָלַךְ לְפִּילָאט וְשָׁאַל לוֹ הַגּוּף מִיֵשׁוּעַ וְפִּלָאט צִוָּה שֶׁיַּתְנוּהוּ לוֹ

59.
וְיוֹסֵף לָקְחוּ וְכָרְכוּ בִּבֶּגֶד מֶשִׁי חָשׁוּב מְאֹד

60.
וְשָׂם אוֹתוֹ בְּקִבְרוֹ שֶׁהָיָה נֶחְצַב חָדָשׁ מֵאֶבֶן וְשָׂם אֶבֶן גְּדוֹלָה עַל פִּי הַקֶּבֶר

61.

62.
וּמִמָּחֳרָת יוֹם הַפֶּסַח גְּדוֹלֵי הַכֹּהֲנִים וְהַפְּרוּשִׁים בָּאוּ לְפִּילָאט

63.
וְאָמְרוּ לוֹ אֲדוֹנֵנוּ אָנוּ מַזְכִּירִים שֶׁזֶּה הַשַּׁקְרָן הָיָה אוֹמֵר בְּעוֹדוֹ בַּחַיִּים שְׁלְקֵּץ שְׁלֹשָׁה יָמִים יַעֲמוֹד וְיִחְיֶה

64.
וְלָכֵן צִוָּה לִשְׁמוֹר קִבְרוֹ עַד יוֹם הַשְּׁלִישִׁי שֶׁבַּאוּלַי אֵיזֶה מִתַּלְמִידָיו יִבֵּא וְיִגְנוֹב אוֹתוֹ וְאַחֵר יֹאמְרוּ לָעָם שֶׁעָמַד מֵהַמָּוֶת וְאִם זֶה יַעֲשׂוּ גָּדַל יִהְיֶה עָוֹן הָאַחֲרוֹן מִן הָרִאשׁוֹן

65.
וְפִּילָאט אָמַר לָהֶם בַּקְּשׁוּ שׁוֹמְרִים וְשָׁמְרוּ הַיּוֹתֵר טוֹב שֶׁתּוּכְלוּ

66.
וְהֵם שָׁלְמוּ בִּנְיָן הַקֶבֶר וַהֲתָמוּהוּ וְהֵנִיחוּ שָׂם שׁוֹמְרִים
