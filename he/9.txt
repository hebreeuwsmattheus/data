1.
וְאָז יֵשׁוּעַ בָּא בָּאֳנִיָּה וְיָשׁוּטוּ וְיָשׁוּבוּ לְעִירוֹ

2.
וְיִקְרְבוּ לְפָנָיו חוֹלֶה אֶחָד מִכִּוּוּץ בְּלֹעֵז פַּרַלֵאטִיקוֹ מוּשְׁכָּב עַל מִטָּתוֹ וְיָרֵא יֵשׁוּעַ אֱמוּנָתָם וַיֹּאמֶר לְחוֹלֶה הִתַחְזֵק בְּנִי בֶּאֱמוּנַת הָאֵל כִּי נִמְחֲלוּ עוֹנוֹתֶיךָ

3.
וּקְצָת הַחֲכָמִים אוֹמְרִים בְּלִבָּם זֶהוּ מְגַדֵּף

4.
וְיָרֵא יֵשׁוּעַ מַחְשָׁבוֹתָם וַיֹּאמֶר אֲלֵיהֶם לָמָּה תְּחַשְׁבוּ רָעָה בִּלְבַבְכֶם

5.
זֶהוּ קַל לֵאמֹר נִמְחַל עוֹנֶיךָ אוֹ קוּם וְלֵךְ

6.
רַק לְהוֹדִיעֲכֶם שֶׁבֶּן־אָדָם יָכוֹל אָדָם לַמָּחוֹל עוֹנוֹת בָּאָרֶץ אָז אָמַר לְחוֹלֶה קוּם וְקַח מִיטָּתְךָ וְלֵךְ

7.
וְיִקֹּם וְיֵלֵךְ אֶל בֵּיתוֹ

8.
וְיָרְאוּ הַחֲבוּרוֹת וְיָרְאוּ מְאֹד וִיהַלְּלוּ לָאֵל אֲשֶׁר נָתַן יָכוֹלְתְּ לִבְנֵי אָדָם לַעֲשׂוֹת כָּאֵלֶּה

9.
וַיְהִי כַּאֲשֶׁר עָבַר יֵשׁוּעַ מִשָּׁם וְיָרֵא אִישׁ אֶחָד יוֹשֵׁב עַל שֻׁלְחָן הַחֲלוֹף מַתִּתְיָה שְׁמוֹ בְּלֹעֵז מָאטיֵאוֹ וַיֹּאמֶר לוֹ לְךָ אַחֲרִי וְיִקֹּם וְיֵלֵךְ אַחֲרָיו

10.
וְיוֹלִיכוּהוֹ לְבֵיתוֹ לְאָכוּל וַיְהִי בַּעַת אָכְלוּ וְהִנֵּה פָּרִיצִים רַבִּים וּרְשָׁעִים בַּשֻּׁלְחָן וְסוֹעֲדִים עִם יֵשׁוּעַ וְתַלְמִידָיו

11.
וְיָרְאוּ הַפְּרוּשִׁים וַיֹּאמֵרוּ לְתַלְמִידָיו לָמָּה רַבְּכֶם יוֹשֵׁב וְאוֹכֵל עִם הַפָּרִיצִים וְהַרְשָׁעִים

12.
וְיִשְׁמָע יֵשׁוּעַ וַיֹּאמֶר הַבְּרִיאִים אֵינָם צְרִיכִים לִרְפוּאָה כִּי אִם הַחוֹלִים

13.
לְכוּ וְלָמְדוּ הַכָּתוּב כִּי חֶסֶד חָפַצְתִּי וְלֹא־זָבַח וְלֹא בָּאתִי לְהָשִׁיב הַצַּדִּיקִים כִּי אִם הָרְשָׁעִים

14.
אָז קָרְבוּ אֵלָיו תַּלְמִידֵי יוֹחָנָן וַיֹּאמֵרוּ לוֹ לָמָּה אָנוּ וְהַפְּרוּשִׁים מִתְעַנִּים הַרְבֵּה פְּעָמִים וְתַלְמִידֶיךָ אֵינָם מִתְעַנִּים

15.
וַיַּעַן לָהֶם יֵשׁוּעַ וַיֹּאמֶר לֹא יוּכְלוּ חַבֵירֵי הֶחָתָן לִבְכּוּת וּלְהִתְעַנּוֹת בִּהְיוֹתוֹ עִמָּהֶם יָבוֹאוּ יָמִים וְיִּלָּקַח מֵהֶם הֶחָתָן וְיָצוּמוּ

16.
לֹא יְאַבֵּד אִישׁ חֲתִיכַת מַלְבּוּשׁ חָדָשׁ בְּמַלְבּוּשׁ יָשָׁן שֶׁחוֹזֶק הַחֲתִיכָה הַחֲדָשָׁה יִמְשׁוּךְ מֵהַמַּלְבּוּשׁ הַבְּלוּיָה וְיִקְרַע יוֹתֵר

17.
וְלֹא יָשִׂימוּ יַיִן חָדָשׁ בַּכֵּלִים יְשָׁנִים פָּן יִשְׁבְּרוּ הַכֵּלִים וְיִשְׁפּוֹךְ הַיַּיִן וְהַכֵּלִים יְאַבְּדוּ רַק יַיִן חָדָשׁ בַּכְּלִי חָדָשׁ וּשְׁנֵיהֶם יִשְׁמְרוּ

18.
וַיְהִי בִּדְבָרוּ אֲלֵיהֶם וְיִקְרַב שַׂר אֶחָד וְיִשְׁתַּחֲוֶה לוֹ לֵאמֹר אֲדוֹנִי בִּתִּי אִתָּהּ עַתָּה מֵתָה בָּא נָא וְשָׂיַם יָדְךָ עָלֶיהָ וְהַחַיָּה

19.
וְיִקֹּם יֵשׁוּעַ וְיֵלֵךְ הוּא וְתַלְמִידָיו עִמּוֹ

20.
וְהִנֵּה אִשָּׁה אַחַת שׁוֹפַעַת דָּם שְׁתַּיִם עֲשָׂרָה שָׁנָה בָּא אַחֲרָיו וְתִגַּע בַּצִּיצִית בִּגְדּוֹ

21.
אוֹמֶרֶת בִּלְבָבָהּ אִם אֶגַּע בִּלְבוּשׁוֹ לְבַד אֵרָפֵא מִיָּד

22.
וְיָסֵב פָּנָיו וַיֹּאמֶר אֵלֶיהָ הִתְחַזְּקִי בִּתִּי בְּשֵׁם יהוה שֶׁאֱמוּנַתְךָ רְפְּאֲךָ וּבְּאוֹתָהּ שָׁעָה נִרְפֵּאת

23.
וַיְהִי בְּבוֹאוֹ בְּבֵית הַשַּׂר וְיָרֵא אֲנָשִׁים רַבִּים בּוֹכִים

24.
וַיֹּאמֶר לָהֶם צְאוּ כּוּלְּכֶם חוֹצֶה וְאַל תִּבְכּוּ שֶׁהַנַּעֲרָה יְשָׁנָה וְלֹא מֵתָה וַיְהִי כִּמִצְחָק בְּעֵינֵיהֶם וְאוֹמְרִים הַלֹּא אָנוּ רוֹאִים שֶׁהִיא מֵתָה

25.
וּבְהוֹצִיאָם אוֹתָם הַחוּצָה בָּא אֵלֶיהָ יֵשׁוּעַ וְיָגַע בְּיָדָהּ וְתִקֹּם הַנַּעֲרָה

26.
וַתֵּצֵא שְׁמוּעָה זֹאת בְּכָל הָאָרֶץ הַהִיא

27.
וְיַעֲבוֹר מִשָּׁם יֵשׁוּעַ וְהִנֵּה שְׁנֵי עִוְּרִים רָצִים אַחֲרָיו וְצוֹעֲקִים אֵלָיו חָנִינוּ בֶּן־דָּוִד

28.
וַיְהִי בְּבוֹאוֹ הַבַּיִת וְיִקְרְבוּ אֵלָיו הָעִוְּרִים וַיֹּאמֶר

29.
אֱמוּנַתְכֶם תְּרַפֵּא אֶתְכֶם

30.
וְתִפְקַחְנָה עֵינֵי שְׁנֵיהֶם מִיָּד וְיָרְאוּ וְיָצוּם לֵאמֹר הִשָּׁמְרוּ פֶּן יוֹדֵעַ הַדָּבָר

31.
וְהֵם יָצְאוּ וַיִּגַּלוּהוּ בְּכָל הָאָרֶץ הַהִיא

32.
וַיֵצֵא מִשָּׁם יֵשׁוּעַ וַיָּבִיאוּ לְפָנָיו אִישׁ אִלֵּם וְהַשֵּׁד בְּתוֹכוֹ

33.
וְיוֹצִיא אֶת־הַשֵּׁד וִידַבֵּר הָאֵלֶם וְיִפְלֵאוּ הַחֲבוּרוֹת וַיֹּאמֵרוּ לֹא נִרְאָה כָּזֶה בְּיִשְׁרָאֵל

34.
וַיֹּאמֵרוּ הַפְּרוּשִׁים בֶּאֱמֶת בַּשֵּׁם הַשֵּׁדִים מוֹצִיא הַשֵּׁדִים

35.
וְיָסֵב אֵת כָּל הֶעָרִים וְהַמִּגְדָּלִים מְלַמֵּד בְּבָתֵּי כְּנֵסִיּוֹת וּמְבַשֵּׂר בְּשׂוֹרוֹת וּמַרְפֵּא כָּל חֹלִים וְכָל מַדְוֶה

36.
וְיָרֵא יֵשׁוּעַ הַחֲבוּרוֹת וְיַחְמוֹל עֲלֵיהֶם שֶׁהָיוּ יְגֵעִים וְשׁוֹכְבִים כַּצֹאן אֲשֶׁר אֵין לָהֶם רוֹעֶה

37.
אָז אָמַר לְתַלְמִידָיו הֲקָמָה מְרוּבֶּה וְהַקּוֹצְרִים מְעַטִּים

38.
חָלוּ־נָא פְּנֵי־בַּעַל הֲקָמָה וְיִשְׁלַח הַקּוֹצְרִים רַבִּים לְקָצוּר קוֹמָתוֹ
