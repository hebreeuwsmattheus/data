1.
בַּיָמִים הָהֵם בָּא יוֹחָנָן הַמַּטְבִּיל דּוֹרֵשׁ בַּמִדְבָּר יְהוּדָה

2.
וְאָמַר חָזַרוּ בַּתְּשׁוּבָה בְּחַיִּים כִּי שֶׁמַלְכוּת הַשָׁמַיִם קָרוֹבָה לָבֹא

3.
לְקַיֵּים מָה שֶׁנֶאֱמַר עַל פִּי יְשַׁעְיָהוּ הַנָּבִיא קוֹל קוֹרַא בַּמִּדְבָּר פָּנּוּ דֶרֶךְ יַהְוֶה יַשְׁרוּ בַּעֲרָבָה מְסיִלָּה לֵאלֹהֵינוּ

4.
וְהִנֵּה יוֹחָנָן הָיָה לָבוּשׁ מִצֶמֶר הַגְּמַלִיָם וְאָזוּר וְעוֹר שָׁחוֹר אֵזוֹר בַּמָתְנָיו וְמָזוֹנוֹ הַאַרְבֶּה וּדְבַשׁ וְיְעָרִים

5.
אָז יֵצְאוּ אֵלָיו מֵיְרוּשָׁלַם וְמֵכֹּל יְהוּדָה וְמִכֹּל הַמַלְכוּת סְבִיבוֹת הַיַרְדֵּן

6.
וְאָז מִתְוַדִּים חֵטְאתֶּם וְטוֹבֵלִים בַּיַּרְדֵּן עַל מִאִמְרוֹ

7.
וַיַּרְא כִּי רַבִּיּם מִהַפְּרוּשִׁים בַּלֹעֵז פָארִיזֶיאִי וְמִן צְדוּקִים בָּאוּ לְטְבִּילתוֹ וַיֹּאמֶר אֲלֵהֶם  שֹׁרֶשׁ פְּתָנִים מִי לָמַד לָהֶם לְבָּרוֹחַ מִן הַקֶצֶף לְעָתִיד לְבָּא מִהַאֵל

8.
עֲשׂוֹ פְּרִי תְּשׁוּבָה הָשְׁלְמָה

9.
וְאַל-תֹּאמְרוּ בַּלֵבַבכֶם אָבִינוּ אַבְרָהָם אָמֵן אֲנִי אוֹמֶר לָכֶם שֶׁיוּכַל הָאֱלֹהִים לְהָקִים עַתָּ בְּנוֹ אַבְרָהָם מִן הַאֲבָנִים הָאִלָּה

10.
וּכְבִר הִגִּיעַ הַגַּרְזֶן לְשֶׁרֶשׁ הָעֵץ אֲשֶׁר לֹא יַעֲשֶׂה פְּרִי טוֹב יִכָּרֵת וּבָּאֵשׁ יִשְׂרֹף 

וְיִשְׁאֲלוּ לוֹ הַחַבּוּרוֹת אִם-כֵּן מָה נַעֲשָׂה וַיַּעַן לָהֶם יוֹחָנָן מִי שִׂישׂ לוֹ שְׁתִי כָּתְנוֹת יִתְן הֵא לְמִי שֶׁאֵין לוֹ וְיִבָּאוּ הָעָם להַטְבִּיל 
וַיִּשְׁאָלֵוהוּ רַבִּים מָה נַעֲשָׂה וַיַּעַן לָהֶם יוֹחָנָן תִצְטַעֵרוּ לְשׂוּם אִישׁ וְלֹא תַּעֱנָשׁוּם וְתִּשְׂמְחוּ בַּחֶלְקֵיכֶם וְכָּל הָעָם הֱיו חוֹשְׁבִים וְמֵדָּמִים בַּלִבָּם נִמּוֹל אִם בַּאוּלַי יוֹחָנָן הוּא יְשׁוּעָה

11.
וְיוֹחָנָן עָנָה לְכּוּלָּם בַּאֱמֶת הִנְנִי מִטְבִּיל אֶתְכֶם בַּימַי תְּשׁוּבָהָ וְאַחֵר יְבִּא אַחֲרָי חָזָק מָמֶּנִי שֶׁאֲינִי רָאוּי לְהַתִיר שְׂרוֹךְ נַעֲלָוֹ הוּא יְטַבֵּיל אֶתֲכֶם בַּאֵשׁ רוּחַ הַקָדוֹשׁ

12.
אֲשֶׁר בַּיָדוֹ מִזְרֵה לִזְרוֹת אֶת-גָרְנוֹ וְאַסָף הַדָּגָן לְאוֹצָרוֹ וְהַתֶּבֶן יִשְׂרוֹף בָאֵשׁ אֲשֶׁר לֹא תִּכְבֶּה עַד-כִּי

13.
אָז בָּא יֵשׁוּעַ מִהַגָּלִיל אֶל אֵת הַיַרְדֵּן לְהַטְבִּיל מִיוֹחָנָן

14.
וְיוֹחָנָן הָיָה מִסֶפֶק לְהַטְבִּילוֹ וַיֹאמֶר אֲנִי רָאוּי לְהַטְבִּיל מִיָדֶךָ וְאַתָּה בָּא אֵלָי

15.
וַיַּעַן יֵשׁוּעַ וַיֹאמֶר לוֹ הַנַּח שְׁכֵּן אָנוּ חִיִּיבִים לְהַשְׁלִים כָּל צְדָקָהָ וְאָז הַטְבִּילוּהוֹ יוֹחָנָן

16.
וְמִיַּד שֶׁעָלָה מִן הַמַּיִם נִפִתְּחו לוֹ הַשָׁמַיִם וִיַּרְא רוּחַ אֱלֹהִים יוֹרֶדֶת כִיְונּה וּשָׁרַתה עָלָיו

17.
וְהִנֵּה קוֹל מִן הַשָׁמַיִם אוֹמֵר זֶה בְּנִי אֲהוֹבִי מְאֹד נֶאֱהָב וְחֶפְצִי בּוֹ
