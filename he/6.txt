1.
הִשָּׁמֵרוּ פֶּן תַּעֲשׂוּ צִדְקַתכֶם לִפְּנִי הַאָדָם לְהָלֵל אֵתְכֶם וְאִם תַּעֲשׂוּ לֹא יִהְיֶה לָכֶם שָׂכָר מֵאֵת אֲבִיכֶם שֶׁבַּשָׁמַיִם

2.
עוֹד אָמַר לָהֶם יֵשׁוּעַ כַּאֲשֶׁר תַּעֲשׂוּ צְדָקָה לֹא תִּרְצוּ לְהַעֲבִיר כָּרוֹז וְחֲצוֹצְרוֹת לְפְּנִיכֶם כְּמוֹ הַחֲנֵפִים בַּלֹעֵז אִיפּוֹקְרִיטֶיש שֶׁעוֹשִׂים צִדְקָתָם בְּרְחוֹבוֹת וְבַּשׁוָוקִים בַּעַד שֶׁיְכַבְּדוּ אוֹתָם בְּנֵי אָדָם 
אָמֵן אֲנִי אוֹמֶר לָכֶם שֶׁכְּבַר קִבְּלוּ שְׂכָרֶם

3.
וְאַתֶּם כַּאֲשֶׁר תַּעֲשֵׂה צְדָקָה אַל יָדַע שְׂמֹאלֶךָ מָה יַעֲשֶׂה יְמִינֶךָ

4.
לְהֱיוֹת מַתְּנָתָךְ בַּסֵּתֶר וְאָבִיךָ הַרוֹאֶה הַנִסְתָּרוֹת יְשַׁלֶּם לְךָ

5.
בַּעִתּ הַהיִא אָמַר יֵשׁוּעַ לְתַּלְמִידָיו בַּשָׁעַה שֶׁתִּתְפַּלֵּלוּ אַל תָּרִימוּ קוֹל וְאַל תִּהְיוּ כְּחֲנֵפִים הַעֲצָבִּים הַאוֹהָבִים לְהִתְפַּלֵּל בַּבָּתֵּי כְּנֵסֵיוֹת וְבַּמִקְצוֹע חֲצֵרוֹת וְמִתְפַּלֵּלִים בַּגְּבוֹהוֹת שֶׁיֵרָאוּ אוֹתָם וְיְשַׁבְּחוּ בְּנֵי אָדָם 
אָמֵן אֲנִי אוֹמֶר לָכֶם שֶׁכְּבַר קִבְּלוּ שְׂכָרֶם

6.
וְאַתָּה בַּהִתְפַּלֵּלךָ בָּא לְמִשְׁכָּבךָ וְסְגוֹר דַּלְתֵיךָ בַּעַדךָ וְהִתְפַּלֵּל לְאָבִיךָ בַּסֵּתֶר וְאָבִיךָ הַרוֹאֶה בַּסֵּתֶר יְשַׁלֶּם לְךָ

7.
וְאַתֶּם כַּאֲשֶׁר תִתְפַּלֵּלוּ אַל תִּרְבּוּ דְּבָרִים כְּמוֹ שֶׁהַמִינִים חוֹשְׁבִים שֶׁבְּרוֹב דְּבָרִים יִשְׁמְעוּם

8.
וְאַתֶּם אַל תִּדָּמוּ לָהֶם שֶׁאֲבִיכֶם שֶׁבַּשָׁמַיִם יוֹדֵעַ דְּבָרִיכֶם קוֹדְּם שֶׁתִּשִׁאֵלוּ מִמֶּנּוּ

9.
וְכֵן תִתְפַּלֵּלוּ
אָבִינוּ שֶׁבַּשָׁמַיִם יִתְקַדֵּשׁ שִׁמְךָ

10.
וְיִתְבָּרֵךְ מַלְכוּתְךָ רְצוֹנְךָ יִהְיֶה עָשׂוּי בַּשָׁמַיִם וּבָּאָרֶץ

11.
וַתִתֵּן לַחְמֵנוּ תְּמִידִית תֶּן-הַיּוֹם לָנוּ

12.
וּמְחוֹל לָנוּ חַטֹּאתֵינוּ כַּאֲשֶׁר אֲנַחְנוּ מוֹחֲלִים לַחוֹטִאים לָנוּ

13.
וְאַל תְּבִיאֶנוּ לִידֵי נִסָּיוֹן וְשָׁמְרֵינוּ מִכָּל רָע אָמֵן

14.
אִם-תִּמְחֲלוּ לְבְּנֵי אָדָם עֲוֹנוֹתֵיהֶם יְמְחוֹל אֲבִיכֶם שֶׁבַּשָׁמַיִם עֲוֹנוֹתֵיכֶם

15.
וְאִם-לֹא תִּמְחֲלוּ לָהֶם לֹא יְמְחוֹל לָכֶם עֲוֹנוֹתֵיכֶם לָכֶם

16.
עוֹד אָמַר לָהֶם וְכַּאֲשֶׁר תָּצוּמוּ אַל תִּהְיוּ כְּחֲנֵפִים הַעֲצָבִּים שֶׁמִרֹאִים עַצְמָם עֲצָבִּים וְמִשְׁנִים פְּנֵיהֶם לְהֵרָאוֹת צוֹמוּתֶם לְבְּנֵי אָדָם 
אָמֵן אֲנִי אוֹמֶר לָכֶם שֶׁכְּבַר קִבְּלוּ שְׂכָרֶם

17.
וְאַתֶּם בַּצוֹמְכֶם רַחֲצוּ רֹאשֵׁיכֶם

18.
שֶׁלֹא תִּרְאוּ מִתְּעַנִים לְבְּנֵי אָדָם אֶלָּא לְאָבִיךָ שֶׁהוּא בַּסֵּתֶר וְאָבִיךָ שֶׁהוּא בַּסֵּתֶר יְשַׁלֶּם לְךָ

19.
עוֹד אָמַר לָהֶם אַל תַּרְבּוּ לְצְבוֹר אוֹצָרוֹת בָּאָרֶץ כְּדֵי שֶׁיִאָכַלְנוּ רָקַב וְתּוֹלֵעַה אוֹ יַחְפְּרוּ הַגַּנָּבִים וְיִגָּנְבוּם

20.
עֲשׂוּ לָכֶם אוֹצָרוֹת בַּשָׁמַיִם בַּמָקוֹם שֶׁרִימָּה וְתּוֹלֵעַה לֹא יֹאכְלֵם וְבַּמָקוֹם שֶׁהַגַּנָּבִים לֹא יַחְפְּרוּ וְיִגָּנְבוּ

21.
בַּאוֹתוֹ מָקוֹם שֶׁיִהְיֶה אוֹצָרךָ שָׁם יִהְיֶה לִבֶּךָ

22.
וְנִיר גּוּפְךָ עֵינֶיךָ אִם עֵינֶיךָ לְנוֹכַח יַבִּיטוּ כָּל גּוּפְךָ יַזהִיר

23.
וְאִם עֵינֶיךָ כֵּהוֹ כָּל-גּוּפְך יַחֲשִׁיךָ וְאִם הַאוֹר שֶׁבְךָ יַחֲשִׁיךְ מַחְשָׁכִים יהָיוּ כָּל דְּרָכֶיךָ 

24.
בַּעִתּ הַהיִא אָמַר יֵשׁוּעַ לְתַּלְמִידָיו לֹא יוּכַל אִישׁ לְעֲבוֹד לְשְׁנֵי אֲדוֹנִים כִּי אִם הָאֶחָד יִשְׂנָא וְהָאַחֵר יִאָהַב אוֹ לְאֶחָד יְכַבֵּד וְלְאַחֵר יִבָּזַה לֹא תּוּכְלוּ לְעֲבוֹד הָאֵל וְהַעוֹלָם

25.
לָכֵן אֲנִי אוֹמֶר לָכֶם שֶׁלֹא תְּדָאַגוּ לְמַאֲכָל לְנַפְשׁוֹתֵיכֶם וְלֹא בַּמַלְבּוּשׁ לְגּוּפָכֶם שֶׁהַנֶפֶשׁ יְקָרָה מִהַמָזוֹן וְהַגּוּף מִהַמַלְבּוּשׁ

26.
הִסְתָכֵּלוּ בְעוֹף הַשָׁמַיִם אֲשֶׁר לֹא יִזְרוֹעוֹ וְלֹא יִקְצוֹרוֹ וְלֹא יַאַסְפוּ לָאוֹצָרוֹת וְאֲבִיכֶם הַעֶלְיוֹן מְכַלְכֵּל אֶתְכֶם הֲלֹא אַתֶּם יְקָרִים מִהֵם

27.
מִי בַּכֶם מִהַדּוֹאֲגִים שֶׁיוּכַל לְהוֹסִיף בַּקוֹמָתוֹ אַמָּה אַחַת

28.
אִם-כֵּן אַל מָה תִדְאֲגוּ בַּלְבוּשׁ רָאוּ חֲבַצֶּלֶת הַשָׁרוֹן לֹעֵז גְיִליְוֹן בַּהַגְּדֹלָה אַיִנָה טָווּה וְאוֹרְגֹת

29.
וַאֲנִי אוֹמַר לָכֶם שֶׁהַמֶלֶךְ שְׁלֹמֹה בַּכֹּל כָּבוֹדוֹ לֹא הָיַה מִלְבוּשׁ כָּמוֹהָ

30.
וְאִם תֶּבֶן הַנִשְׁאַר בַּקמוֹת לֹעֵז פֶינֹן אֲשֶׁר הַיּוֹם לֵחֹה וְמָחָר יָבְשָׁה וְמִשִׂימִים אוֹתָהּ בַּתַּנּוּר הַאֵל מִלְבִּישׁ אוֹתָהּ כֹּל שְׁכֵּן אַתֶּם קַטַנֵּי אֱמֻנָה

31.
וְאִם-כֵּן שַׁהַאֵל יַחְשׁוֹב מִכֶּם אַל תִדְאֲגוּ לוֹמַר מָה נֶאֱכַל וְמָה נִשְׁתֶּה

32.
שַׁכָּל אֵלֶּה הַגוּפִים מְבַקְּשִׁים וַיּוֹדַע אֲבִיכֶם שַׁכָּל אֵלֶּה אַתֶּם צָרִיכִּים

33.
בִּקְשׁוּ קוֹדֵם מַלְכוּת אֱלֹהִים וְצִדְקָתוֹ וְכָּל אֵלֶּה הַדְּבָרִים יִנָּתְנוּ לָכֶם

34.
לָכֵן אַל תִדְאֲגוּ לְיוֹם מָחָר שֶׁיוֹם מָחָר יִדְאָגוּ מִמֶּנּוּ דַּי לוֹ לְיוֹם בַּצַרתוֹ
